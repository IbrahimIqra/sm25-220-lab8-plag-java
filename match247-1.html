public class TaskDistributorBasic {
    public static int[] distributeTasks(int[] tasks, int m) {
        // Step 1: Initialize machine loads
        int[] machineLoads = new int[m];

      
        for (int i = 0; i < tasks.length; i++) {
            int minIndex = 0;

         
            for (int j = 1; j < m; j++) {
                if (machineLoads[j] < machineLoads[minIndex]) {
                    minIndex = j;
                }
            }

            machineLoads[minIndex] += tasks[i];
        }

        
        return machineLoads;
    }

    


    public static void main(String[] args) {
        int[] tasks = {2, 4, 7, 1, 6};
        int m = 4;

        int[] result = distributeTasks(tasks, m);


        System.out.print("Output: [");
        for (int i = 0; i < result.length; i++) {
            System.out.print(result[i]);
            if (i < result.length - 1) System.out.print(", ");
        }
        System.out.println("]");
    }
}

public class TopKLargestElements {

   
    public static void heapify(int[] arr, int n, int i) {
        int largest = i;       


        int left = 2 * i + 1;        
        int right = 2 * i + 2; 
       
        if (left < n && arr[left] > arr[largest]) {
            largest = left;
        }

        
        if (right < n && arr[right] > arr[largest]) {
            largest = right;
        }

        
        if (largest != i) {
           
            int temp = arr[i];
            arr[i] = arr[largest];
            arr[largest] = temp;
            heapify(arr, n, largest);
        }
    }

   
    public static void buildMaxHeap(int[] arr, int n) {
        // Index of last non-leaf node
        for (int i = n / 2 - 1; i >= 0; i--) {
            heapify(arr, n, i);
        }
    }

  


    public static int[] getTopKLargest(int[] arr, int k) {
        int n = arr.length;

        buildMaxHeap(arr, n);

        int[] result = new int[k];

        for (int i = 0; i < k; i++) {
            
            result[i] = arr[0];

          
            arr[0] = arr[n - 1];
            n--;

          
            heapify(arr, n, 0);
        }

        return result;
    }

 


    public static void main(String[] args) {
        int[] nums = {4, 10, 2, 8, 6, 7};
        int k = 3;

        int[] topK = getTopKLargest(nums, k);

        System.out.print(k)
        for (int num : topK) {
        System.out.print(num + " ");
        }
    }
}
