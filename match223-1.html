public void insert(int value){
      if(heap.length == size){
        System.out.println("Heap Full");
       return ;
      }
      
      heap[size] = value;
      swim(size);
      size++;
    }
    
    public int extractMin(){
      if(size <= 0){
        System.out.println("Heap is empty.");


        return -1;
      }
      
      int min = heap[0];
      heap[0] = heap[size-1];
      size--;
      sink(0);      
      return min;
    }
    
    private void swap(int a, int b){
      int temp = heap[a];
      heap[a] = heap[b];
      heap[b] = temp;
    }
    
    private void swim(int i){
      while(i>0 && heap[i] < heap[(i-1)/2]){
        swap(i, (i-1)/2);
        i = (i-1)/2;
      }
    }
    
    private void sink(int i){
      int minimum = i;
      int left = (2*i+1); 
      int right = (2*i+2);


      if(left < size && heap[minimum] > heap[left]){
        minimum = left;
      }
      
      if(right < size && heap[minimum] > heap[right]){
        minimum = right;
      }
      
      if(minimum != i){
        swap(i, minimum);
        sink(minimum);
      }
    }
    
    public void sort(){
      int s = size;
      for(int i = size - 1; i> 0; i--){    
        swap(0, i);
        size--;
        sink(0);
      }
      
      size = s;
    }
    
    public Integer [] getheap(){
      return heap;
    }
    
    public int getsize(){


      return size;
    }
}
