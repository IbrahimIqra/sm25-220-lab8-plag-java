public class taskkk03{
  public static  class MinHeap{
    int [] heap;
    int s;
    
    MinHeap(int c){
      heap=new int [c];
      s=0;
    }
    void insert(int i){
      heap[s]=i;
      int curr=s;
      s++;
      while(curr>0){
        int p=(curr-1)/2;
        if(heap[curr]<heap[p]){
          swap(curr,p);
          curr=p;
        }
        else{
          break;
        }
      }
    } 
    
     void swap(int a,int b){
      int t=heap[a];
      heap[a]=heap[b];
      heap[b]=t;
    }
    int extractMin(){


      if(s==0){
        return -1;
      }
      int min=heap[0];
      heap[0]=heap[s-1];
      s--;
      int curr=0;
      while(true){
        int lc=curr*2+1;
        int rc=curr*2+2;
        int small=curr;


        if(lc<s && heap[lc]<heap[small]){
          small=lc;
        }
        if(rc<s && heap[rc]<heap[small]){
          small=rc;
        }
        if(small!=curr){
          swap(curr,small);
          curr=small;

        }
        else{
          break;
        }
      }
    
    return min;
      }

     
    }



    public static void main(String [] args){
      int [] tasks={2,4,7,1,6};
      int m=4;
      MinHeap heap=new MinHeap(tasks.length+m);
      for(int i=0;i<m;i++){
        heap.insert(0);
      }

      for(int i=0;i<tasks.length;i++){
        int ml=heap.extractMin();
        ml+=tasks[i];
        heap.insert(ml);
      }
      int [] lo=new int [heap.s];
      for(int i=0;i<lo.length;i++){
            lo[i]=heap.extractMin();
      }
      System.out.print("[");
      for(int i=0;i<lo.length;i++){


        System.out.print(lo[i]);
        if(i<lo.length-1){
          System.out.print(",");
        }
      }
      System.out.print("]");
    
    }
}

public class taskkk04{
  public static  class MaxHeap{
    int [] heap;
    int s;
    
    MaxHeap(int c){
      heap=new int [c];
      s=0;
    }
    void insert(int i){
      heap[s]=i;
      int curr=s;
      s++;
      while(curr>0){
        int p=(curr-1)/2;
        if(heap[curr]>heap[p]){
          swap(curr,p);
          curr=p;
        }
        else{
          break;
        }
      }
    } 
    
     void swap(int a,int b){
      int t=heap[a];
      heap[a]=heap[b];
      heap[b]=t;
    }
    int extractMax(){


      if(s==0){
        return -1;
      }
      int max=heap[0];
      heap[0]=heap[s-1];
      s--;
      int curr=0;
      while(true){
        int lc=curr*2+1;
        int rc=curr*2+2;
        int large=curr;


        if(lc<s && heap[lc]>heap[large]){
          large=lc;
        }
        if(rc<s && heap[rc]>heap[large]){
          large=rc;
        }
        if(large!=curr){
          swap(curr,large);
          curr=large;

        }
        else{
          break;
        }
      }
    


    return max;
      }

     
    }
  
  public static int[] topk(int[]nums,int k){
     MaxHeap heap=new MaxHeap(nums.length);
      for(int i=0;i<nums.length;i++){


        heap.insert(nums[i]);
      }
      int[] result=new int[k];
       for(int i=0;i<k;i++){


         result[i]=heap.extractMax();
       }
       return result;
  }
       


    public static void main(String [] args){

    }
