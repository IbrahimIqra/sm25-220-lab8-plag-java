public class taskSchedule {
    public static int[] scheduleTasks(int[] tasks, int m) {
        
        int[] heap = new int[m];
        for (int i = 0; i < m; i++) {
            heap[i] = 0;
        }


        
        for (int i = 0; i < tasks.length; i++) {
            int task = tasks[i];   


           int minLoad = heap[0];
            heap[0] = minLoad + task;


            
            heapifyDown(heap, 0, m);
        }


        return heap;
    }


    


    private static void heapifyDown(int[] heap, int i, int size) {
        int smallest = i;
        int left = 2 * i + 1;
        int right = 2 * i + 2;


        
        if (left < size && heap[left] < heap[smallest]) {
            smallest = left;
        }


        
        if (right < size && heap[right] < heap[smallest]) {
            smallest = right;
        }


        
        if (smallest != i) {
            int temp = heap[i];
            heap[i] = heap[smallest];
            heap[smallest] = temp;
            heapifyDown(heap, smallest, size);
        }
    }


    


    public static void main(String[] args) {
        int[] tasks = {2, 4, 7, 1, 6};
        int m = 4;


        int[] result = scheduleTasks(tasks, m);


        
        System.out.print("[");
        for (int i = 0; i < result.length; i++) {
            System.out.print(result[i]);
            if (i < result.length - 1) System.out.print(", ");
        }
        System.out.println("]");
    }
}

public class topkelements {
    public static int[] topkFinder(int[] nums, int k) {
        int n = nums.length;


        
        maxHeap(nums, n);


        


        int[] result = new int[k];
        for (int i = 0; i < k; i++) {
           
            result[i] = nums[0];


            nums[0] = nums[n - 1 - i];
            heapifyDown(nums, 0, n - 1 - i);
        }


        return result;
    }


    
    private static void maxHeap(int[] arr, int n) {
        
        for (int i = n / 2 - 1; i >= 0; i--) {
            heapifyDown(arr, i, n);
        }
    }


    


    private static void heapifyDown(int[] arr, int i, int size) {
        int largest = i;
        int left = 2 * i + 1;
        int right = 2 * i + 2;


        
        if (left < size && arr[left] > arr[largest]) {
            largest = left;
        }


        
        if (right < size && arr[right] > arr[largest]) {
            largest = right;
        }


        
        if (largest != i) {
            int temp = arr[i];
            arr[i] = arr[largest];
            arr[largest] = temp;
            heapifyDown(arr, largest, size);
        }
    }
