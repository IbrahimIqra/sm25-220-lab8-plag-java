public class Task3 {

    private int size;

    private int[] heap;

    private int capacity;

    public Task3(int capacity) {

        this.capacity = capacity;

        heap = new int[capacity];

        size = 0;

    }

    public void insert(int val) {



        if (size >= capacity) {

            System.out.println("Heap is full");

            return;

        }

        heap[size] = val;

        swim(size);

        size++;

    }

    public void swim(int index) {

        int parent  = (index - 1) / 2;

        while (index > 0 && heap[index] < heap[parent]) {

            int temp = heap[index];

            heap[index] = heap[parent];

            heap[parent] = temp;

            index = parent;

            parent = (index - 1) / 2;

        }

    }

    public int min(){

        if (size <= 0){

            System.out.println("Heap is empty");

            return -1;

        }

        int temp = heap[0];

        heap[0] = heap[size-1];

        heap[size-1] = 0;

        size--;

        heapify(0);

        return temp;

    }

    public void heapify(int index){

        int left  = 2 * index + 1;

        int right = 2 * index + 2;

        if (left >= size){

            return;

        }



        int min = left;

        if (right < size && heap[right] < heap[min]){

            min = right;

        }

        if (heap[index] > heap[min]){

            int temp = heap[index];

            heap[index] = heap[min];

            heap[min] = temp;

            heapify(min);

        }

    }

    public static void main(String [] args) {

        int[] arr = {2, 4, 7, 1, 6};

        int m = 4;

        Task3 t = new Task3(m * arr.length);

        for (int i = 0; i < m; i++) {

            t.insert(arr[i]);

        }

        for (int i = m; i < arr.length; i++) {

            int amount = t.min();

            amount += arr[i];



            t.insert(amount);

        }

        int [] vals = new int [m];

        for (int i = 0; i < m; i++) {

            vals[i] = t.min();

        }

        for (int i = 0; i < vals.length; i++) {

            System.out.print(vals[i] + " ");

        }

    }

}
