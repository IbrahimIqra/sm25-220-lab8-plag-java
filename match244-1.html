        }

            if (size < heap.length) {
                heap[size] = val;
                swim(size);
                size++;
            } 
                

               else {
                System.out.println("Heap full ");
            }
        }

        public Integer delete() {
            if (size == 0) {
                System.out.println("Heap empty ");
                return null;
            }
            int item = heap[0];
            swap(0, size - 1);
            size--;
            sink(0);
            return item;
        }

        private void swim(int child) {
            int parent = (child - 1) / 2;


            while (child > 0 && heap[parent] > heap[child]) {
                swap(parent, child);
                child = parent;
                parent = (child - 1) / 2;
            }
        }

        private void sink(int parent) {
            int left = 2 * parent + 1;
            int right = left + 1;
            while ((left < size && right < size) || left < size) {
                if (left < size && right < size) {
                    if (heap[left] < heap[parent] && heap[right] < heap[parent]) {

                        if (heap[left] < heap[right]) {
                            swap(parent, left);
                            parent = left;

                            parent = right;

                    }
                     
                        else if (heap[left] < heap[parent]) {

        private void swap(int i, int j) {

            heap[i] = heap[j];
            heap[j] = t;

import java.util.Arrays;
public class task4{
 static class MaxHeap {
    private int[] heap;
    private int size;

    public MaxHeap(int capacity) {
        heap = new int[capacity + 1];
        size = 0;
    }

    public void insert(int val) {
        size++;
        heap[size] = val;
        swim(size);
    }

    private void swim(int child) {
        int parent = child / 2;
        while (parent > 0 && heap[parent] < heap[child]) {
            swap(parent, child);
            child = parent;
            parent = child / 2;
        }
    }

    public int delete() {
        int item = heap[1];
        swap(1, size);
        size--;
        sink(1);
        return item;
    }

    private void sink(int parent) {
        int left = 2 * parent;
        int right = left + 1;
        while ((left <= size && right <= size) || left <= size) {
            if (left <= size && right <= size) {
               
                   if (heap[left] > heap[parent] && heap[right] > heap[parent]) {

                    if (heap[left] > heap[right]) {
                        swap(parent, left);
                        parent = left;


                    } else {
                        swap(parent, right);
                        parent = right;
                    }


                } else if (heap[left] > heap[parent]) {
                    swap(left, parent);
                    parent = left;


                } else if (heap[right] > heap[parent]) {
                    swap(right, parent);
                    parent = right;


                } else {
                    break;
                }


            } else if (heap[left] > heap[parent]) {
                swap(left, parent);
                parent = left;


            } else {
                break;
            }


            left = 2 * parent;
            right = left + 1;
        }
    }



    private void swap(int i, int j) {
        int tmp = heap[i];
        heap[i] = heap[j];
        heap[j] = tmp;
    }

 }


    public static int[] topK(int[] nums, int k) {
        MaxHeap heap = new MaxHeap(nums.length);

        for (int i = 0; i < nums.length; i++) {
            heap.insert(nums[i]);
        }

        int[] res = new int[k];

        for (int i = 0; i < k; i++) {
            res[i] = heap.delete();
        }

        return res;
    }
