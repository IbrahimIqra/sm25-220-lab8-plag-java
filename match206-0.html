import java.util.Scanner;
public class Task3{
    public static void main(String[]args){
        Scanner sc=new Scanner(System.in);
        
        System.out.println("Enter the amount of tasks");
        int totalTasks=sc.nextInt();
        int[]taskArr=new int[totalTasks];

        for(int i=1; i<=totalTasks; i++){
            System.out.println("Enter the processing time of task-"+i+": ");
            taskArr[i-1]=sc.nextInt();
        }
        
        System.out.println("Enter the number of machines");


        int machines=sc.nextInt();

        MinHeap heap= new MinHeap(machines);

        for(int i=0; i<machines; i++){
            heap.insert(0);
        }

        for(int i=0;i<totalTasks; i++){
            int load=heap.extractMin();
            int newLoad=load+taskArr[i];
            heap.insert(newLoad);
        }
        heap.sort();
        System.out.print("Final machine loads: ");
        heap.printHeap();

        sc.close();
    }
}

import java.util.Scanner;
public class Task4 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.println("Enter the number of elements:");
        int n = sc.nextInt();

        int[] nums = new int[n];
        System.out.println("Enter the elements:");
        for (int i=0;i<n; i++){
            nums[i]=sc.nextInt();
        }


        System.out.println("Enter the value of k:");
        int k = sc.nextInt();



        MaxHeap heap = new MaxHeap(n);
        for (int i=0;i<nums.length;i++) {
            heap.insert(nums[i]);
        }


        int[] result = new int[k];
        for (int i=0;i<k; i++){
            result[i]=heap.extractMax();
        }

        System.out.print("Top " + k + " largest elements: [");
        for (int i=0;i<k; i++){
            System.out.print(result[i]);
            if(i<k-1) System.out.print(", ");
        }
        System.out.println("]");
        
        sc.close();
    }
}




//MinHeap class
public class MinHeap {
    private int size;
    private int capacity;


    private int[] heap;

    public MinHeap(int capacity) {
        this.capacity = capacity;
        this.heap = new int[capacity + 1];
        this.size = 0;
    }

    public void insert(int value) {
        if (capacity == size) {
            System.out.println("Heap full");
            return;
        }
        size++;
        heap[size] = value;
        swim(size);
    }

    private void swim(int i) {
        while (i > 1 && heap[i] < heap[i / 2]) {
            swap(i, i / 2);
            i = i / 2;
        }
    }



    private void sink(int i) {
        while (2 * i <= size) {
            int left = 2 * i;
            int right = 2 * i + 1;
            int smallerChild;
            if (right <= size && heap[right] < heap[left]) {
                smallerChild = right;
            } else {
                smallerChild = left;
            }
            if (heap[i] <= heap[smallerChild]) {
                break;
            } else {
                swap(i, smallerChild);
                i = smallerChild;
            }
        }
    }



    public int extractMin() {
        if (size == 0) {
            System.out.println("Heap empty");
            return -1;
        }
        int min = heap[1];
        heap[1] = heap[size];
        size--;
        sink(1);
        return min;
    }

    public void sort() {
        int originalSize = size;
        int[] sorted = new int[size + 1];

        for (int i = 1; i <= originalSize; i++) {
            sorted[i] = extractMin();
        }

        for (int i = 1; i <= originalSize; i++) {
            heap[i] = sorted[i];
        }
        size = originalSize;
    }

    private void swap(int idx1, int idx2) {
        int temp = heap[idx1];
        heap[idx1] = heap[idx2];
        heap[idx2] = temp;
    }

    public void printHeap() {
    System.out.print("[");
    for (int i = 1; i <= size; i++) {
        System.out.print(heap[i]);
        if (i < size) System.out.print(", ");
    }
    System.out.println("]");
}

}

    public class MaxHeap{
            private int size;
            private int capacity;
            private int[]heap;

        public MaxHeap(int capacity){
            this.capacity=capacity;
            this.heap=new int[capacity+1];


            this.size=0;
            }
            
        public void insert(int value){
            if(capacity==size){
                System.out.println("Heap full");
                return;
                }
                size++;
                heap[size]=value;
                swim(size);
            }

        private void swim(int i){
                while(i>1 && heap[i]>heap[i/2]){
                    swap(i, i/2);
                    i=i/2;
                    }
                }



                private void sink(int i){
                    while(2*i<=size){
                        int left=2*i;
                        int right=2*i+1;
                        int biggerChild;
                        if(right<=size && heap[right]>heap[left]){
                            biggerChild=right;
                        }
                        else{
                            biggerChild=left;
                        }
                        if(heap[i]>=heap[biggerChild]){
                            break;
                        }
                        else{
                            swap(i, biggerChild);
                            i=biggerChild;
                        }
                    }
                }



                public int extractMax(){
                    if(size==0){
                        System.out.println("Heap empty");
                        return -1;
                }
                int max=heap[1];
                heap[1]=heap[size];
                size--;
                sink(1);
                return max;
                }

                public void sort() {
                int originalSize = size;
                int[] sorted = new int[size + 1];

                for (int i = 1; i <= originalSize; i++) {
                    sorted[i] = extractMax();
                }

                // copy sorted values back into heap[]
                for (int i = 1; i <= originalSize; i++) {
                    heap[i] = sorted[i];
                }
                size = originalSize; // restore size
            }

            private void  swap(int idx1, int idx2){
                    int temp=heap[idx1];
                    heap[idx1]=heap[idx2];
                    heap[idx2]=temp;
                }

                public void printHeap() {
        System.out.print("[");
        for (int i = 1; i <= size; i++) {
            System.out.print(heap[i]);
            if (i < size) System.out.print(", ");
        }
        System.out.println("]");
    }
