class MinHeap {
private int[] heap;
private int size;
public MinHeap(int capacity) {
heap = new int[capacity + 1];
size = 0;
}
public void insert(int value) {
heap[++size] = value;
swim(size);
}
private void swim(int k) {
while (k > 1 && heap[k / 2] > heap[k]) {
swap(k / 2, k);
k = k / 2;
}
}
public int extractMin() {
int min = heap[1];
swap(1, size--);
sink(1);
return min;
}
private void sink(int k) {
while (2 * k <= size) {
int j = 2 * k;
if (j < size && heap[j] > heap[j + 1]) j++;
if (heap[k] <= heap[j]) break;
swap(k, j);
k = j;
}
}
private void swap(int i, int j) {
int temp = heap[i];
heap[i] = heap[j];
heap[j] = temp;
}
public void sort() {
int originalSize = size;
for (int i = size; i > 1; i--) {
swap(1, i);
size--;
sink(1);
}
size = originalSize;
}
public int[] getArray() {
int[] arr = new int[size];
for (int i = 1; i <= size; i++) {
arr[i-1] = heap[i];
}
return arr;
}
}
}

public int[] assignTasks(int[] tasks, int m) {
MinHeap minHeap = new MinHeap(m);
for (int i = 0; i < m; i++) {
minHeap.insert(0);
}
for (int task : tasks) {
int currentLoad = minHeap.extractMin();
minHeap.insert(currentLoad + task);
}
return minHeap.getArray();
}
}

public int[] topKLargest(int[] nums, int k) {
MaxHeap maxHeap = new MaxHeap(nums.length);
for (int num : nums) {
maxHeap.insert(num);
}
int[] result = new int[k];
for (int i = 0; i < k; i++) {
result[i] = maxHeap.extractMax();
}
return result;
}
