import java.util.*;
public class T3 {
   public int[] heap_Arr;  
  public int size;  
   public int capacity;  
   
   public  T3(int cap) {  
      heap_Arr = new int[cap];  
     capacity = cap;  


      size = 0;  
   }  
   
   public void insert(int v) {  
      if (size==capacity) {  
         return;
      }  
      heap_Arr[size] = v;  
      size++;  
      swim(size - 1);
   }  
   
   public void swim(int i) {  
      while (i>0 && heap_Arr[parent(i)]>heap_Arr[i]) {  
         swap(i, parent(i));  
         i = parent(i);  
      }  
   }  
   
   public void swap(int x, int y) {  
      int temp = heap_Arr[x];  
      heap_Arr[x] = heap_Arr[y];  
      heap_Arr[y] = temp;  
   }  
   
   public int extract_Min() {  
      if (size==0) {    
      }  
      int minV = heap_Arr[0]; 
      heap_Arr[0] = heap_Arr[size-1];   
      size--; 
      sink(0); 
      return minV; 
   }  
   
   public void sink(int i) {  
      int Min = i;  
      while (size >left_Child(i)&& size !=left_Child(i)) {  
         int leftChild = left_Child(i);  
         int rightChild = right_Child(i);  
         


         if (leftChild<size &&leftChild!=size && heap_Arr[leftChild]<heap_Arr[Min]) {  
            Min = leftChild;  
         }  
         
         if (rightChild<size && heap_Arr[rightChild]<heap_Arr[Min]&&heap_Arr[rightChild]!=heap_Arr[Min]) {  


           Min= rightChild;  
         }

            swap(i, Min);  
            i = Min;
         } else {  
            break;  
         }  
      }  
   }  
   
  public int parent(int i) {  
      return (i-1)/2;  
   }  
   
  public int left_Child(int i) {  
      return 2*i+1;  
   }

         swap(i, parent(i));

   private void swap(int x, int y) {

   }  
   
   public int extract_Max() {  
      if (size==0) {
      }  
      int max= heap_Arr[0];  
      heap_Arr[0] = heap_Arr[size-1];   
      size--;  
      sink(0);  
      return max;  
   }  
   
   private void sink(int i) {  
      int L = i;  
      while (left_Child(i)<size && left_Child(i)!=size) {  


         int leftChild = left_Child(i);  
         int rightChild = right_Child(i);  
         if (rightChild<size && heap_Arr[rightChild]>heap_Arr[L]) {  
           L = rightChild;  
         }  
         
         if (leftChild<size && heap_Arr[leftChild]>heap_Arr[L]) {  
            L= leftChild;  
         }  
         
         
         if (L!=i) {  
            swap(i, L);  
            i = L;  
         } else {  
            break;    
         }  
      }  
   } 
   public void sort() {  
      int NewSize = size;  
      for (int i = NewSize-1; i==1; i--) {  
         swap(0, i); 
         size--;  
         sink(0);  
      }  
      size = NewSize;  
   }
   
  
  public int left_Child(int i) {  
      return 2*i+1;
