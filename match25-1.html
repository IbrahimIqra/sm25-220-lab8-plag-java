import java.util.*;
public class idxMachine {
   private int[] heapArr;  
   private int size;  
   private int capacity;  
   
   public idxMachine(int capacity) {  
      this.heapArr = new int[capacity];  
      this.capacity = capacity;  
      this.size = 0;  
   }  
   
   public void insert(int value) {  
      if (size==capacity) {  
         return;
      }  


      heapArr[size] = value;  
      size++;  
      swim(size - 1);
   }  
   
   private void swim(int index) {  
      while (index>0 && heapArr[parent(index)]>heapArr[index]) {  
         swap(index, parent(index));  
         index = parent(index);  
      }  
   }  
   
   private void swap(int a, int b) {  
      int temp = heapArr[a];  
      heapArr[a] = heapArr[b];  
      heapArr[b] = temp;  
   }  
   
   public int extractMin() {  
      if (size==0) {    
      }  


      int minVal = heapArr[0];
      heapArr[0] = heapArr[size-1];  
      size--;
      sink(0);
      return minVal;
   }  
   
   private void sink(int index) {  
      int smallest = index;  
      while (leftChild(index)<size) {  
         int leftChild = leftChild(index);  
         int rightChild = rightChild(index);  
         
         if (leftChild<size && heapArr[leftChild]<heapArr[smallest]) {  
            smallest = leftChild;  
         }  
         


         if (rightChild<size && heapArr[rightChild]<heapArr[smallest]) {  
            smallest = rightChild;  
         }  
         
         if (smallest!=index) {  
            swap(index, smallest);  
            index = smallest;
         } else {  
            break;  
         }  
      }  
   }  
   
   private int parent(int index) {  
      return (index-1)/2;  
   }  
   
   private int leftChild(int index) {  
      return 2*index+1;  
   }  
   
   private int rightChild(int index) {  
      return 2*index+2;  
   }  
   
   public void sort() {  
      int orgSize = size;  
      for (int i=orgSize-1; i>0; i--) {  
         swap(0, i);
         size--;  
         sink(0);
      }  
      size = orgSize;
   }  
   
   public void printArray() {
      System.out.println(Arrays.toString(heapArr));
   }
   
   
   public static void main(String[] args) {
      Scanner sc = new Scanner(System.in);
      System.out.println("Please enter the number of tasks:");
      int n = sc.nextInt();
     
      System.out.println("Estimated time:");
      int[] tasks = new int[n];


      for (int i=0; i<n; i++) {
         tasks[i] = sc.nextInt();
      }
     
      System.out.println("Please enter the number of machines:");
      int m = sc.nextInt();
     
      idxMachine minHeap = new idxMachine(m);
      for(int i=0; i<m; i++) {
         minHeap.insert(0);
      }
     
      for (int i=0; i<tasks.length; i++) {
         int currentTask = tasks[i];        
         int minLoad = minHeap.extractMin();
         minLoad += currentTask;            
         minHeap.insert(minLoad);            
      }
     


      int[] machineLoads = new int[m];
      for(int i=0; i<m; i++) {
         machineLoads[i] = minHeap.extractMin();
      }
     
      System.out.println(Arrays.toString(machineLoads));
   }
   
}
