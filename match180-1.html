   private void turnheap(int j) {
        int left = 2 * j + 1;
        int right = 2 * j + 2;


        int big = j;


        if (left < s && a[left] > a[big]) {
          big = left;
        }
        if (right < s && a[right] > a[big]) {
          big = right;
        }


        if (big != j) {
            exchange(j, big);
            turnheap(big);
        }
    }
    private void exchange(int f, int g) {
        int b = a[f];
        a[f] = a[g];
        a[g] = b;
