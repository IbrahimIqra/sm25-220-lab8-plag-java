import java.util.Arrays;

public class Task3{
     public static void buildMinHeap(int[]arr){
             int l=arr.length;

        for(int p=l/2-1;p>=0;p--){

            sink(arr,l,p)
            ;
        }
    }static void sink(int[]arr,int m,int i) {


        int smallest=i;

        int k=2*i+1;

        int t=2*i+2;

        if (k<t && arr[k]<arr[smallest]){
        smallest=k;
        }
        if (t<m && arr[t]<arr[smallest]){
        smallest=t;
        }
        if (smallest!=i){
            int tmp=arr[i];

                arr[i]=arr[smallest];
            arr[smallest] = tmp;

            sink(arr,m,smallest );


        }
    }
    static int[]machinary(int[] arr,int n) {
        if (n>arr.length){
             return null;
        }
           int[]newArr=new int[n];

        for(int i=0;i<n;i++){
            newArr[i] =arr[i];
        }

        buildMinHeap(newArr)
        ;
        for(int i=n;i<arr.length;i++){

              newArr[0]=newArr[0]+arr[i];

            buildMinHeap(newArr);

        }
        return newArr;
    }



    public static void main(String[] args) {
        int[] tasks = {2, 4, 7, 1, 6};
        int m = 4;
        System.out.println(Arrays.toString(machinary(tasks, m)));
    }
}




import java.util.Arrays;
public class Task4{
   public static class MaxHeap{
       int[]heap;
       int size;

    public MaxHeap(int[]arr){
              this.heap = Arrays.copyOf(arr, arr.length);



             this.size=arr.length;
            buildMaxHeap();
        }private void buildMaxHeap(){
            for(int j=size/2-1;j>=0;j--) {
              
                sink(j);
            }
        } 
        private void sink(int k){
            int lar= k;
            int l=2*k+1;
            int r=2*k+2;


            if (l<size && heap[l]>heap[l]){
                lar=l;
            }
            if (r<size && heap[r]>heap[l]){
                lar=r;
            }

            if (lar!=k){

                int temp=heap[k]
                ;
                    heap[k]=heap[lar];
                  heap[lar]=temp;
                
                  sink(lar);
            }
        }

        public int extractMax() {
            if(size<=0)throw new RuntimeException("Heap is empty");


            int max=heap[0];

            heap[0]=heap[size-1];
               size--;

        sink(0);
            return max;
        }
    }

    public static int[]extraction(int[]arr,int l){
        if(l>arr.length){return null;
        }




          MaxHeap maxHeap=new MaxHeap(arr);
        int[]result=new int[l];

        for(int j=0;j<j;j++){ 

            result[j]=maxHeap.extractMax();
        }

        return result;
    }

    public static void main(String[] args) {
        int[] array = {4, 10, 2, 8, 6, 7};
        int k = 3;
        int[] topK = extraction(array, k);
        System.out.println(Arrays.toString(topK));  // Output: [10, 8, 7]
    }
}
