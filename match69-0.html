 class MaxHeap {
    private int[] arrh;
    private int size;
    private int total;
    MaxHeap(int n){
        this.arrh = new int[n];
        this.size = 0;
        this.total = n;
    }



    public void insert(int x){
        if(this.size >= this.total){
        System.out.println("The heap is full.");
        return;
    }
    this.arrh[this.size] = x;
    swim(this.size);
    this.size++;
}
   public void swim(int indx){
    int parent = (indx-1)/2;


    while(indx > 0 && this.arrh[indx] > this.arrh[parent]){
        int store = this.arrh[indx];
        this.arrh[indx] = this.arrh[parent];
        this.arrh[parent] = store;

        indx = parent;
        parent = (indx-1)/2;
    }
   }

   public int extractMax(){
    if(this.size <= 0){
        System.out.println("There's no more elements to remove.");
        return -1;
    }


   int store = this.arrh[0];
   this.arrh[0] = this.arrh[this.size - 1];
   this.arrh[this.size - 1] = 0;
   this.size--;
   sink(0);
   return store;
}
public void sink(int indx){
    int left = 2*indx + 1;


    int right = 2*indx + 2;
    if(left >= this.size){
        return;
    }
    int big = left;
    if(right < this.size && this.arrh[big] < this.arrh[right]){
        big = right;
    }
    if(this.arrh[indx] < this.arrh[big]){
        int store = this.arrh[indx];
        this.arrh[indx] = this.arrh[big];
        this.arrh[big] = store;
        sink(big);
    }

}
  public int[] sort(){
    int[] arr = new int[this.size];
    int bsize = this.size;
for(int i = 0; i < bsize; i++){
    int store = extractMax();
    arr[i] = store;
}
return arr;
}
}


 class MinHeap{

    public void insert(int x){
        if(this.size >= this.total){
        System.out.println("The heap is full.");
        return;
    }
    this.arrh[this.size] = x;
    swim(this.size);
    this.size += 1;
}
   public void swim(int indx){
    int parent = (indx-1)/2;
    while(indx > 0 && this.arrh[indx] < this.arrh[parent]){
        int store = this.arrh[indx];


        this.arrh[indx] = this.arrh[parent];

        indx = parent;
        parent = (indx-1)/2;
    }
   }

   public int extractMin(){

        System.out.println("There's no more elements to remove.");
        return -1;
    }
   int store = this.arrh[0];
   this.arrh[0] = this.arrh[this.size - 1];
   this.arrh[this.size - 1] = 0;
   this.size--;
   sink(0);
   return store;
}
public void sink(int indx){
    int left = 2*indx + 1;
    int right = 2*indx + 2;
    if(left >= this.size){
        return;
    }
    int small = left;

        int store = this.arrh[indx];

        this.arrh[small] = store;
        sink(small);
    }

}
  public int[] sort(){

    int bsize = this.size;

    int store = extractMin();
