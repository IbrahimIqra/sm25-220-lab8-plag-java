public class Task4_KMaxHeap {
    public static void main(String[] args) {
        int[] nums = {4, 10, 2, 8, 6, 7};
        int k = 3;

        int[] ans = KMaxHeap(nums, k);
        printArray(ans); 
    }

    
    public static int[] KMaxHeap(int[] nums, int k) {


        MaxHeap heap = new MaxHeap(nums.length);

        
        for (int i = 0; i < nums.length; i++) {
            heap.insert(nums[i]);
        }

        
        int[] result = new int[k];
        for (int i = 0; i < k; i++) {
            result[i] = heap.extractMax();
        }
        return result;
    }

            }
        }
        System.out.println();
    }
}




public class MaxHeap {
    private int[] heap;
    private int size;
    private int capacity;

    public MaxHeap(int capacity) {
        this.capacity = capacity;
        this.heap = new int[capacity + 1]; 
        this.size = 0;
    }

    public void insert(int value) {
        if (size == capacity) {
            System.out.println("Heap is full! Cannot insert.");
            return;
        }
        size++;
        heap[size] = value;  


        swim(size);          
    }

    public int extractMax() {
        if (size == 0) {
            System.out.println("Heap is empty!");
            return -1; 
        }
        int max = heap[1];           
        heap[1] = heap[size];    
        size--;
        sink(1);                     
        return max;
    }

    public void swim(int index) {
        while (index > 1) {
            int parent = index / 2;


            if (heap[index] > heap[parent]) {
                swap(index, parent);
                index = parent;
            } else {
                break;
            }
        }
    }

    public void sink(int index) {
        while (2 * index <= size) {       
            int left = 2 * index;
            int right = 2 * index + 1;
            int largerChild = left;

            if (right <= size && heap[right] > heap[left]) {
                largerChild = right;
            }

            if (heap[index] < heap[largerChild]) {


                swap(index, largerChild);
                index = largerChild;
            } else {
                break;
            }
        }
    }

    public void swap(int i, int j) {
        int t = heap[i];
        heap[i] = heap[j];
        heap[j] = t;
    }
}

public class Task3_Scheduler {
    public static int[] scheduleTasks(int[] task, int m) {

      if (m <= 0) {
        int [] empty = new int[0];
        return empty;
      }



        MinHeap heap = new MinHeap(m);

        
        for (int i = 0; i < m; i++) {
            heap.insert(0);
        }

        
        for (int j = 0; j < task.length; j++) {
            int t = task[j];
            int smallestLoad = heap.extractMin();
            int updateLoad = smallestLoad + t;
            heap.insert(updateLoad);
        }

        return heap.toArrayCopy();
    }

    public static void printArray(int[] arr) {
        if (arr == null) {
            System.out.println("null");
            return;
        }
        System.out.print("[");
        for (int i = 0; i < arr.length; i++) {
            System.out.print(arr[i]);
            if (i != arr.length - 1) {
              System.out.print(", ");
            }
        }
        System.out.println("]");
    }



    public static void main(String[] args) {
        int[] task1 = {2, 4, 7, 1, 6};
        int m1 = 4;
        int[] result1 = scheduleTasks(task1, m1);
        printArray(result1); 
    }
}



public class MinHeap {
    private int[] heap;
    private int size;
    private int capacity;

    public MinHeap(int capacity) {
        this.capacity = capacity;
        heap = new int[capacity + 1]; 
        size = 0;
    }

    public void insert(int value) {
        if (size == capacity) {
            System.out.println("Heap is full! Cannot insert.");
            return;
        }
        size++;
        heap[size] = value;


        swim(size);
    }

    public int extractMin() {
        if (size == 0) {
            System.out.println("Heap is empty!");
            return -1;
        }
        int min = heap[1];
        heap[1] = heap[size];
        size--;
        sink(1);
        return min;
    }

    public void swim(int index) {
        while (index > 1) {
            int parent = index / 2;
            if (heap[index] < heap[parent]) {
                int temp = heap[index];
                heap[index] = heap[parent];
                heap[parent] = temp;
                index = parent;
            } else {
                break;
            }
        }
    }

    public void sink(int index) {
        while (2 * index <= size) {
            int left = 2 * index;
            int right = 2 * index + 1;
            int smallerChild = left;

            if (right <= size && heap[right] < heap[left]) {
                smallerChild = right;
            }

            if (heap[index] > heap[smallerChild]) {
                int temp = heap[index];
                heap[index] = heap[smallerChild];
                heap[smallerChild] = temp;
                index = smallerChild;
            } else {
                break;
            }
        }
    }

    public int[] toArrayCopy() {
        int[] arr = new int[size];
        for (int i = 1; i <= size; i++) {
            arr[i - 1] = heap[i];
        }
        return arr;
    }
}
