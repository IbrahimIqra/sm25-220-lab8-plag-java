import java.util.Arrays;
public class TaskDistributor {
  
  public static void minHeapify(int[] heap, int i, int size) {
    int smallest = i;
    int left = 2*i;


    int right = 2*i+1;
    
    if (left < size && heap[left] < heap[smallest]) {
      smallest = left;
    }
    if (right < size && heap[right] < heap[smallest]) {
      smallest = right;
    }
    if (smallest != i) {
      int temp = heap[i];
      heap[i] = heap[smallest];
      heap[smallest] = temp;
      minHeapify(heap, smallest, size);
    }
  }
  
  public static int[] distributeTasks(int[] tasks, int m) {
    int[] heap = new int[m];
    for (int i = 0; i < m; i++) {
      heap[i] = 0;
    }
    for (int i = 0; i < tasks.length; i++) {
      heap[0] += tasks[i];           
      minHeapify(heap, 0, m);
    }


    return heap;
  }
  
  public static void main(String[] args) {
    int[] tasks = {2, 4, 7, 1, 6};
    int m = 4;
    System.out.println(Arrays.toString(distributeTasks(tasks, m)));
  }
}

import java.util.Arrays;
public class tLargest {
  
  public static void maxHeapify(int[] heap, int i, int size) {
    int largest = i;


    int left = 2*i;
    int right = 2*i+1;
    
    if (left < size && heap[left] > heap[largest]) {
      largest = left;
    }
    if (right < size && heap[right] > heap[largest]) {
      largest = right;
    }
    if (largest != i) {
      int temp = heap[i];
      heap[i] = heap[largest];
      heap[largest] = temp;
      maxHeapify(heap, largest, size);
    }
  }
  
  public static void build_Max_Heap(int[] heap) {
    for (int i = heap.length / 2; i >= 0; i--) {
      maxHeapify(heap,i,heap.length);
    }
  }
  


  public static int extract_Max(int[] heap, int size) {
    int max = heap[0];
    heap[0] = heap[size - 1];
    heap[size - 1]=0;
    maxHeapify(heap,0,size - 1);
    return max;
  }
  
  public static int[] tpLargest(int[] nums, int k) {
    int[] heap = new int[nums.length];
    for (int i = 0; i < nums.length; i++) {
      heap[i] = nums[i]; 
    }
    build_Max_Heap(heap);
    int[] temp = new int[k];
    for (int i = 0; i < k; i++) {
      temp[i] = extract_Max(heap, nums.length - i);
    }


    return temp;
  }
  
  public static void main(String[] args) {
    int[] nums = {4, 10, 2, 8, 6, 7};
    int k = 3;
    System.out.println(Arrays.toString(tpLargest(nums, k)));
  }
}
