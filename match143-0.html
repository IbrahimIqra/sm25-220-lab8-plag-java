public class task3 {

    public static int [] processingTime(int [] tasks, int m){
        MinHeap mHeap=new MinHeap(m);
        for(int i=0;i<m;i++){
            mHeap.insert(0);
        }
        for(int i=0;i<tasks.length;i++){
            int current=mHeap.extractMin();
            int sum=current+tasks[i];
            mHeap.insert(sum);
        }
        int [] finalArr=new int[m];
        for(int i=0;i<m;i++){
            finalArr[i]=mHeap.extractMin();
        }
        return finalArr;
    }

    //Driver code


    public static void main(String[] args) {
        int [] tasks={2,4,7,1,6};
        int m=4;
        int [] mLoads=processingTime(tasks, m);
         System.out.print("[");
        for(int j=0;j<mLoads.length;j++){
            System.out.print(mLoads[j]);
            if(j<mLoads.length-1){
                System.out.print(",");
            }
        }
        System.out.print("]");
    }
}

public class task4 {
    
    public static int [] largestElems(int [] nums, int k){
        MaxHeap mHeap=new MaxHeap(nums.length);
        for(int i=0;i<nums.length;i++){
            mHeap.insert(nums[i]);
        }
        int [] resultArr=new int[k];
        for(int i=0;i<k;i++){
            resultArr[i]=mHeap.extractMax();
        }
        return resultArr;
    }

    //Driver code
    public static void main(String[] args) {
        int [] nums={4,10,2,8,6,7};
        int k=3;
        int [] largest=largestElems(nums, k);
        System.out.print("[");
        for(int j=0;j<largest.length;j++){
            System.out.print(largest[j]);
            if(j<largest.length-1){
                System.out.print(",");
            }
        }
        System.out.print("]");
    }
}
