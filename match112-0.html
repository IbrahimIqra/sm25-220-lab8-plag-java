import java.util.Arrays;
public class Task3 {
    static class MinHeap {
        private int[] arr;
        private int size;
        private int capacity;

        MinHeap(int cap) {
            capacity = cap;
            arr = new int[cap];


            size = 0;
        }

        private int parent(int i) {
            return (i - 1) / 2;
        }

        private int left(int i) {
            return 2 * i + 1;
        }

        private int right(int i) {
            return 2 * i + 2;
        }

        private void swap(int i, int j) {
            int temp = arr[i];
            arr[i] = arr[j];
            arr[j] = temp;
        }
        private void up(int i) {
            if (i <= 0) return;
            int p = parent(i);
            if (arr[i] < arr[p]) {
                swap(i, p);


                up(p);
            }
        }

        private void down(int i) {
            int l = left(i);
            int r = right(i);
            int s = i;
            if (l < size && arr[l] < arr[s]) {
                s = l;
            }

            if (r < size && arr[r] < arr[s]) {
                s = r;
            }

            if (s != i) {
                swap(i, s);
                down(s);
            }
        }

        public void insert(int v) {
            if (size == capacity) return;
            arr[size] = v;
            size++;
            up(size - 1);
        }

        public int eMin() {
            if (size == 0) return 0;
            int root = arr[0];
            arr[0] = arr[size - 1];
            arr[size - 1] = 0;
            size--;
            if (size > 0) {
                down(0);
            }
            return root;
        }

        public int[] toArray() {
            int[] res = new int[size];
            for (int i = 0; i < size; i++) {
                res[i] = arr[i];
            }

            for (int i = 0; i < res.length - 1; i++) {
                int minIdx = i;
                for (int j = i + 1; j < res.length; j++) {
                    if (res[j] < res[minIdx]) {
                        minIdx = j;
                    }
                }
                int temp = res[i];
                res[i] = res[minIdx];
                res[minIdx] = temp;
            }

            return res;
        }
    }



    public static int[] t3(int[] tasks, int m) {
        MinHeap heap = new MinHeap(m);

        for (int i = 0; i < m; i++) {
            heap.insert(0);
        }
        for (int i = 0; i < tasks.length; i++) {
            int smallest = heap.eMin();
            int updated = smallest + tasks[i];
            heap.insert(updated);
        }
        return heap.toArray();
    }
    public static void main(String[] args) {
        int[] tasks = {2, 4, 7, 1, 6};
        int m = 4;
        int[] result = t3(tasks, m);
        System.out.println(Arrays.toString(result));
    }

import java.util.Arrays;
public class Task4 {
    static class MaxHeap {
        private int[] arr;
        private int size;
        private int capacity;

        MaxHeap(int cap) {
            capacity = cap;
            arr = new int[cap];


            size = 0;
        }

        private int p(int i) {
            return (i - 1) / 2;
        }

        private int l(int i) {
            return 2 * i + 1;
        }

        private int r(int i) {
            return 2 * i + 2;
        }

        private void up(int i, int j) {
            int temp = arr[i];
            arr[i] = arr[j];
            arr[j] = temp;
        }

        private void down(int i) {
            if (i <= 0) {
                return;
            }
            int p = p(i);
            if (arr[i] > arr[p]) {
                up(i, p);


                down(p);
            }
        }
        private void sink(int i) {
            int l = l(i);
            int r = r(i);
            int big = i;
            if (l < size && arr[l] > arr[big]) {
                big = l;
            }

            if (r < size && arr[r] > arr[big]) {
                big = r;
            }

            if (big != i) {
                up(i, big);
                sink(big);
            }
        }

        public void insert(int v) {
            if (size == capacity) {
                return;
            }
            arr[size] = v;
            size++;
            down(size - 1);
        }

        public int eMax() {
            if (size == 0) {
                return 0;
            }
            int root = arr[0];
            arr[0] = arr[size - 1];
            arr[size - 1] = 0;
            size--;
            if (size > 0) {
                sink(0);
            }
            return root;
        }
    }

    public static int[] t4(int[] nums, int k) {
        MaxHeap heap = new MaxHeap(nums.length);
        for (int i = 0; i < nums.length; i++) {
            heap.insert(nums[i]);
        }

        int[] res = new int[k];


        for (int i = 0; i < k; i++) {
            res[i] = heap.eMax();
        }

        return res;
    }
    public static void main(String[] args) {
        int[] nums = {4, 10, 2, 8, 6, 7};
        int k = 3;
        int[] result = t4(nums, k);
        System.out.println(Arrays.toString(result));  // [10, 8, 7]
    }
