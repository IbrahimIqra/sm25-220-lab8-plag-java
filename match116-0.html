class MinHeap {
    private int[] h;
    private int sz;
    private int cap;


    public MinHeap(int cap) {
        this.cap = cap;
        this.sz = 0;
        this.h = new int[cap];


    }


    private int p(int i){
        return (i-1)/2;}
    private int l(int i){
        return 2*i+1;}
    private int r(int i){
        return 2*i+2;}


    private void swap(int i, int j){
        int t = h[i];
        h[i] = h[j];
        h[j] =t;
    }


    public void insert(int v){
        h[sz] = v;
        int i = sz;
        while (i>0 && h[i]<h[p(i)]){
            swap(i, p(i));
            i=p(i);
        }
        sz++;
    }


    public int extractMin(){


        int min = h[0];
        h[0] = h[sz-1];
        sz--;
        sink(0);
        return min;
    }


    private void sink(int i){
        int s = i;
        int L = l(i);


        int R = r(i);


        if (L<sz && h[L]<h[s]){
            s = L;
        }
        if (R<sz && h[R]<h[s]){
            s = R;
        }


        if (s != i){
            swap(i, s);


            sink(s);
        }
    }
}


public class a8t3{
    public static int[] distribute(int[] tasks, int m) {
        MinHeap heap = new MinHeap(m);
        for (int i = 0; i < m; i++){
            heap.insert(0);
        }
        int p=tasks.length;
        for (int i = 0; i <p; i++){
            int load = heap.extractMin();
            heap.insert(load + tasks[i]);
        }


        int[] res = new int[m];
        for (int i = 0; i < m; i++) {
            res[i] = heap.extractMin();
        }
        return res;
    }

        }
    }

class MaxHeap{
    private int[] h;
    private int sz;
    private int cap;


    public MaxHeap(int cap){
        this.cap = cap;
        this.sz = 0;
        this.h = new int[cap];
    }


    private void swap(int i, int j){
        int t = h[i];
        h[i] = h[j];
        h[j] = t;
    }


    public void insert(int v){
        int i = sz;
        h[sz] = v;


       
        while (i>0 && h[i]>h[(i-1)/2]) {
            swap(i, (i-1)/2);
            i = (i-1)/2;
        }
        sz++;
    }


    public int extractMax(){


        int max = h[0];
        h[0]=h[sz-1];
        sz--;
        sink(0);
        return max;
    }


    private void sink(int i) {
        int lg=i;
        int L=2*i+1;
        int R=2*i+2;


        if (L<sz && h[L]>h[lg]){


            lg = L;
        }
        if (R<sz && h[R]>h[lg]) {
            lg = R;
        }


        if (lg != i){
            swap(i, lg);
            sink(lg);
        }
    }
}


public class a8t4{
    public static int[] topK(int[] arr, int k) {
        MaxHeap heap = new MaxHeap(arr.length);
        for (int i = 0; i < arr.length; i++){
            heap.insert(arr[i]);
        }


        int[] res = new int[k];
        for (int i = 0; i < k; i++) {
            res[i] = heap.extractMax();
        }
        return res;
    }


    public static void main(String[] args) {


        int[] nums = {4, 10, 2, 8, 6, 7};
        int k = 3;
        int[] ans = topK(nums, k);
        for (int i = 0; i < ans.length; i++) {
            System.out.print(ans[i] + " ");
        }
    }
