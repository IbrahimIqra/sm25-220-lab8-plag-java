        int[] arr = {2, 4, 7, 1, 6};

        System.out.print(Arrays.toString(res));

    public static class MinHeap {
        private int[] heap;
        private int size;
        private int cap;

        public MinHeap(int cap) {
            this.cap = cap;
            this.heap = new int[cap + 1];
            this.size = 0;
        }

        public void insert(int val) {
          if(size >= cap){
            System.out.println("Heap is full!");
          }
          else{
            heap[++size] = val;
            swim(size);
          }
        }

        public int extractMin() {
          if(size == 0){
           System.out.println("Heap is empty!");
          }
            int min = heap[1];
            swap(1, size--);
            sink(1);


            return min;
        }

        private void swim(int i) {
            while(i > 1 && (heap[i] < heap[i / 2])){
                swap(i, i / 2);
                i = i / 2;
            }
        }

        private void sink(int i) {


            while((2 * i) <= size) {
                int j = (2 * i);
                if ((j < size) && (heap[j] > heap[j + 1])){
                 j++;
                }
                if (heap[i] <= heap[j]){
                  break;
                }
                swap(i, j);
                i = j;
            }
        }

        private void swap(int i, int j) {


            int temp = heap[i];
            heap[i] = heap[j];
            heap[j] = temp;
        }
    }

/////////////////////////////////////////////////////////////
    public static int[] tasks(int[] tasks, int m) {
        MinHeap heap = new MinHeap(m);
        for(int i=0; i < m; i++) {
            heap.insert(0);
        }
        for(int i = 0; i< tasks.length; i++){
            int time = tasks[i];
            int minLoad = heap.extractMin();
            heap.insert(minLoad + time);
        }
        int[] res = new int[m];
        for (int i = 0; i < m ; i++){
            res[i] = heap.extractMin();
        }
        return res;
    }

}

    static class MaxHeap {
        private int[] heap;
        private int size;
        private int cap;

        public MaxHeap(int cap) {
            this.cap = cap;
            this.heap = new int[cap + 1]; 
            this.size = 0;
        }

        public void insert(int val) {
          if(size >= cap){
            System.out.println("Heap is full!");
          }
          else{
            heap[++size] = val;
            swim(size);
          }
        }
        public int extractMax() {
          if(size == 0){
           System.out.println("Heap is empty!");
          }
          int max = heap[1];
          swap(1, size--);
          sink(1);
          return max;
        }



        public void swim(int i) {
            while(i > 1 && (heap[i] > heap[i / 2])) {
                swap(i, i / 2);
                i = i / 2;
            }
        }

        public void sink(int i) {
            while ((2 * i) <= size) {
                int j = (2 * i);
                if((j < size) && (heap[j] < heap[j + 1])){
                 j++;
                }
                if (heap[i] >= heap[j]){
                 break;
                }
                swap(i, j);
                i = j;
            }
        }
        public void swap(int i, int j) {
            int temp = heap[i];
            heap[i] = heap[j];


            heap[j] = temp;
        }
    }
//////////////////////////////////////////////////////////
    public static int[] top(int[] num, int k) {
        MaxHeap heap = new MaxHeap(num.length);
        for(int i = 0; i < num.length; i++) {
          heap.insert(num[i]);
        }
        int[] res = new int[k];
        for(int i = 0; i < k; i++) {
          res[i] = heap.extractMax();
        }
        return res;
    }
