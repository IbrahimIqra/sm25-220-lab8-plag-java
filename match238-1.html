public class Task301
{
    public static int[] distributeTasks(int[] tasks, int m) {
         int[] heap  = new int[ m ];  


        for (int t  = 0; t <  tasks.length;  t=t+1) {
            int task  = tasks[t ], minIndex =  0;
                for (int i = 1; i < m; i++) {
                    if (heap[i] < heap[minIndex]) minIndex = i;
                    
                }

           
            System.out.print("Task " + (t+1) + " (Processing Time = " + task + "): ");
            System.out.print("Assign to Machine " + (minIndex+1) ); 
            System.out.println(" (current load = " + heap[minIndex] + ")");

             heap[minIndex ] +=  task;

         System.out.print("Current machine loads: [");
                for (int i = 0; i < m; i++) {
                    System.out.print (heap[ i ]);
                    if (i <  m - 1)  System.out.print(", ");
                }
                System.out.println("]");
                System.out.println("--------------------------------");
        }

         return heap;
    }

    public static void main(String[] args) {
          int m = 4;


        int[] tasks = { 2, 4, 7 , 1 ,  6}, NormResulty = distributeTasks(tasks, m);
      



        System.out.print( " Final result: [");
        for (int i = 0; i <  NormResulty.length; i++) {
            System.out.print (NormResulty[i ]);
            if (i < NormResulty.length  - 1) System.out.print(" , ");
        }
        System.out.println(  " ]");
    }
}

System.out.print (" Top " + k + " largest elements: [");

        public static int[] getTopK(int[] nums, int k) {
        int iterator = nums.length;  int[] heap = new int[iterator];

      


        for (int i = 0; i < iterator; i++)  heap[i] = nums[i];
        
        buildMaxHeap(heap, iterator);

        int[] result = new int[k];
        for (int i = 0; i < k; i++) {
         result[i] = heap[0];   heap[0 ] = heap[iterator - 1 - i]; 
             maxHeapify(heap, 0, iterator - 1 - i); 
        }

         return result; 
    }

    // Build max-heap
    private static void buildMaxHeap(int[] heap, int n) {
        for (int i = n / 2 - 1; i >= 0; i--) 


            maxHeapify(heap, i, n);
   
    }

    // Max-heapify function
    private static void maxHeapify(int[] heap, int i, int n) {
        int bigINT = i, left = 2 * i + 1, right = 2 * i + 2;



        if (left < n && heap[left] > heap[bigINT])  bigINT = left;
        
        if (right < n && heap[right] > heap[bigINT])    bigINT = right;
        

        if ( bigINT  !=  i) { int temp = heap[i ]; heap[i]  = heap[ bigINT]; heap[bigINT]  = temp;
            maxHeapify(heap, bigINT, n);
        }
    }
