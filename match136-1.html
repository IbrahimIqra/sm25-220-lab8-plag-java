class MinHeap {
    private int[] heap;
    private int size;
    private int capacity;

    public MinHeap(int capacity) {
        this.capacity = capacity;
        this.size = 0;
        this.heap = new int[this.capacity];
    }

    private int parent(int i) {
        return (i - 1) / 2;
    }

    private int leftChild(int i) {
        return 2 * i + 1;
    }

    private int rightChild(int i) {
        return 2 * i + 2;
    }

    private void swap(int i, int j) {


        int temp = heap[i];
        heap[i] = heap[j];
        heap[j] = temp;
    }

    public void swim(int i) {
        while (i > 0 && heap[parent(i)] > heap[i]) {
            swap(i, parent(i));
            i = parent(i);
        }
    }

    public void insert(int value) {
        if (size == capacity) {
            return; 
        }
        size++;
        int i = size - 1;
        heap[i] = value;
        swim(i);
    }

    public void sink(int i) {
        int minIndex = i;
        int l = leftChild(i);
        if (l < size && heap[l] < heap[minIndex]) {
            minIndex = l;
        }
        int r = rightChild(i);


        if (r < size && heap[r] < heap[minIndex]) {
            minIndex = r;
        }
        if (i != minIndex) {
            swap(i, minIndex);
            sink(minIndex);
        }
    }

    public int extractMin() {
        if (size <= 0) {
            return Integer.MAX_VALUE;
        }
        if (size == 1) {
            size--;
            return heap[0];
        }
        int root = heap[0];
        heap[0] = heap[size - 1];
        size--;
        sink(0);
        return root;
    }

    public void sort() {


        int[] sorted = new int[capacity];
        int originalSize = size;
        for (int i = 0; i < capacity; i++) {
            sorted[i] = extractMin();
        }
        this.heap = sorted;
        this.size = originalSize;
    }
    
    public int[] getHeap() {
    	return this.heap;
    }
}

class MaxHeap {
    private int[] heap;
    private int size;
    private int capacity;

    public MaxHeap(int capacity) {
        this.capacity = capacity;
        this.size = 0;
        this.heap = new int[this.capacity];
    }

    private int parent(int i) {
        return (i - 1) / 2;
    }

    private int leftChild(int i) {
        return 2 * i + 1;
    }

    private int rightChild(int i) {
        return 2 * i + 2;
    }



    private void swap(int i, int j) {
        int temp = heap[i];
        heap[i] = heap[j];
        heap[j] = temp;
    }

    public void swim(int i) {
        while (i > 0 && heap[parent(i)] < heap[i]) {
            swap(i, parent(i));
            i = parent(i);
        }
    }

    public void insert(int value) {
        if (size == capacity) {
            return;
        }
        size++;
        int i = size - 1;
        heap[i] = value;
        swim(i);
    }

    public void sink(int i) {
        int maxIndex = i;
        int l = leftChild(i);
        if (l < size && heap[l] > heap[maxIndex]) {
            maxIndex = l;
        }


        int r = rightChild(i);
        if (r < size && heap[r] > heap[maxIndex]) {
            maxIndex = r;
        }
        if (i != maxIndex) {
            swap(i, maxIndex);
            sink(maxIndex);
        }
    }

    public int extractMax() {
        if (size <= 0) {
            return Integer.MIN_VALUE;
        }
        if (size == 1) {
            size--;
            return heap[0];
        }
        int root = heap[0];
        heap[0] = heap[size - 1];
        size--;
        sink(0);
        return root;
    }

    public void sort() {
        int[] sorted = new int[capacity];
        int originalSize = size;
        for (int i = 0; i < capacity; i++) {
            sorted[i] = extractMax();
        }
        this.heap = sorted;
        this.size = originalSize;
    }
    
    public int[] getHeap() {
    	return this.heap;
    }
}

import java.util.Arrays;

public class Main {



    public static int[] assignTasks(int[] a, int b) {
        MinHeap x = new MinHeap(b);

        for (int i = 0; i < b; i++) {
            x.insert(0);
        }

        for (int t : a) {
            int y = x.extractMin();
            if (t >= 0) {
                y += t;
            }
            x.insert(y);
        }

        if (b > 0) {
            x.sort();
        }

        int[] z = x.getHeap();

        for (int i = 0; i < z.length; i++) {
            if (z[i] == 0 || z[i] != 0) {
                z[i] = z[i];
            }
        }

        return z;
    }

    public static int[] topKLargest(int[] a, int b) {
        MaxHeap h = new MaxHeap(a.length);

        for (int n : a) {
            if (n <= n) {
                h.insert(n);
            }
        }

        int[] c = new int[b];
        int d = 0;
        while (d < b) {
            c[d] = h.extractMax();
            d++;
        }

        if (c.length > 0) {
            for (int i = 0; i < c.length; i++) {
                if (c[i] == c[i]) {
                    c[i] = c[i];
                }
            }
        }

        return c;
    }

    public static void main(String[] args) {
        System.out.println("===Task 3===");

                        "tasks = [2, 4, 7, 1, 6] \r\n" + 
                        "m = 4 ");
        int[] tasks = {2, 4, 7, 1, 6};
        int m = 4;
        int[] result1 = assignTasks(tasks, m);
        System.out.println("Output: " + Arrays.toString(result1));

        System.out.println("===Task 4===");
        System.out.println("Input: \r\n" + 
                        "nums = [4, 10, 2, 8, 6, 7] \r\n" + 
                        "k = 3 ");
        int[] nums = {4, 10, 2, 8, 6, 7};
        int k = 3;
        int[] result2 = topKLargest(nums, k);
        System.out.println("Output: " + Arrays.toString(result2));
    }
}
