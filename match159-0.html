public class Task3{


  public static int[] taskProcessing(int[]tasks,int m){
    MinHeap heap=new MinHeap(m);
    for(int i=0;i<m;i++){
      heap.insert(0);
    }
    for(int i=0;i<tasks.length;i++){
      int ld=heap.extractMin();
      ld+=tasks[i];
      heap.insert(ld);
    }
    int r[]=new int[m];


    for(int i=0;i<m;i++){
      r[i]=heap.extractMin();
    }
    return r;
  }
    public static void main(String[]args){


      int[]a={2,4,7,1,6};

        if(j==res.length-1){

        }
      }
