    public static void insert(int[] arr,int val,int a){
        arr[a]=val;
        int i=a;
        while(i>0){
        int par=(i-1)/2;
        if(arr[par]>arr[i]){
        swap(arr,par,i);
        i=par;} 
        else{break;}}}


    public static void swap(int[] arr,int i,int j){
        int temp=arr[i];
        arr[i]=arr[j];
        arr[j]=temp;}




    public static void sink(int[] arr,int j,int i){
        int val=i;
        int left=2*i+1;
        int right=2*i+2;
        if(left<j&&arr[left]<arr[val]){
        val = left;}
        if(right<j&&arr[right]<arr[val]){
        val=right;}
        if(val!=i){
        swap(arr,i,val);


        sink(arr,j,val);}}


    public static int extractMin(int[] arr,int i){
        int min=arr[0];
        arr[0]=arr[i-1];
        sink(arr,i - 1,0);
        return min;}


    public static int[] mainfunction(int[] arr,int i){
        int len=arr.length;
        int[] ar=new int[len+i];
        int size=0;
        int j=0;
        
        while(j<i){
        insert(ar,0,size);
        size++;
        j++;}
        j = 0;
        
        while(j<len){
        int min=extractMin(ar,size);
        size--;
        insert(ar,min + arr[j],size);
        size++;
        j++;}
        int[] result = new int[i];
        j=0;
        
        while(j<i){
        result[j]=extractMin(ar,size);
        size--;
        j++;}
        return result;}
}

import java.util.Arrays;
public class Task04{


public static void main(String[] shaan){

public static void swap(int[] arr,int i,int j){
  int temp=arr[i];
  arr[i]=arr[j];
  arr[j]=temp;}




public static void sink(int[] arr,int n,int i){
  int max=i;
  int left=2*i+1;
  int right=2*i+2;
if(left<n&&arr[left]>arr[max]){
  max= left;}
if(right<n&&arr[right]>arr[max]){
  max = right;}
if(max!=i){
  swap(arr,i,max);
  sink(arr, n, max);}}


public static void buildHeap(int[] arr,int n){
   int j=(n/2)-1;
   while(j>=0){
   sink(arr,n,j);
   j--;}}


public static int extractMax(int[] arr,int n){


    int max=arr[0];
    arr[0]= arr[n-1];
    sink(arr,n-1,0);
    return max;}


public static int[] mainfunction(int[] arr,int k){
    int n=arr.length;
    int[] result=new int[k];
    buildHeap(arr,n);
    int j=0;
    while(j<k){
    result[j]=extractMax(arr,n);
    n--;
    j++;}
    return result;}}
