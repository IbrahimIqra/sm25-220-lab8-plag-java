class MinHeap {
    private int[] heap;
    private int size;
    private int capacity;

    MinHeap(int cap) {
        this.heap = new int[cap];
        this.size = 0;
        this.capacity = cap;
    }

    public void insert(int val) {
        if (this.size >= this.capacity) {
            System.out.println("Heap is full");
            return;
        }
        this.heap[this.size] = val;
        swim(this.size);
        this.size += 1;
    }

    private void swim(int indx) {
        int pIndx = (indx - 1) / 2;
        while (indx > 0 && this.heap[indx] < this.heap[pIndx]) {
            int temp = this.heap[indx];
            this.heap[indx] = this.heap[pIndx];
            this.heap[pIndx] = temp;
            indx = pIndx;
            pIndx = (indx - 1) / 2;
        }
    }

    public int removeMin() {
        if (this.size <= 0) {
            System.out.println("No more elements to remove!");
            return -1;
        }
        int temp = this.heap[0];
        this.heap[0] = this.heap[this.size - 1];
        this.heap[this.size - 1] = 0;
        this.size -= 1;
        heapify(0);
        return temp;
    }

    private void heapify(int indx) {
        int left = 2 * indx + 1;
        int right = 2 * indx + 2;
        if (left >= this.size){
            return;
        }
        int smallest = left;
        if (right < this.size && this.heap[right] < this.heap[left]) {
            smallest = right;
        }
        if (this.heap[indx] > this.heap[smallest]) {
            int temp = this.heap[indx];
            this.heap[indx] = this.heap[smallest];
            this.heap[smallest] = temp;
            heapify(smallest);
        }
    }
}

public class TaskDistributor {


    public static int[] distributeTasks(int[] tasks, int m) {
        MinHeap heap = new MinHeap(m);
        for (int i = 0; i < m; i++) {
            heap.insert(0);
        }
        for (int i = 0; i < tasks.length; i++) {
            int minLoad = heap.removeMin();
            int updatedLoad = minLoad + tasks[i];
            heap.insert(updatedLoad);
        }
        int[] result = new int[m];


        for (int i = 0; i < m; i++) {
            result[i] = heap.removeMin();
        }
        return result;
    }

    public static void printArray(int[] arr) {
        System.out.print("[");
        for (int i = 0; i < arr.length; i++) {
            System.out.print(arr[i]);
            if (i < arr.length - 1) {
                System.out.print(", ");
            }
        }
        System.out.println("]");
    }

    public static void main(String[] args) {


        int[] tasks = {2, 4, 7, 1, 6};
        int m = 4;
        int[] result = distributeTasks(tasks, m);
        System.out.println("Expected Result: [2, 4, 7, 7]");
        System.out.println("My result:");
        printArray(result);
    }
}

class MaxHeap {
    private int[] heap;
    private int size;
    private int capacity;

    MaxHeap(int cap) {
        this.heap = new int[cap];
        this.size = 0;
        this.capacity = cap;
    }

    public void insert(int val) {
        if (this.size >= this.capacity) {
            System.out.println("Heap is full");
            return;
        }
        this.heap[this.size] = val;
        swim(this.size);
        this.size += 1;
    }

    private void swim(int indx) {
        int pIndx = (indx - 1) / 2;
        while (indx > 0 && this.heap[indx] > this.heap[pIndx]) {
            int temp = this.heap[indx];
            this.heap[indx] = this.heap[pIndx];
            this.heap[pIndx] = temp;
            indx = pIndx;
            pIndx = (indx - 1) / 2;
        }
    }

    public int removeMax() {
        if (this.size <= 0) {
            System.out.println("No more elements to remove!");
            return -1;
        }
        int temp = this.heap[0];
        this.heap[0] = this.heap[this.size - 1];
        this.heap[this.size - 1] = 0;
        this.size -= 1;
        heapify(0);
        return temp;
    }

    private void heapify(int indx) {
        int left = 2 * indx + 1;
        int right = 2 * indx + 2;
        int largest = indx;

        if (left < this.size && this.heap[left] > this.heap[largest]) {
            largest = left;
        }
        if (right < this.size && this.heap[right] > this.heap[largest]) {
            largest = right;
        }
        if (largest != indx) {
            int temp = this.heap[indx];
            this.heap[indx] = this.heap[largest];
            this.heap[largest] = temp;
            heapify(largest);
        }
    }
}

public class TopKElements {


    public static int[] findTopK(int[] nums, int k) {
        MaxHeap heap = new MaxHeap(nums.length);

        for (int i = 0; i < nums.length; i++) {
            heap.insert(nums[i]);
        }

        int[] result = new int[k];
        for (int i = 0; i < k; i++) {
            result[i] = heap.removeMax();
        }



        return result;
    }

    public static void printArray(int[] arr) {
        System.out.print("[");
        for (int i = 0; i < arr.length; i++) {
            System.out.print(arr[i]);
            if (i < arr.length - 1) {
                System.out.print(", ");
            }
        }
        System.out.println("]");
    }

    public static void main(String[] args) {
