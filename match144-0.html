public class TaskScheduler{
  public static int[] assignTasks(int[] tasks, int machineCount){
        MinHeap heap = new MinHeap(machineCount); 
                                                                   //Initialize machines with 0 load
        for (int i = 0; i < machineCount; i++){ 
          heap.insert(0);
        }
        for (int task : tasks){
            int leastLoad = heap.removeMin();     //pick machine with smallest load
            leastLoad += task;                
            heap.insert(leastLoad);               //put updated load back
        }
      return heap.getHeap(); // final machine
    }

 // Driver code


    public static void main(String[] args){
        int[] tasks ={2, 4, 7, 1, 6};
        int machines = 4;
      int[] finalLoads = assignTasks(tasks, machines);

        System.out.print("Final machine loads: ");
        for (int load : finalLoads) System.out.print(load+" ");
    }
}

public class MinHeap {
    private int[] heap;  
    private int size;   
// Constructor: create empty heap for given capacity
  public MinHeap(int capacity) {
     heap = new int[capacity + 1]; 
       size = 0;
    }



    public void insert(int load) {
        size++;
        heap[size] = load;
        swim(size);  //fix heap upwards
    }

    public int removeMin() {
        int smallest = heap[1];
        heap[1] = heap[size];
        size--;
        sink(1);     //fix heap downwards
        return smallest;
    }

   private void swim(int index) {   //Move up to maintain min-heap
        while (index > 1) {
            int parent = index / 2;
            if (heap[index] < heap[parent]) {
                swap(index, parent);
                index = parent;


            } else break;
        }
    }

    private void sink(int index) {  // Move down to maintain min-heap
        while (true) {
            int left = 2 * index;
            int right = 2 * index + 1;
            int smallest = index;

            if (left <= size && heap[left] < heap[smallest]) smallest = left;
            if (right <= size && heap[right] < heap[smallest]) smallest = right;

            if (smallest != index) {
                swap(index, smallest);
                index = smallest;
            } else break;
        }
    }

    private void swap(int i, int j) {


        int temp = heap[i];
        heap[i] = heap[j];
        heap[j] = temp;
    }

    public int[] getHeap() {
        int[] result = new int[size];
        for (int i = 1; i <= size; i++) result[i - 1] = heap[i];
        return result;
    }
}

    public MaxHeap(int capacity){

    }

   public void insert(int number) {
       size++;
       heap[size] = number;  
       swim(size);          
    }

    public int removeMax() {
        int largest = heap[1];      // root
        heap[1] = heap[size];       // move last element to root
        size--;
        sink(1);                    // move down to fix heap
        return largest;
    }

    private void swim(int index) {  // Move element up to maintain maxheap 
        while (index > 1) {
           int parent = index /2;
            if (heap[index] > heap[parent]) {
                swap(index, parent);
                index = parent;


            } else break;       //parent bigger â†’ stop
        }
    }

  private void sink(int index) {
        while (true) {
            int left = 2 * index;
            int right = 2 * index + 1;
            int largest = index;

            if (left <= size && heap[left] > heap[largest]) largest = left;

                index = largest;
            } else break;
