class MinHeap {
    private int[] data;
    private int count;

    public inHeap(int capacity) {
        data = new int[capacity];
        count = 0;


    }
    public int Min() {
        int root = data[0];
        swapValues(0, count - 1);
        count--;
        sinkDown(0, count);
        return root;
    }
    private void ensureCapacity() {


        if (count == data.length) {
            int[] bigger = new int[data.length + 1];
            System.arraycopy(data,0, bigger, 0, data.length);
            data = bigger;
        }
    }

    private void swapValues(int a, int b) {
        int temp = data[a];
        data[a] = data[b];
        data[b] = temp;
    }
    

   public void insertValue(int num) {
    ensureCapacity();
    int position = count;
    data[position] = num;

    while (position > 0) {
        int parentIndex = (position - 1) / 2;
        if (data[position] < data[parentIndex]) {
            swapValues(position, parentIndex);
            position = parentIndex;
        } else {


            break;
        }
    }
    count++;
}


    private void rise(int child, int parent) {
        while (child > 0 && data[child] < data[parent]) {
            swapValues(child, parent);
            child = parent;
            parent = (child - 1) / 2;
        }
    }

    public int top() {
        return data[0];
    }

    private void sinkDown(int i, int x) {
        int smallest = i;
        int left = 2*i + 1;
        int right = 2*i + 2;



        if (left <x && data[left] < data[smallest]) {
            smallest = left;
        }
        if (right< x&& data[right] <data[smallest]) {
            smallest = right;
        }
        if (smallest != i) {
            swapValues(i, smallest);
            sinkDown(smallest,x);
        }
    }
    public boolean empty() {


        return count == 0;
    }
  
}
