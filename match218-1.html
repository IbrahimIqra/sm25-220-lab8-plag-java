public class MinHeap {
    private int[] heap;
    private int size;

    public MinHeap(int n) {
        size = 0;
        heap = new int[n];
    }

    public void insert(int x) {
        if (size >= heap.length) {
            return;
        }
        heap[size] = x;
        swim(size);
        size++;
    }



    public void swim(int y) {
        while (y > 0) {
            int p = (y - 1) / 2;
            if (heap[y] < heap[p]) {
                int temp = heap[y];
                heap[y] = heap[p];
                heap[p] = temp;
                y = p;
            } else {
                break;
            }
        }
    }

    public int extractMin() {
        if (size == 0) {
            return 0;
        }


        int min = heap[0];
        heap[0] = heap[size - 1];
        size--;
        sink(0);
        return min;
    }

    public void sink(int i) {
        while (2 * i + 1 < size) {
            int left = 2 * i + 1;


            int right = 2 * i + 2;
            int smallest = left;

            if (right < size && heap[right] < heap[left]) {
                smallest = right;
            }

            if (heap[i] > heap[smallest]) {
                int temp = heap[i];
                heap[i] = heap[smallest];
                heap[smallest] = temp;
                i = smallest;
            } else {
                break;
            }
        }
    }

    public void sort() {
        int ogsize = size;
        for (int i = ogsize - 1; i > 0; i--) {
            int temp = heap[0];
            heap[0] = heap[i];
            heap[i] = temp;
            size--;
            sink(0);
        }
        size = ogsize;
    }

    public int[] get() {
        return heap;
    }
}

public class Task3 {


    public static int[] arrange(int[] tasks, int m) {
        MinHeap heap = new MinHeap(m);

        for (int i = 0; i < m; i++) {
            heap.insert(0);
        }

        for (int i = 0; i < tasks.length; i++) {
            int task = tasks[i];

            int currLoad = heap.extractMin();
            currLoad += task;

            heap.insert(currLoad);
        }

        return heap.get();
    }
}

public class MaxHeap {
    private int[] heap;
    private int size;

    public MaxHeap(int n) {
        size = 0;
        heap = new int[n];
    }

    public void insert(int x) {
        if (size >= heap.length) {
            return;
        }
        heap[size] = x;
        swim(size);
        size++;
    }



    public void swim(int i) {
        while (i > 0) {
            int p = (i - 1) / 2;
            if (heap[i] > heap[p]) {
                int temp = heap[i];
                heap[i] = heap[p];
                heap[p] = temp;
                i = p;
            } else {
                break;
            }
        }
    }

    public int extractMax() {
        if (size == 0) {
            return 0;
        }


        int max = heap[0];
        heap[0] = heap[size - 1];
        size--;
        sink(0);
        return max;
    }

    public void sink(int i) {
        while (2 * i + 1 < size) {
            int left = 2 * i + 1;
            int right = 2 * i + 2;


            int largest = left;

            if (right < size && heap[right] > heap[left]) {
                largest = right;
            }

            if (heap[i] < heap[largest]) {
                int temp = heap[i];
                heap[i] = heap[largest];
                heap[largest] = temp;
                i = largest;
            } else {
                break;
            }
        }
    }

    public void sort() {
        int originalSize = size;
        for (int i = originalSize - 1; i > 0; i--) {
            int temp = heap[0];
            heap[0] = heap[i];
            heap[i] = temp;
            size--;
            sink(0);
        }
        size = originalSize;
    }

    public int[] get() {
        return heap;
    }
}

public class Task4 {
    public static int[] arrange(int[] nums, int k) {


        MaxHeap heap = new MaxHeap(nums.length);

        for (int i = 0; i < nums.length; i++) {
            heap.insert(nums[i]);
        }

        int[] result = new int[k];
        for (int i = 0; i < k; i++) {
            result[i] = heap.extractMax();
        }

        return result;
    }
