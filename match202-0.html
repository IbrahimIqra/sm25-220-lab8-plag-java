public class Assignment8_task3
{



	public static int[] distribute(int[]tasks, int m)
	{

		MinHeap heap = new MinHeap(m);
		
		for (int i=0; i<m; i++)
		{
			heap.insert(0);
		}


		for (int i=0; i<tasks.length; i++) 
		{
			int min_load = heap.extractMin();
			int new_load = min_load + tasks[i];


			heap.insert(new_load);
		}
		return heap.getHeap();
	}

	public static void main(String[] args) 
	{
		int[] tasks = {2, 4, 7, 1, 6};
		int m = 4;
		int[] res = distribute(tasks, m);

		System.out.print("Final machine loads: [");
		for (int i = 0; i < res.length; i++) 
		{
			System.out.print(res[i]);
			
			if (i<res.length-1) 
			{
			  System.out.print(", ");
			}
		}


		System.out.println("]");
	}
}

public class Assignment8_task4 
{
	public static int[] top_k(int[]nums, int k) 
	{
		MaxHeap maxHeap = new MaxHeap(nums.length);

    for (int i=0; i<nums.length; i++) 
    {
        maxHeap.insert(nums[i]);
    }


		int[] res = new int[k];


		for (int i=0; i<k; i++) 
		{
			res[i] = maxHeap.extractMax();
		}
		return res;
	}

	public static void main(String[] args) 
	{
		int[] nums = {4, 10, 2, 8, 6, 7};
		int k = 3;
		int[] res = top_k(nums, k);

		System.out.print("[");
		for (int i=0; i<res.length; i++)
		{
			System.out.print(res[i]);
			
			if (i<res.length - 1) 
			{
			  System.out.print(", ");
			}
		}
		System.out.println("]");
	}
}
