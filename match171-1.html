public class Task3 {
    public static int[] Tasks(int[] tasks, int m) {
        MinHeap heap = new MinHeap(m);

        for (int i = 0; i < m; i++) {
            heap.insert(0);
        }

        for (int i = 0; i < tasks.length; i++) {
            int task = tasks[i];
            int min = heap.extractMin();

            heap.insert(min + task);
        }

        int[] result = new int[m];


        for (int i = 0; i < m; i++) {
            result[i] = heap.extractMin();
        }

        return result;
    }

    public static void main(String[] args) {
        int[] tasks = {2, 4, 7, 1, 6};
        int m = 4;

        int[] loads = Tasks(tasks, m);

        System.out.print("[");
        for (int i = 0; i < loads.length; i++) {
            System.out.print(loads[i]);
            if (i != loads.length - 1) {
                System.out.print(", ");
            }
        }
        System.out.println("]");
    }
}

public class Task4 {
    public static int[] TopK(int[] num, int k) {
        MaxHeap heap = new MaxHeap(num.length);

        for (int i = 0; i < num.length; i++) {
            heap.insert(num[i]);
        }

        int[] result = new int[k];
        for (int i = 0; i < k; i++) {
            result[i] = heap.extractMax();
        }

        return result;
    }

    public static void main(String[] args) {
        int[] nums = {4, 10, 2, 8, 6, 7};
        int k = 3;

        int[] topK = TopK(nums, k);

        System.out.print("[");
        for (int i = 0; i < topK.length; i++) {
            System.out.print(topK[i]);
            if (i != topK.length - 1) {
                System.out.print(", ");
            }
        }
        System.out.println("]");
    }
}
