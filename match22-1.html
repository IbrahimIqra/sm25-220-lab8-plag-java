public class Task03{
    public static class MinHeap {
        private int [] array;
        private int size;
    
        public MinHeap(int capacity){
        array = new int [capacity];
        }



        public void insert (int data){
            array[size] = data;
            int indx = size;
            int idx = (indx-1)/2;
            swim(indx, idx);
            size++;
        }

        private void swim(int indx, int idx){
            while (indx>0 && array[indx] < array[idx]) {
                swap(indx,idx);
                indx = idx;
                idx = (indx-1)/2;
            }
        }
        private void swap(int a, int b){
            int temp = array[a];
            array[a] = array[b];
            array[b] = temp;
        }
        public int extractMin(){
            int store = array[0];
            swap(0, size-1);
            size--;
            sink(0, size);
            return store;
        }
        private void sink(int k, int m){
            int left = 2*k+1;
            int right = 2*k+2;
            int minIndex = k;
       
            if (left<m && array[minIndex] > array[left]) {
                minIndex = left;
            }
            if (right<m && array[minIndex] > array[right]) {
                minIndex = right;
            }
            if (minIndex != k) {
                swap(k, minIndex);
                sink(minIndex, m);
            }
        }

        public boolean isEmpty() {
            return size == 0;
        }
    }


    public static void heapInsert (int [] array, int m){
        MinHeap heap = new MinHeap(m);
        for (int i = 0; i < m; i++) {
             heap.insert(array[i]);
        }
        int indx = m;
        while (heap.isEmpty() == false){
            int v = heap.extractMin();
            heap.insert(v + array[indx]);
            indx++;

            if(indx == array.length){
                break;
            }
        }
        if (heap.isEmpty() == false){
            for (int i = 0; i < array.length-1; i++) {
                System.out.print(heap.extractMin()+" ");
            }
            System.out.println();
        }
    }


    public static void main (String []args){
        int[] array = {2,4,7,1,6};
        int m = 4;
        heapInsert(array, m);
    }
}

        int store = array[0];
        swap(0, size-1);
        size--;
        sink(0, size);

    }
    private void sink(int k, int m){

        if (left<m && array[minIndex] < array[left]) {
            minIndex = left;
        }

        }

        }
    }

    public boolean isEmpty() {

}


    public static void largestElement(int [] num, int k){
