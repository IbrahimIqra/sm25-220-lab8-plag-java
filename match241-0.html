//MinHeap


import java.util.Arrays;
public class MinHeap {
    private int[] heap;
    private int size;
    private int capacity;
    public MinHeap(int capacity){
        this.capacity=capacity;
        this.size=0;
        this.heap=new int[this.capacity];
    }
    private int parent(int i){
        return (i-1)/2;
    }
    private int leftChild(int i){
        return 2*i+1;
    }
    private int rightChild(int i){
        return 2*i+2;
    }
    public void swim(int curr){
        while(curr > 0&&heap[curr]<heap[parent(curr)]){
            swap(curr,parent(curr));
            curr=parent(curr);
        }
    }
    public void insert(int elem){
        if (size >= capacity) {
            System.out.println("Heap is full. Cannot insert.");
            return;
        }
        heap[size]=elem;
        size++;
        swim(size-1);
    }
    private void swap(int x, int y){
        int temp=heap[x];
        heap[x]=heap[y];
        heap[y]=temp;
    }
    public int extractMin(){
        int pop=heap[0];
        heap[0]=heap[--size];
        sink(0);
        return pop;
    }
    public void sink(int idx){
        int minidx=idx;
        while (true) {
            int lidx=leftChild(idx);
            int ridx=rightChild(idx);


            if(lidx<size && heap[lidx] < heap[minidx]){
                minidx=lidx;
            }
            if(ridx<size && heap[ridx] < heap[minidx]){
                minidx=ridx;
            }
            if(idx!=minidx){
                swap(idx,minidx);
                idx=minidx;
            }
            else{
                break;
            }
        }
    }
    public int[] sort() {


        int[] sortedArray = new int[size];
        int originalSize = size;
        for (int i = 0; i < originalSize; i++) {
            sortedArray[i] = extractMin();
        }
        // Restore the heap state after sorting
        this.size = originalSize;
        return sortedArray;
    }
    public void printHeap() {
        System.out.println(Arrays.toString(Arrays.copyOf(heap, size)));
    }
}

import java.util.Arrays;
public class MaxHeap {
    private int[] heap;
    private int size;
    private int capacity;
    public MaxHeap(int capacity){
        this.capacity=capacity;
        this.size=0;
        this.heap=new int[this.capacity];
    }
    private int parent(int i){
        return (i-1)/2;
    }
    private int leftChild(int i){
        return 2*i+1;
    }
    private int rightChild(int i){
        return 2*i+2;
    }
    public int[] getHeapArray() {
        int[] result = Arrays.copyOf(heap, size);
        Arrays.sort(result); // Sorts to match the sample output format
        return result;
    }
    public void swim(int curr){
        while(curr > 0 && heap[curr]>heap[parent(curr)]){
            swap(curr,parent(curr));
            curr=parent(curr);
        }
    }
    public void insert(int elem){
        if (size >= capacity) {
            System.out.println("Heap is full. Cannot insert.");
            return;
        }
        heap[size]=elem;
        size++;
        swim(size-1);
    }


    private void swap(int x, int y){
        int temp=heap[x];
        heap[x]=heap[y];
        heap[y]=temp;
    }
    public int extractMax(){
        int pop=heap[0];
        heap[0]=heap[--size];
        sink(0);
        return pop;
    }
    public void sink(int idx){
        int maxidx =idx;
        while (true) {
            int lidx=leftChild(idx);


            int ridx=rightChild(idx);
            if(lidx<size && heap[lidx] > heap[maxidx]){
                maxidx =lidx;
            }
            if(ridx<size && heap[ridx] > heap[maxidx]){
                maxidx =ridx;
            }
            if(idx!= maxidx){
                swap(idx, maxidx);
                idx= maxidx;
            }
            else{
                break;
            }
        }
    }
    private void sort(){
        int osize=this.size;
        for (int i = osize-1; i > 0; i--) {
            swap(0,i);
            this.size--;
            sink(0);
        }
        this.size=osize;
    }
    public void printHeap() {
        System.out.println(Arrays.toString(Arrays.copyOf(heap, size)));
    }
}

import java.util.Arrays;

public class Task_3Scheduler {


    public static void scheduler(int arr[],int m){
        MinHeap load=new MinHeap(m);
        for (int i = 0; i < m; i++) {
            load.insert(0);
        }
        for (int tasks : arr) {
            int min= load.extractMin();
            int sum= min+tasks;
            load.insert(sum);
        }
        System.out.println("Final loads: "+Arrays.toString(load.sort()));
    }
    public static void main(String[]args){
        int[] arr={2,4,7,1,6};
        int m=4;
        scheduler(arr,m);
    }
}
