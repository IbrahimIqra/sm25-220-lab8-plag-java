class MinHeap{
  private int[]h;
  private int lng;
  private int c;
  MinHeap(int cc){
    this.h=new int[cc];
    this.c=cc;
  }


  public int removeMin(int cc){
    int t=h[0];
    h[0]=h[lng-1];
    this.h[this.lng-1]=cc;
    heapify(0);
    return t;
  }
  public void heapify(int idx){
    int l=2*idx+1;
    int r=2*idx+2;
    if(l>=this.lng){
      return;
    }
    int s=l;
    if(r<this.lng&&this.h[s]>this.h[r]){
      s=r;
    }
    if(this.h[idx]>this.h[s]){
      int t=this.h[idx];
      this.h[idx]=this.h[s];
      this.h[s]=t;
      heapify(s);
    }
  }
  public void insert(int cc){
    if(this.lng>=this.c){
      cc=cc+h[0];
      removeMin(cc);
    }
    else{
      h[this.lng]=cc;
      swim(this.lng);
      this.lng++;
    }
  }
  public void swim(int n){
    int p=(n-1)/2;
    while(n>0&&h[p]>h[n]){
      int t=h[p];
      h[p]=h[n];
      h[n]=t;
      n=p;
      p=(n-1)/2;
    }
  }
  public static int[] nt(int tt[],int m){
    MinHeap heap=new MinHeap(m);
    for(int i=0; i<tt.length; i++){
      int c=heap.removeMin(m);
      int ins=c+tt[i];
      heap.insert(ins);
    }
    return heap.getheap();
  }
  public static void main(String[]args){
    int[]t={2,4,7,1,6};
    int m=4;
    int r[]=nt(t,m);
    for(int i=0; i<r.length; i++){
      System.out.print(r[i]+" ");
    }
    System.out.println();
  }
  public int[]getheap(){
    return h;
  }
  
}
