public class Task3 {
  static class MinHeap {
    private int[] data;
    private int lngth;
    public MinHeap(int capacity) {
      data= new int[capacity];
      lngth = 0;}
    public void insert(int elem) {
      if (lngth == data.length) throw new IllegalStateException("MinHeap overflow");
      data[lngth] =elem;
      swim(lngth);
      lngth++;}
    
    public int extractMin() {
      if(lngth== 0) throw new IllegalStateException("MinHeap underflow");


      int min = data[0];
      data[0] =data[lngth -1];
      lngth--;
      sink(0);
      return min;}


    
    public int[] toArray() {
      int[] out= new int[lngth];
      for (int i =0; i <lngth; i++)out[i] =data[i];
      return out;}
    
    private void swim(int k) {
      while(k > 0) {
      int p = (k - 1) / 2;
      if (data[k] < data[p]) {
      int temp = data[k]; data[k] =data[p];data[p] = temp;
          k= p;
        } else break; }}
    
    private void sink(int k) {
      while (true) {
        int lft = 2*k + 1;
        if (lft >= lngth) break;


        int rit =lft + 1;
        int smallest= lft;
        if (rit< lngth && data[rit] < data[lft]) smallest= rit;
        if (data[k] <=data[smallest]) break;
        int temp= data[k]; data[k] = data[smallest]; data[smallest]= temp;
        k = smallest;
      }}}
  


  public static int[] distributeTasks(int[] tasks, int m) { //Task 3 Function
    MinHeap heap = new MinHeap(m);
    for (int i =0;i <m; i++) heap.insert(0);  
    for (int i =0; i < tasks.length; i++) {
      int currentLoad = heap.extractMin();
      currentLoad +=tasks[i];
      heap.insert(currentLoad);}
    return heap.toArray();}
  private static void sortArray(int[] data) {
    for (int i = 0;i < data.length - 1; i++) {
      for (int j = 0; j<data.length - i - 1; j++) {
        if (data[j] > data[j + 1]) {
          int temp = data[j];
          data[j] =data[j + 1];
          data[j + 1] =temp;
        }}}}
  


  public static void main(String[] args) {  // Tester Code 
    
    int[]tasks = {2, 4, 7, 1, 6};
    int m = 4;
    int[] loads = distributeTasks(tasks, m);
    sortArray(loads);
    print("Task3 output: ", loads);  }

public class Task4 {
  static class MaxHeap {
    private int[] data;
    private int lngth;
    public MaxHeap(int capacity) {
      data = new int[capacity];
      lngth= 0;}
    public void insert(int elem) {
      if (lngth ==data.length) throw new IllegalStateException("MaxHeap overflow");
      data[lngth] = elem;
      swim(lngth);
      lngth++;}
    public int extractMax(){
      if (lngth == 0) throw new IllegalStateException("MaxHeap underflow");


      int max = data[0];
      data[0]= data[lngth -1];
      lngth--;
      sink(0);
      return max;  }


    private void swim(int k) {
      while (k > 0) {
        int p = (k - 1) / 2;
        if (data[k] > data[p]) {
          int temp = data[k]; data[k] = data[p]; data[p] = temp;
          k = p;
        } else break;
      }}
    private void sink(int k) {
      while(true) {
        int lft = 2 * k+ 1;
        if (lft>= lngth) break;
        int rit = lft +1;
        int biggest =lft;
        if (rit < lngth &&data[rit] > data[lft]) biggest =rit;
        if (data[k] >= data[biggest])break;
        int temp = data[k]; data[k]= data[biggest];data[biggest] = temp;
        k = biggest;
      }}}
  public static int[] topK(int[] nums,int k){   // Task 4 Function 
    if (k <0 || k >nums.length) throw new IllegalArgumentException("Invalid k");


    MaxHeap heap = new MaxHeap(nums.length);
    for (int i= 0; i <nums.length; i++) heap.insert(nums[i]);
    int[] result = new int[k];
    for (int i =0; i< k;i++) result[i] =heap.extractMax();
    return result;
  }

  }
