class MaxHeap {


    private int[] heap;
    private int count;

    public MaxHeap(int capacity) {
        heap = new int[capacity];
        count = 0;
    }
    private void expand() {
        if (count >= heap.length) {   // changed == to >=
            int[] bigger = new int[heap.length + 1];
            System.arraycopy(heap, 0, bigger, 0, heap.length);
            heap = bigger;
        }
    }
    private void swap(int a, int b) {
        int tmp = heap[a];
        heap[a] = heap[b];
        heap[b] = tmp;
    }
    public boolean empty() {
        return count <= 0;   // changed ==0 to <=0 (same effect)
    }
    public void add(int val) {
        expand();
        heap[count] = val;
        int child = count;
        int parent = (child - 1) / 2;
        rise(child, parent);
        count++;
    }
    private void rise(int child, int parent) {
        while (child > 0 && heap[child] > heap[parent]) {
            swap(child, parent);
            child = parent;
            parent = (child - 1) / 2;
        }
    }
    public int top() {
        return heap[0];
    }
    public int removeMax() {
        int res = heap[0];
        swap(0, count - 1);
        count--;
        sink(0, count);
        return res;
    }
    private void sink(int i, int n) {


        int left = 2 * i + 1, right = 2 * i + 2;
        int big = i;

        if (left < n && heap[left] > heap[big]) big = left;
        if (right < n && heap[right] > heap[big]) big = right;

        if (big != i) {
            swap(i, big);
            sink(big, n);
        }
    }


    public void sort() {
        for (int i = count / 2 - 1; i >= 0; i--) {
            sink(i, count);
        }
        for (int i = count - 1; i > 0; i--) {
            swap(0, i);
            sink(0, i);
        }
    }
    public void print() {
        for (int i = 0; i < count; i++) {
            System.out.print(heap[i] + " ");
        }
    }
}


public class PracticeHeap {
    public static void main(String[] args) {
        int[] numbers = {4, 10, 2, 8, 6, 7};
        int k = 3;
        findTopK(numbers, k);
    }
    public static void findTopK(int[] data, int k) {
        MaxHeap pq = new MaxHeap(0);
        for (int i = 0; i < data.length; i++) {
            pq.add(data[i]);
        }

        int[] arr = new int[k];
        int idx = 0;

        while (!pq.empty() && idx < k) {
            arr[idx] = pq.removeMax();
            idx++;
        }

        for (int i = 0; i < arr.length; i++) {
            System.out.print(arr[i] + " ");
        }
    }
}

class MinHeap {


    private int[] heap;
    private int count;

    public MinHeap(int cap) {
        heap = new int[cap];
        count = 0;
    }
    private void grow() {
        if (count >= heap.length) {
            int[] temp = new int[heap.length + 1];
            System.arraycopy(heap, 0, temp, 0, heap.length);
            heap = temp;
        }
    }
    private void swap(int a, int b) {


        int t = heap[a];
        heap[a] = heap[b];
        heap[b] = t;
    }
    public boolean empty() {
        return count == 0;
    }
    public void add(int val) {
        grow();
        heap[count] = val;
        int child = count;
        int parent = (child - 1) / 2;
        upHeap(child, parent);
        count++;
    }
    private void upHeap(int child, int parent) {
        while (child > 0 && heap[child] < heap[parent]) {
            swap(child, parent);
            child = parent;
            parent = (child - 1) / 2;
        }
    }
    public int top() {
        return heap[0];
    }
    private void downHeap(int i, int n) {
        int left = 2 * i + 1, right = 2 * i + 2;
        int min = i;

        if (left < n && heap[left] < heap[min]) min = left;
        if (right < n && heap[right] < heap[min]) min = right;

        if (min != i) {
            swap(i, min);


            downHeap(min, n);
        }
    }
    public int removeMin() {
        int res = heap[0];
        swap(0, count - 1);
        count--;
        downHeap(0, count);
        return res;
    }
}
public class PracticeHeap {


    public static void main(String[] args) {
        int[] jobs = {2, 4, 7, 1, 6};
        int workers = 4;
        distribute(jobs, workers);
    }
    public static void distribute(int[] jobs, int workers) {
        MinHeap pq = new MinHeap(workers);
        for (int i = 0; i < workers; i++) {
            pq.add(0);
        }
        int idx = 0;
        while (!pq.empty()) {
            int load = pq.removeMin();
            pq.add(load + jobs[idx]);
            idx++;
            if (idx >= jobs.length) {
                break;
            }
        }
        while (!pq.empty()) {
            System.out.println(pq.removeMin());
        }
    }
