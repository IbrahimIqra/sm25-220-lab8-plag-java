public class TimeProcess{
  public static int[] taskStep(int[] tasks,int m){
    if(tasks==null||tasks.length==0||m<=0){
      return new int[0];
    }
    
    MinHeap heap=new MinHeap(m);
    for(int i=0;i<m;i++){
      heap.insert(0);
    }
    
    for(int i=0;i<tasks.length;i++){
      int task=tasks[i];
      int minLoad=heap.extractMin();
      heap.insert(minLoad+task);
    }
    
    return heap.getSrtArray();
  }



class MinHeap {
  private int[] heap;
  private int size;
  private int cap;
  
  public MinHeap(int cap){
    this.cap=cap;
    this.heap=new int[cap];
    this.size=0;
  }
  
  public void insert(int value){
    if(size==cap){
      rsz();
    }
    heap[size]=value;
    swim(size);
    size++;
  }
  
  public int extractMin(){
    if(size==0){
      return -1;
    }
    int min=heap[0];
    heap[0]=heap[size-1];
    size--;
    sink(0);
    return min;
  }
  
  private void swim(int ix){
    int prnt=(ix-1)/2;
    while(ix>0 && heap[ix]<heap[prnt]){
      swap(ix,prnt);
      ix=prnt;
      prnt=(ix-1)/2;
    }
  }
  
  private void sink(int ix){
    int lftChld=2*ix+1;
    int rghtChld=2*ix+2;
    int smlst=ix;
    


    if(lftChld<size && heap[lftChld]<heap[smlst]){
      smlst=lftChld;
    }
    if(rghtChld<size && heap[rghtChld]<heap[smlst]){
      smlst=rghtChld;
    }
    if(smlst!=ix){
      swap(ix,smlst);
      sink(smlst);
    }
  }
  
  private void swap(int i,int j){
    int temp=heap[i];
    heap[i]=heap[j];
    heap[j]=temp;
  }
  
  private void rsz(){
    cap*=2;
    int[] newHeap=new int[cap];
    for(int i=0;i<size;i++){
      newHeap[i]=heap[i];
    }
    heap=newHeap;
  }
  
  public int[] getHeapArray(){
    int[] rslt=new int[size];
    for(int i=0;i<size;i++){
      rslt[i]=heap[i];
    }
    return rslt;
  }
  public int[] getSrtArray(){
    int[] rslt=getHeapArray();
    java.util.Arrays.sort(rslt);
    return rslt;
  }
  


  public void sort(){
    int orgSize=size;
    for(int i=size-1;i>0;i--){
      swap(0,i);
      size--;
      sink(0);
    }
    size=orgSize;
  }
}

public class FindLargest{
  public static int[] findingLrgst(int[] nums,int k){
    if(nums==null||nums.length==0||k<=0||k>nums.length){
      return new int[0];
    }
    


    MaxHeap heap=new MaxHeap(nums.length);
    for(int i=0;i<nums.length;i++){
      heap.insert(nums[i]);
    }
    
    int[] rslt=new int[k];
    for(int i=0;i<k;i++){
      rslt[i]=heap.fndMax();
    }
    
    return rslt;
  }
}
