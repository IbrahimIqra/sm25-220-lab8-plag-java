//TASK03



public class task03{
    public static int[] heapArray(int[] a,int m){
        int[] heap=new int[m];
        
        for(int i=0;i<m;i++){
            heap[i]=0;
        }
        for(int i=0;i<a.length;i++){
            int b=a[i];
            
        int minLoad=heap[0];
        heap[0]=minLoad+b;
        heapify(heap,0,m);
    }
    
        return heap;
    }

private static void heapify(int[] heap,int i,int size){
    int smallest=i;
    int left=2*i+1;
    int right=2*i+2;

    if(left<size && heap[left]<heap[smallest]){
        smallest=left;
    }
    
    if(right<size && heap[right]<heap[smallest]){
        smallest=right;
    }
    
    if(smallest!=i){
        int temp=heap[i];
        heap[i]=heap[smallest];
        heap[smallest]=temp;
        heapify(heap,smallest,size);
    }
}



public static void main(String[]args){
        int[] a={2,4,7,1,6};
        int m=4;
        int[] res=heapArray(a,m);
        
        System.out.print("[");
        for(int i=0;i<res.length;i++){
            System.out.print(res[i]);
            
            if(i<res.length-1){
                System.out.print(", ");
            }
        }
        
        System.out.print("]");
    }
}

public class task04{


    public static int[] findK(int[] a,int k){
        int n=a.length;
        
        buildMaxHeap(a,n);
        
        int[] res=new int[k];
        
        for(int i=0;i<k;i++){
            res[i]=a[0];
            a[0]=a[n-1-i];
            heapify(a,0,n-1-i);
        }
        
        return res;
    }
    
    private static void buildMaxHeap(int[] a,int n){
        for(int i=n/2;i>=0;i--){
            heapify(a,i,n);
        }
    }
    
    private static void heapify(int[] a,int i, int size){
        int largest=i;


        int left=2*i+1;
        int right=2*i+2;
        
        if(left<size && a[left]>a[largest]){
        largest=left;
    }
    
    if(right<size && a[right]>a[largest]){
        largest=right;
    }
    
    if(largest!=i){
        int temp=a[i];
        a[i]=a[largest];
        a[largest]=temp;
        heapify(a,largest,size);
    }
}
    


    public static void main(String[]args){
        int[] x={4,10,2,8,6,7};
        int k=3;
        int[] res=findK(x,k);
        
        System.out.print("[");
        for(int i=0;i<res.length;i++){
            System.out.print(res[i]);
            
            if(i<res.length-1){
                System.out.print(", ");
            }
        }
        
        System.out.print("]");
    }
}
