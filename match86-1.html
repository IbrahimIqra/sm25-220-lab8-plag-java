import java.util.Scanner;
public class Main {
    private int[] heap;
    private int size;
    private int capacity;

    public Main(int capacity) {
        this.capacity = capacity;
        heap = new int[capacity];


        size = 0;
    }
    
    public void insert(int value) {
        if (size == capacity) {
            System.out.println("Heap is full");
            return;
        }
        heap[size] = value;
        swim(size);
        size++;
    }

    private void swim(int index) {
        while (index > 0 && heap[index] < heap[parent(index)]) {
            swap(index, parent(index));
            index = parent(index);
        }
    }

    private int parent(int index) {
        return (index - 1) / 2;
    }



    public int deleteMin() {
        if (size == 0) {
            throw new IllegalStateException("Heap is empty");
        }
        int min = heap[0];
        swap(0, size - 1);
        size--;
        heapify(0);
        return min;
    }



    private void heapify(int index) {
        int smallest = index; 
        int left = 2 * index + 1;
        int right = 2 * index + 2;
    
        if (left < size && heap[left] < heap[smallest])  
            smallest = left;
        if (right < size && heap[right] < heap[smallest]) 
            smallest = right;
    
        if (smallest != index) {
            swap(index, smallest);
            heapify(smallest);
        }
    }


    private void swap(int i, int j) {
        int temp = heap[i];
        heap[i] = heap[j];
        heap[j] = temp;
    }

    public static void main(String[] args) {
        int tasks[] = {2, 4, 7, 1, 6};
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter m: ");


        int m = sc.nextInt();
        Main minHeap = new Main(m);
        for(int i = 0; i < m; i++){
            minHeap.insert(0);
        }
        int x[] = process(minHeap, tasks, m);
        for(int i = 0; i < x.length; i++){
            System.out.print(x[i] + " ");
        }
    }
    public static int[] process(Main MinHeap, int[] tasks, int m){
        for(int i = 0; i < tasks.length; i++){
            int ft = tasks[i];
            int dmin = MinHeap.deleteMin();
            MinHeap.insert(ft+dmin);
        }
        int arr[] = new int[m];
        for(int i = 0; i < m; i++){
            arr[i] = MinHeap.deleteMin();
        }
        return arr;
    }
}

import java.util.Scanner;
public class Main {
    private int[] heap;
    private int size;
    private int capacity;

    public Main(int capacity) {
        this.capacity = capacity;
        heap = new int[capacity];


        size = 0;
    }

    public void insert(int value) {
        if (size == capacity) {
            System.out.println("Heap is full");
            return;
        }
        heap[size] = value;
        swim(size);
        size++;
    }
    private void swim(int index) {
        while (index > 0 && heap[index] > heap[parent(index)]) {
            swap(index, parent(index));
            index = parent(index);
        }
    }

    private int parent(int index) {
        return (index - 1) / 2;
    }


    public int deleteMax() {
        if (size == 0) {
            throw new IllegalStateException("Heap is empty");
        }
        int max = heap[0];
        swap(0, size - 1);
        size--;
        heapify(0);
        return max;
    }


    private void heapify(int index) {
        int largest = index;
        int left = 2 * index + 1;
        int right = 2 * index + 2;

        if (left < size && heap[left] > heap[largest])
            largest = left;
        if (right < size && heap[right] > heap[largest])
            largest = right;

        if (largest != index) {
            swap(index, largest);
            heapify(largest);
        }
    }
    private void swap(int i, int j) {
        int temp = heap[i];
        heap[i] = heap[j];
        heap[j] = temp;
    }


    public static void main(String[] args) {
        int nums[] = {4, 10, 2, 8, 6, 7};


        Scanner sc = new Scanner(System.in);
        Main maxHeap = new Main(nums.length);
        
        for(int i = 0; i < nums.length; i++){
            maxHeap.insert(nums[i]);
        }
        
        System.out.print("Enter k: " );
        int k = sc.nextInt();
        while(k < 0 || k > maxHeap.size){
            System.out.print("Please enter a valid k: ");
            k = sc.nextInt();
        }
        
        int x[] = process(maxHeap, k);
        for(int i = 0; i < x.length; i++){
            System.out.print(x[i] + " ");
        }
    }
    public static int[] process(Main maxHeap, int k){
        int arr[] = new int[k];
        for(int i = 0; i < k; i++){
            arr[i] = maxHeap.deleteMax();
        }
        return arr;
    }
}
