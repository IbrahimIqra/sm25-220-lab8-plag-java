public class Task4 {


   
    static class MinHeap {
        private int[] heap;
        private int size;
        private int capacity;


        MinHeap(int cap) {
            this.capacity = cap;
            this.heap = new int[cap];
            this.size = 0;
        }


        public void insert(int val) {
            if (size >= capacity) {
                System.out.println("Heap is full");
                return;
            }
            heap[size] = val;
            swim(size);
            size++;
        }


        public int extractMin() {
            if (size <= 0) {


                System.out.println("Heap is empty");
                return -1;
            }
            int root = heap[0];
            heap[0] = heap[size - 1];
            size--;
            sink(0);
            return root;
        }


        private void swim(int i) {
            while (i > 0 && heap[parent(i)] > heap[i]) {
                swap(i, parent(i));
                i = parent(i);
            }
        }


        private void sink(int i) {
            while (true) {
                int left = leftChild(i);
                int right = rightChild(i);


                int smallest = i;
          if (left < size && heap[left] < heap[smallest]) smallest = left;
       if (right < size && heap[right] < heap[smallest]) smallest = right;
                if (smallest == i) break;
                swap(i, smallest);
                i = smallest;
            }
        }


        private int parent(int i) { return (i - 1) / 2; }
        private int leftChild(int i) { return 2 * i + 1; }
        private int rightChild(int i) { return 2 * i + 2; }


        private void swap(int i, int j) {
            int t = heap[i];
            heap[i] = heap[j];
            heap[j] = t;
        }
    }


   


    public static int[] distributeTasks(int[] tasks, int m) {
        MinHeap minHeap = new MinHeap(m);


        for (int i = 0; i < m; i++) {
            minHeap.insert(0);
        }


       for (int i = 0; i < tasks.length; i++) {
    int smallestLoad = minHeap.extractMin();
    int newLoad = smallestLoad + tasks[i];
    minHeap.insert(newLoad);


}




        int[] result = new int[m];
        for (int i = 0; i < m; i++) {
            result[i] = minHeap.extractMin();
        }


        return result;
    }

    public static void main(String[] args) {
        int[] tasks = {2, 4, 7, 1, 6};

public class Task5 {


static class MaxHeap {
    private int[] heap;
    private int size;
    private int capacity;


    MaxHeap(int cap) {
        this.capacity = cap;
        this.heap = new int[cap];
        this.size = 0;
    }


    public void insert(int val) {
        if (size >= capacity) {
            System.out.println("Heap is full");
            return;
        }
        heap[size] = val;
        swim(size);
        size = size + 1;
    }


    public int extractMax() {
        if (size <= 0) {


            System.out.println("Heap is empty");
            return -1;
        }
        int root = heap[0];
        heap[0] = heap[size - 1];
        size = size - 1;
        sink(0);
        return root;
    }


    private void swim(int i) {
        while (i > 0 && heap[parent(i)] < heap[i]) {
            swap(i, parent(i));
            i = parent(i);
        }
    }


    private void sink(int i) {
        while (true) {
            int left = leftChild(i);
            int right = rightChild(i);


            int largest = i;


          if (left < size && heap[left] > heap[largest]) largest = left;
         if (right < size && heap[right] > heap[largest]) largest = right;
            if (largest == i) break;


            swap(i, largest);
            i = largest;
        }
    }


    private int parent(int i) { return (i - 1) / 2; }
    private int leftChild(int i) { return 2 * i + 1; }
    private int rightChild(int i) { return 2 * i + 2; }




    private void swap(int i, int j) {
        int t = heap[i];
        heap[i] = heap[j];
        heap[j] = t;
    }
}




    public static int[] findLargest(int[] nums, int k) {


        MaxHeap maxHeap = new MaxHeap(nums.length);


        for (int i = 0; i < nums.length; i++) {
            maxHeap.insert(nums[i]);
        }


        int[] largest = new int[k];
        for (int i = 0; i < k; i++) {
            largest[i] = maxHeap.extractMax();
        }


        return largest;
    }


    //Driver code:




    public static void main(String[] args) {
        int[] nums = {4, 10, 2, 8, 6, 7};
        int k = 3;


        int[] result = findLargest(nums, k);


        System.out.println(java.util.Arrays.toString(result));
    }
