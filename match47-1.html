class minHeap {
    public int[] arr;
    public int size;
    
    public minHeap(int capacity)
    {
        arr = new int[capacity];
        size = 0;
    }
    
    public boolean isEmpty()
    {
        return size == 0;
    }
    
    public void add(int value)
    {
        arr[size] = value;
        int child = size;
        int parent = (child - 1) / 2;
        
        while (child > 0 && arr[child] < arr[parent])
        {
            int temp = arr[child];
            arr[child] = arr[parent];
            arr[parent] = temp;
            
            child = parent;
            parent = (child - 1) / 2;
        }
        size++;
    }
    
    public int removeMin()
    {
        int minValue = arr[0];
        arr[0] = arr[size - 1];
        size--;
        
        int current = 0;
        while (true) {


            int left = 2 * current + 1;
            int right = 2 * current + 2;
            int smallest = current;
            
            if (left < size && arr[left] < arr[smallest])
            {
                smallest = left;
            }
            if (right < size && arr[right] < arr[smallest]) 
            {
                smallest = right;
            }
            
            if (smallest != current)
            {
                int temp = arr[current];
                arr[current] = arr[smallest];
                arr[smallest] = temp;
                current = smallest;
            } 
            else
            {
                break;
            }
        }
        
        return minValue;
    }
    
    public int peekMin()
    {
        return arr[0];
    }
}






    public static void main(String[] args) {
        int[] tasks = {2, 4, 7, 1, 6};
        int workers = 4;
        
        minHeap heap = new minHeap(workers);
        
        for (int i = 0; i < workers; i++) {
            heap.add(0);
        }
        
        for (int i = 0; i < tasks.length; i++)
        {
            int earliestTime = heap.removeMin();
            heap.add(earliestTime + tasks[i]);
        }
        
        while (!heap.isEmpty())
        {
            System.out.print(heap.removeMin() + " ");
        }
    }
}

class MaxHeap {


    public int[] heap;
    public int count;
    
    public MaxHeap(int n)
    {
        heap = new int[n];
        count = 0;
    }
    
    public void makeBigger()
    {
        if (count == heap.length)
        {
            int[] newHeap = new int[heap.length + 1];
            System.arraycopy(heap, 0, newHeap, 0, heap.length);
            heap = newHeap;
        }
    }
    
    public void swap(int i, int j)
    {
        int temp = heap[i];
        heap[i] = heap[j];


        heap[j] = temp;
    }
    
    public boolean isEmpty()
    {
        return count == 0;
    }
    
    public void add(int value)
    {
        makeBigger();
        heap[count] = value;
        int child = count;
        int parent = (child - 1) / 2;
        


        while (child > 0 && heap[child] > heap[parent])
        {
            swap(child, parent);
            child = parent;
            parent = (child - 1) / 2;
        }


        count++;
    }
    
    public void sinkDown(int index, int n)
    {
        int left = 2 * index + 1;
        int right = 2 * index + 2;
        int largest = index;
        
        if (left < n && heap[largest] < heap[left])
        {
            largest = left;
        }
        if (right < n && heap[largest] < heap[right])
        {


            largest = right;
        }
        if (largest != index)
        {
            swap(index, largest);
            sinkDown(largest, n);
        }
    }
    
    public int removeMax()
    {
        int maxValue = heap[0];
        swap(0, count - 1);
        count--;
        sinkDown(0, count);
        return maxValue;
    }

            heap.add(nums[i]);
