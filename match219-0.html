public class Task03 
{


  public static int[] taskDis(int[] tasks, int m) 
  {
    MinHeap x = new MinHeap(m);
    
    for (int i=0; i<m; i++) 
    {
      x.insert(0);
    }
    
    for (int i=0; i<tasks.length; i++) 
    {
      int temp = x.extractMin();
      x.insert(temp + tasks[i]);
    }
    
    int[] result = new int[m];


    for (int i=0; i<m; i++)
    {
      result[i] = x.extractMin();
    }
    return result;
  }
  
  public static void print(int[] arr) 
  {
    System.out.print("[");
    for (int i = 0; i<arr.length; i++)
    {
      System.out.print(arr[i]);
      if (i<arr.length-1) 
      {
        System.out.print(", ");
      }
    }
    System.out.println("]");
  }
}


public class Task03Tester 
{


  public static void main(String[] args)
  {
    int[] tasks = {2, 4, 7, 1, 6};
    int m = 4;
    int[] output = Task03.taskDis(tasks, m);
    Task03.print(output);
  }
}

public class Task04
{


  public static int[] topVal(int[] arr, int k)
  {
    MaxHeap x = new MaxHeap(arr.length);
    
    for (int i = 0; i < arr.length; i++)
    {
      x.insert(arr[i]);
    }
    
    int[] sortedArr = x.sort(); 
    int[] result = new int[k];
    
    for (int i = 0; i < k; i++)
    {
      result[i] = sortedArr[i];
    }    


    return result;
  }
  
  public static void print(int[] arr) 
  {
    System.out.print("[");
    for (int i = 0; i < arr.length; i++)
    {
      System.out.print(arr[i]);
      if (i<arr.length-1)
      {
        System.out.print(", ");
      }
    }
    System.out.println("]");
  }
}


public class Task04Tester 
{


  public static void main(String[] args) 
  {
    int[] arr = {4, 10, 2, 8, 6, 7};
    int k = 3;
    int[] result = Task04.topVal(arr, k);
    Task04.print(result);
  }
}
