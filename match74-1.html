import java.util.*;

class MinHeap {
    private int[] heap;
    private int size;
    private int capacity;

    public MinHeap(int capacity) {
        this.heap = new int[capacity];
        this.capacity = capacity;
        this.size = 0;
    }
    public void insert(int val) {
        if (size==capacity) {
            return;
        }
        heap[size]=val;
        swim(size);
        size++;
    }
    
    private void swim(int idx) {
        int parent = (idx - 1) / 2;


        while (idx > 0 && heap[idx] < heap[parent]) {
            int temp=heap[idx];
            heap[idx]=heap[parent];
            heap[parent]=temp;
            idx=parent;


            parent=(idx-1)/2;
        }
    }

    
    public int extractMin() 
    {
        if (size==0) 
        {
            return -1;
        }
        int top=heap[0];
        heap[0]=heap[size-1];
        size--;
        sink(0);
        return top;
    }
    private void sink(int idx) {


        int left=2*idx+1;
        int right=2*idx+2;
        int small=idx;

        if (left<size&&heap[left]<heap[small]) 
        {
            small=left;
        }
        if (right<size&&heap[right]<heap[small]) 
        {
            small=right;
        }
        if (small!=idx) 
        {
            int temp=heap[idx];
            heap[idx]=heap[small];
            heap[small]= temp;
            sink(small);
        }
    }
}

import java.util.*;

class MaxHeap {
    private int[] heap;  
    private int size;  
    private int capacity;  
    
    public MaxHeap(int capacity) {  
       this.heap=new int[capacity];  
       this.capacity=capacity;  
       this.size = 0;  
    }  
    
    public void insert(int val) 
    {  
       if(size==capacity) 
       {  
          return;
       }  
       heap[size]= val;  
       
       swim(size);
       size++;  
    }  
    
    private void swim(int idx) 
    {  
        int pidx = (idx-1)/2;


        while (idx>0 && heap[idx]>heap[pidx]) 
        {
            int temp=heap[idx];
            heap[idx]=heap[pidx];
            heap[pidx]=temp;
            idx=pidx;


            pidx=(idx-1)/2;
        }
    }  
    
    public int extractMax() 
    {  
       if (size==0) 
       {    
          return -1;
       }  
       int top=heap[0]; 
       heap[0]=heap[size-1];   
       size--;
       sink(0); 
       return top; 
    }  
    public void sink (int idx)
    {


        int left =2*idx+1;
        int right=2*idx+2;
        int largest=idx;
        if (left<size && heap[left]>heap[largest])
        {
            largest=left;
            
        }
        if (right<size && heap[right]>heap[largest])
        {
            largest=right;
        }
        if (largest!=idx)
        {
            int temp=heap[idx];
            heap[idx]=heap[largest];
            heap[largest]=temp;
            sink(largest);
        }
    }
}

public class Task4
{

    public static int[] maxreturner(int[] received, int k) 
    {
        if (k ==0 || received==null ) 
        {
            System.out.println("Invalid");
            return new int[0];
        }
        MaxHeap heap= new MaxHeap(received.length);

        for(int i=1;i<=received.length;i++)
        {
            heap.insert(received[i-1]);
        }
        
        int[] returner = new int[k];

        for(int i=0;i<k;i++) 
        {
            returner[i]=heap.extractMax();
        }

        return returner; 
    }
