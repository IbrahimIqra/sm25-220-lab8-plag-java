   public  static class Minheap{
        int[] heap;
        int size;


        public Minheap(int capacity) {
            this.heap = new int[capacity+1]; 
            this.size = 0;
        }


        public String insert(int value) {
         if (this.size==this.heap.length-1)
            return "Insertion is not possible.";
        this.heap[this.size]=value;
         this.size++;
        this.swim();
        return "Insertion is successful.";
        }


        public void swim() {
           int chld=this.size;
        int paren= chld/2;
        while (paren>1 && this.heap[paren]>this.heap[chld]) {
            this.swap(paren,chld);
            chld= paren;
            paren= chld/2;
        }
        }


        public void swap(int l,int r){
            int a = heap[l];
             heap[l] = heap[r]; 
             heap[r] = a;
        }




        public int delete() {
            if (size == 0){
            return 0;
            }
            int min=heap[1];
            heap[1]=heap[size];
            size--;
            sink(1);
            return min;
        }


        public void sink(int s) {
            while (true){
                int left = 2*s;
                int right = 2*s+ 1;
                int small=s;
                if (left <= size && heap[left]<heap[small]){
                small=left;
                }


                if (right <= size && heap[right]<heap[small]){
                small= right;
                }
                if (small ==s){
                 break;
                }
                swap(s,small);
                s=small;
            }
        }
    }
}

   public  static class MaxHeap{
        int[] heap;
        int size;


        public MaxHeap(int capacity) {
            this.heap = new int[capacity+1]; 
            this.size = 0;
        }


        public String insert(int value) {
         if (this.size==this.heap.length-1)
            return "Insertion is not possible.";
        this.heap[this.size]=value;
         this.size++;
        this.swim();
        return "Insertion is successful.";
        }


        public void swim() {
           int chld=this.size;
        int paren= chld/2;
        while (paren>0 && this.heap[paren]<this.heap[chld]) {
            this.swap(paren,chld);
            chld= paren;
            paren= chld/2;
        }
        }


        public void swap(int l,int r){
            int a = heap[l];
             heap[l] = heap[r]; 
             heap[r] = a;
        }




        public int delete() {
            if (size == 0){
            return 0;
            }
            int top=heap[1];
             size--;
            heap[1]=heap[size];
            sink(0);
            return top;
        }


        public void sink(int large) {
            while (true){
                int left = 2 *large;
                int right = 2 *large+ 1;
                int largest=large;
                if (left <= size && heap[left]>heap[largest]){
                largest=left;
                }


                else if (right <= size && heap[right]>heap[largest]){
                 largest = right;
                }
                if (largest==large){
                    break;
                }
                swap(large,largest);
                largest=large;
            }
        }
    }
