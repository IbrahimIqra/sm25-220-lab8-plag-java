public class MinHeap
{
    private int[] heapArr;
    private int size;

    public MinHeap(int cap)
    {
        heapArr = new int[cap];
        size = 0;
    }

    private void insert(int val)
    {
        if (size==heapArr.length){
            System.out.println("Heap is full");
            return;
        }
        heapArr[size]=val;
        swim(size);
        size++;
    }

    private void swim(int a)
    {
        while (a>0)
        {
            int parent =(a - 1)/2;
            if (heapArr[parent]<heapArr[a])
            {
                break;
            }
            swap(parent, a);
            a = parent;
        }
    }

    private void swap(int c,int d)
    {
        int  temp=heapArr[c];
        heapArr[c]= heapArr[d];
        heapArr[d]= temp;
    }
private void sink(int e)
    {
            int smallest = e;
            int left = 2*e+1;
            int right = 2*e+2;



            if   (left<size && heapArr[left] <heapArr[smallest]) {
                smallest = left;
            }
            if(right<size && heapArr[right]<heapArr[smallest]) {
                smallest = right;
            }
            if(smallest!=e) {
                swap(e, smallest);
                sink(smallest);
            }


    }
public int extractMin()
    {
        if (size==0)
        {
            System.out.println("Heap is empty");
            return 0;
        }
        int min = heapArr[0];
        swap(0,size-1);
        size--;
        sink(0);
        return min;
    }
}

public class MaxHeap
{
    private int [] heapArr;
    private int size;
    public MaxHeap(int cap)
    {
        heapArr=new int[cap];
        size=0;
    }

    private void insert(int val)
    {
        if (size==heapArr.length)
        {
            System.out.println("Heap is full");
            return;
        }
        heapArr[size]=val;
        swim(size);
        size++;
    }


    private void swim(int a)
    {
        while (a>0)
        {
            int parent = (a-1)/2;
            if (heapArr[parent]>heapArr[a])
            {
                break;
            }
            swap(parent,a);
            a= parent;
        }
    }
    private void swap(int c,int d)
    {
        int temp = heapArr[c];
        heapArr[c]=heapArr[d];
        heapArr[d]=temp;
    }
    private void sink(int e)
    {
            int largest = e;
            int left= 2*e+1;
            int right= 2*e+2;



            if (left<size && heapArr[left]>heapArr[largest])
            {
                largest=left;
            }
            if (right<size && heapArr[right]>heapArr[largest])
            {
                largest=right;
            }

            if (largest!=e)
            {
                swap(e, largest);
                sink(largest);
            }
    }
public int extractMax()
    {
        if (size==0)
        {
            System.out.println("Heap is empty");
            return 0;
        }
        int max = heapArr[0];
        swap(0,size-1);
        size--;
        sink(0);
        return max;
    }
}

    {


        for (int i = 0;i<num.length;i++)

        int[] result = new int[k];
