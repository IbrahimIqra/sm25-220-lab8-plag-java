   public class drivercodeforheap {

    // ================== MAX HEAP ==================


    public static class MaxHeap {
        private int[] heap;
        private int size;

        public MaxHeap(int capacity) {
            heap = new int[capacity];
            size = 0;
        }

        public void insert(int val) {
            if (size == heap.length) {
                System.out.println("Heap is full!");
                return;
            }
            heap[size] = val;
            swim(size);
            size++;
        }

      public int extractMax() {


    int max = heap[0];
    heap[0] = heap[size - 1];
    size--;
    sink(0);
    return max;
}



        private void swim(int index) {
            while (index > 0) {
                int parent = (index - 1) / 2;
                if (heap[index] > heap[parent]) {
                    swap(index, parent);
                    index = parent;
                } else break;
            }
        }

        private void sink(int index) {
            while (2 * index + 1 < size) {
                int left = 2 * index + 1, right = 2 * index + 2, larger = left;
                if (right < size && heap[right] > heap[left]) larger = right;
                if (heap[index] < heap[larger]) {
                    swap(index, larger);
                    index = larger;
                } else break;
            }
        }



        private void swap(int i, int j) {
            int tmp = heap[i];
            heap[i] = heap[j];
            heap[j] = tmp;


        }

        public int[] getHeapArray() {
            int[] arr = new int[size];
            for (int i = 0; i < size; i++) arr[i] = heap[i];
            return arr;
        }
    }

    // ================== MIN HEAP ==================


    public static class MinHeap {
        private int[] heap;
        private int size;

        public MinHeap(int capacity) {
            heap = new int[capacity];
            size = 0;
        }

        public void insert(int val) {
            if (size == heap.length) {
                System.out.println("Heap is full!");
                return;
            }
            heap[size] = val;
            swim(size);
            size++;
        }

      public int extractMin() {


    int min = heap[0];
    heap[0] = heap[size - 1];
    size--;
    sink(0);
    return min;
}




        private void swim(int index) {
            while (index > 0) {
                int parent = (index - 1) / 2;
                if (heap[index] < heap[parent]) {
                    swap(index, parent);
                    index = parent;
                } else break;
            }
        }

        private void sink(int index) {
            while (2 * index + 1 < size) {
                int left = 2 * index + 1, right = 2 * index + 2, smaller = left;
                if (right < size && heap[right] < heap[left]) smaller = right;
                if (heap[index] > heap[smaller]) {
                    swap(index, smaller);
                    index = smaller;
                } else break;
            }
        }

        private void swap(int i, int j) {
            int tmp = heap[i];
            heap[i] = heap[j];


            heap[j] = tmp;
        }

        public int[] getHeapArray() {
            int[] arr = new int[size];
            for (int i = 0; i < size; i++) arr[i] = heap[i];
            return arr;


        }
    }

    // ================== TASK SCHEDULER USING MIN HEAP ==================
    public static int[] scheduleTasks(int[] tasks, int m) {
        MinHeap heap = new MinHeap(m);
        for (int i = 0; i < m; i++) heap.insert(0);
        for (int i = 0; i < tasks.length; i++) {
            int task = tasks[i];
            int minLoad = heap.extractMin();
            heap.insert(minLoad + task);
        }
        int[] result = new int[m];
        for (int i = 0; i < m; i++) result[i] = heap.extractMin();


        return result;
    }

    // ================== TOP K ELEMENTS USING MAX HEAP ==================
    public static int[] topKElements(int[] nums, int k) {
        MaxHeap heap = new MaxHeap(nums.length);
        for (int i = 0; i < nums.length; i++) {
            heap.insert(nums[i]);
        }

        int[] result = new int[k];
        for (int i = 0; i < k; i++) {
            result[i] = heap.extractMax();
        }
        return result;
    }
