public class MinHeap{

    static int[] heap;
    static int size;
    public static void Heap(int cap){
        heap=new int[cap];
        size = 0;
    }
    public static void insert(int value){
        if(size>=heap.length){
            return;
        }
        heap[size]=value;
        heap_Up(size);
        size++;
    }
// value swap kori jate heap property maintain kora jai
    public static void swap(int x,int j){
        int temp = heap[x];
        heap[x] = heap[j];
        heap[j] = temp;
    }
    // shobche choto value ta diye return dae
    public static int extracting_Min(){
        if (size == 0){
            return -1;
        }


        int min=heap[0];
        heap[0]=heap[size-1];
        size--;
        heap_Down(0);
        return min;
    }
    //choto value hoile root er theke then oi value 


    public static void heap_Up(int indx){
        while(indx> 0){
            int parent = (indx-1)/2;
            if(heap[parent]>heap[indx]){
                swap(parent,indx);
                indx=parent;
            } 
            else{
                break;
            }
        }

    }
    // boro valiue gula niche pathai
    public static void heap_Down(int indx){
        while(indx<size){


            int left=2*indx;
            int right=2*indx+1;
            int smallest=indx;
            if(left<size && heap[left]<heap[smallest]){
                smallest=left;
            }
            if(right<size && heap[right]<heap[smallest]){
                smallest=right;
            }
            if(smallest!=indx){
                swap(smallest,indx);
                indx=smallest;
            } 
            else{
                break;
            }
        }

    }
    //array te convert kore from heap
    public static int[] getHeap(){
        int[]new_arr = new int[size];
        for (int i=0; i<size; i++){
            new_arr[i]=heap[i];
        }


        return new_arr;
    }
    public static void main(String[] args){
        int[] task = {2,4,7,1,6}; // 
        int m=4;                      
        Heap(m);
        // 0 diye loads start korbo
        for (int i=0; i<m; i++){
            insert(0);
        }
        for (int i=0; i<task.length; i++){
            int minLoad = extracting_Min();
            insert(minLoad + task[i]);
        }
        int[] result = getHeap();
        for (int i = 0; i < result.length; i++){
            System.out.print(result[i] + " ");
        }
    }
}

public class top_large{
    public static int[]heap;
    public static int size;

    public static void new_Heap(int capacity){
        heap=new int[capacity];
        size=0;
    }
    public static void insert(int val){
        if (size>= heap.length){
            return;
        }
        heap[size] = val;
        heap_Up(size);
        size++;
    }
    // max alada korbo from the heap
    public static int extracting_Max(){
        if(size==0){
            return -1;
        }


        int max=heap[0];
        heap[0]=heap[size-1];
        size--;
        heap_Down(0);
        return max;
    }
    public static void heap_Up(int idx){
        while (idx> 0){
            int parent = (idx-1)/2;
            if(heap[parent]<heap[idx]){
                swap(parent, idx);
                idx = parent;
            }
            else{
                break;
            }
        }
    }
    public static void heap_Down(int idx){
        while (idx<size){


            int l=2*idx+1;
            int r=2*idx+2;
            int largest=idx;
            if (l<size && heap[l]>heap[largest]){
                largest =l;
            }
            if(r<size && heap[r]>heap[largest]){
                largest =r;
            }
            if(largest != idx){
                swap(largest, idx);
                idx=largest;
            }
            else{
                break;
            }
        }
    }
    public static void swap(int p,int j){
        int copy=heap[p];
        heap[p]=heap[j];
        heap[j]=copy;
    }
    public static int[]find_top(int[]arr, int k){
        int n=arr.length;
        new_Heap(n); // new heap banachi
        for (int i=0; i<n; i++){


            insert(arr[i]);
        }
        //k bar amra heap theke max alada korbo and k size er array te rakhbo
        int[]result=new int[k];
        for (int i=0; i<k; i++) {
            result[i]=extracting_Max();
        }


        return result;
    }
    public static void main(String[]args){
        int[]arr_origin={4,10,2,8,6,7};
        int k=3;
        int[]decend_result=find_top(arr_origin, k);
        for (int i=0; i<decend_result.length; i++){ // array ke print korar jonno
            System.out.print(decend_result[i] + " ");
        }
    }
}
