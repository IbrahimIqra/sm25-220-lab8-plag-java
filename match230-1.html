class MinHeap {
   private int[] heap;
   private int size;


   public MinHeap(int cap) {
       size = 0;
       heap = new int[cap];
   }




   public void insert(int value) {
       if (size == heap.length) {
           System.out.println("Heap is full!");


           return;
       }
       heap[size] = value;
       swim(size);
       size++;
   }


   // Swim: fix heap upwards
   private void swim(int index) {
      int pindx = (index-1 )/ 2 ;
      while (index>0 && heap[index]<heap[pindx]){
          int temp = heap[index];
          heap[index]= heap[pindx];
          heap[pindx]= temp ;


          index = pindx ;
          pindx = (index-1)/2 ;
      }
   }


   // Extract minimum (root)
   public int extractMin() {
    if(size<=0)return -1 ;


    int temp = heap[0];
    heap[0]= heap[size-1];
    //heap[size-1]= 0 ;
    size -=1 ;
    sink(0);


    return temp ;
   }


   // Sink: fix heap downwards
   private void sink(int index) {
       while (index * 2 + 1 < size) { // while left child exists
           int left = index * 2 + 1;


           int right = index * 2 + 2;


           int smallest = left;
           if (right < size && heap[right] < heap[left]) {
               smallest = right;
           }


           if (heap[index] > heap[smallest]) {
               int t = heap[index];
               heap[index]= heap[smallest];
               heap[smallest]= t ;
               sink(smallest);
           }
           else break;
       }
   }


   public int[] sort() {
       int[]arr = new int[size];
       int ogsize = size;


       for (int i = 0; i < ogsize; i++) {
           int m = extractMin();
           arr[i] = m ;
       }
       return arr ;
   }


   public void sortInplace(){
       int ogsize = size ;


       for (int i=ogsize-1 ; i>0 ; i--){
           int m = extractMin();
           heap[i]=m ;
       }
       size = ogsize ;
   }
  
}

class MaxHeap {
   private int[] heap;
   private int size;


   public MaxHeap(int cap) {
       size = 0;
       heap = new int[cap];
   }


   public void insert(int value) {
       if (size == heap.length) {
           System.out.println("Heap is full!");


           return;
       }
       heap[size] = value;
       swim(size);
       size++;
   }


   private void swim(int index) {
       int pindx = (index - 1) / 2;
       while (index > 0 && heap[index] > heap[pindx]) {
           int temp = heap[index];
           heap[index] = heap[pindx];
           heap[pindx] = temp;


           index = pindx;
           pindx = (index - 1) / 2;
       }
   }




   public int extractMax() {
       if (size <= 0) return -1;


       int temp = heap[0];
       heap[0] = heap[size - 1];
       size -= 1;
       heapify(0);


       return temp;
   }


   private void heapify(int index) {
       while (index * 2 + 1 < size) { // while left child exists
           int left = index * 2 + 1;
           int right = index * 2 + 2;


           int largest = left;
           if (right < size && heap[right] > heap[left]) {
               largest = right;
           }


           if (heap[index] < heap[largest]) {
               int t = heap[index];
               heap[index] = heap[largest];
               heap[largest] = t;
               heapify(largest);
           }
           else break;
       }
   }


   public int[] sort() {
       int[] arr = new int[size];
       int ogsize = size;


       for (int i = 0; i < ogsize; i++) {
           int m = extractMax();
           arr[i] = m;
       }
       return arr;
   }


   public void sortInplace() {
       int ogsize = size;


       for (int i = ogsize - 1; i > 0; i--) {
           int m = extractMax();
           heap[i] = m;
       }
       size = ogsize;
   }


}
