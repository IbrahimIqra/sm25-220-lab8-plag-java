//MinHeap (Needed for Task 3)

public class MinHeap {

    private int [] heap;
    private int size = 0;

    public MinHeap(int maxCapacity){
        heap = new int[maxCapacity];
    }

    public void swim(int idx){
        if(idx < 1){
            return;
        }
        int parent = (idx-1)/2;



        if(heap[parent] > heap[idx]){
            int temp = heap[parent];
            heap[parent] = heap[idx];
            heap[idx] = temp;
            swim(parent);
        }
    }

    public void insert(int key){
        if(size < heap.length){
            heap[size] = key;
            size += 1;


            swim(size-1);
        }
    }

    public void sink(int idx){
        int left = (2 * idx) + 1;
        int right = (2 * idx) + 2;

        if(left >= size){
            return;
        }

        int smaller = left;
        if(right < size && heap[right] < heap[left]){
            smaller = right;
        }

        if(heap[idx] > heap[smaller]){
            int temp = heap[idx];
            heap[idx] = heap[smaller];
            heap[smaller] = temp;
            sink(smaller);
        }
    }

    public int extractMin(){
        if(size == 0){
            return -1;
        }
        int minValue = heap[0];
        heap[0] = heap[size-1];
        size -= 1;
        sink(0);
        return minValue;
    }

}




//Task 3

public class Task3 {

    int[] tasks;
    int m;

    public Task3(int[] tasks, int m){
        this.tasks = tasks;
        this.m = m;
    }



    public int[] distribute(){
        MinHeap heap = new MinHeap(m);
        for(int i = 0; i < m; i++){
            heap.insert(0);
        }

        for(int i = 0; i < tasks.length; i++){
            int load = heap.extractMin();
            load += tasks[i];
            heap.insert(load);
        }

        int[] processTime = new int[m];
        for(int i = 0; i < m; i++){


            processTime[i] = heap.extractMin();
        }

        return processTime;
    }

}

    }
}


//MaxHeap (Needed for Task 4)

public class MaxHeap {

    private int [] heap;
    private int size = 0;

    public MaxHeap(int maxCapacity){
        heap = new int[maxCapacity];
    }

    public void swim(int idx){
        if(idx < 1){
            return;
        }
        int parent = (idx-1)/2;



        if(heap[parent] < heap[idx]){
            int temp = heap[parent];
            heap[parent] = heap[idx];
            heap[idx] = temp;
            swim(parent);
        }
    }

    public void insert(int key){
        if(size < heap.length){
            heap[size] = key;
            size += 1;


            swim(size-1);
        }
    }

    public void sink(int idx){
        int left = (2*idx)+1;
        int right = (2*idx)+2;

        if(left < size && right < size){
            if(heap[left] < heap[idx] && heap[right] < heap[idx]){
                return;
            }
            else{
                int greater = 0;
                if(heap[left] > heap[right]){
                    greater = left;
                }
                else{
                    greater = right;
                }
                int temp = heap[idx];
                heap[idx] = heap[greater];
                heap[greater] = temp;
                sink(greater);
            }
        }
    }

    public int extractMax(){
        if(size == 0){
            return -1;
        }
        size -= 1;
        int maxValue = heap[0];
        heap[0] = heap[size-1];
        sink(0);
        return maxValue;
    }

}



//Task 4


public class Task4 {

    int[] nums;
    int k;

    public Task4(int[] nums, int k){
        this.nums = nums;
        this.k = k;
    }

    public int[] largestElements(){



        MaxHeap heap = new MaxHeap(nums.length);

        for(int i = 0; i < nums.length; i++){
            heap.insert(nums[i]);
        }

        int[] maxElems = new int[k];

        for(int i = 0; i < k; i++){
            maxElems[i] = heap.extractMax();
        }

        return maxElems;

    }

}
