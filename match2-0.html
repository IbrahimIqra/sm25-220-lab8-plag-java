class MinHeapT_3{
    private int[] arr;
    private int size;
    public MinHeapT_3(int capa){
        arr = new int[capa];
        size = 0;
    }

    private void increasecapacity(){
        if (size == arr.length) {
            int[] new_array = new int[arr.length +1];
            System.arraycopy(arr, 0, new_array, 0, arr.length);
            arr = new_array;
        }
    }
    private void swapp(int i, int j) {
        int temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
    }
    public boolean isEmpty() {
        return size == 0;
    }


    public void insertt(int data) {
        increasecapacity();
        arr[size] = data;
        int childIndex = size;
        int parentIndex = (childIndex - 1) / 2;
        swimm(childIndex, parentIndex);
        size++;
    }
    private void swimm(int childIndex,int parentIndex){
        while (childIndex > 0 && arr[childIndex] < arr[parentIndex]) {
            swapp(childIndex, parentIndex);
            childIndex = parentIndex;
            parentIndex = (childIndex - 1) / 2;
        }
    }
    public int peek() {
        return arr[0];
    }
    private void sink(int i, int n) {
        int left = 2 * i + 1;
        int right = 2 * i + 2;
        int minIndex = i;

        if (left < n && arr[minIndex] > arr[left]) {
            minIndex = left;
        }
        if (right < n && arr[minIndex] > arr[right]) {
            minIndex = right;
        }
        if (minIndex != i) {
            swapp(i, minIndex);
            sink(minIndex, n);
        }
    }
    public int extractMin() {
        int data = arr[0];
        swapp(0, size - 1);
        size--;
        sink(0, size);
        return data;
    }
    public void heapsorting() {
        for (int i = size / 2 - 1; i >= 0; i--) {
            sink(i, size);
        }
        for (int i = size - 1; i > 0; i--) {
            swapp(0, i);
            sink(0, i);
        }
    }
    public void printSorted() {
        for (int i = 0; i < size; i++) {
            System.out.print(arr[i] + " ");
        }
    }
}

class MaxHeapT_04 {
    private int[] arr;
    private int size;

    public MaxHeapT_04(int capacity) {
        arr = new int[capacity];
        size = 0;
    }

    private void incrCapacity() {
        if (size == arr.length) {
            int[] newArray = new int[arr.length + 1];
            System.arraycopy(arr, 0, newArray, 0, arr.length);
            arr = newArray;
        }
    }

    private void swapp(int i, int j) {
        int temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
    }

    public boolean isEmptyy() {
        return size == 0;
    }



    public void insertt(int data) {
        incrCapacity();
        arr[size] = data;
        int childIndex = size;
        int parentIndex = (childIndex - 1) / 2;
        swimm(childIndex, parentIndex);
        size++;
    }

    private void swimm(int childIndex, int parentIndex) {
        while (childIndex > 0 && arr[childIndex] > arr[parentIndex]) {
            swapp(childIndex, parentIndex);
            childIndex = parentIndex;
            parentIndex = (childIndex - 1) / 2;
        }
    }

    public int peek() {


        return arr[0];
    }

    private void sinkk(int i, int n) {
        int left = 2 * i + 1;
        int right = 2 * i + 2;
        int maxIndex = i;

        if (left < n && arr[maxIndex] < arr[left]) {
            maxIndex = left;
        }
        if (right < n && arr[maxIndex] < arr[right]) {
            maxIndex = right;
        }
        if (maxIndex != i) {
            swapp(i, maxIndex);
            sinkk(maxIndex, n);
        }
    }

    public int extractMax() {
        int data = arr[0];
        swapp(0, size - 1);
        size--;
        sinkk(0, size);
        return data;
    }

    public void heapSort() {
        for (int i = size / 2 - 1; i >= 0; i--) {
            sinkk(i, size);
        }
        for (int i = size - 1; i > 0; i--) {
            swapp(0, i);
            sinkk(0, i);
        }
    }

    public void printSort() {
        for (int i = 0; i < size; i++) {
            System.out.print(arr[i] + " ");
        }
    }
}
