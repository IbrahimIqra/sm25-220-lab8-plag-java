public class Task03 {


    public static void main(String[]args){
        int [] tasks = {2,4,7,1,6};
        int m =4;
        t03( tasks,m );
    }

    public static void t03( int[]tasks , int m ){

        MinHeap h= new MinHeap( m );

        for(int i=0; i<m; i++){

            h.insert(0);
        }

        int index = 0;

        while(!h.isEmpty()) {

            int val= h.extractMin();
            h.insert ( val+tasks[index] );

            index++;

            if (index==tasks.length){
                break ;

            }

        }

        while( !h.isEmpty() ){

            System.out.print(h.extractMin()+" ");
        }
    }



    public static class MinHeap {
        private int [] heap;
        private int size;

        public MinHeap(int capacity) {
            heap= new int[capacity];
            size= 0;


        }

        public void insert(int data){
            heap[size] = data;
            int childindex = size;
            int parentindex =( childindex-1 )/2;
            swim( childindex,parentindex );
            size++;
        }

        private void swim(int childindex, int parentindex) {

            while(childindex>0 && heap[childindex]<heap[parentindex]){

                swap( childindex,parentindex);
                childindex=parentindex;
                parentindex =( childindex-1 )/2;
            }
        }

        private void swap(int i, int j) {

            int temp= heap[i];
            heap[i] = heap[j];


            heap[j]= temp;
        }

        private int extractMin() {

            int data = heap[0];
            swap(0,size-1);
            size--;
            sink(0,size);

            return data;
        }

        private void sink( int i, int n ) {

            int left= 2*i+1;
            int right= 2*i+2;
            int minindex= i;

            if( left<n && heap[minindex]> heap[left] ){
                minindex=left;
            }

            if(right<n &&  heap[minindex] >heap[right]){
                minindex = right;
            }

            if( minindex!=i ){

                swap( i,minindex );
                sink( minindex,n );
            }
        }

        public boolean isEmpty(){
            return size == 0;
         }
     }  
 }

    public static void  topKlargestElement(int [] nums, int k){
        MaxHeap h =  new MaxHeap (nums.length);

        for(int  i=0; i<nums.length; i++) {
            h.insert(nums[i]);
        }
        int [] arr=  new int [k];
        int index= 0;

        while (!h.isEmpty()  && index <k) {

            arr[index]= h.extractMax();
            index++;
        }

        for(int i= 0; i< arr.length; i++){
            System.out.print(arr[i]+ " ");
        }

    }



    public static class MaxHeap {
        private int [] heap;
        private int size;

        public MaxHeap(int capacity){
            heap =  new int [capacity];
            size = 0;
        }

        public void insert(int data) {
            heap[size] =data;
            int childindex = size;
            int parentindex =( childindex-1 )/2;
            swim( childindex,parentindex );
            size++;
        }

        private void swim(int childindex, int parentindex){



            while(childindex>0 && heap[childindex]>heap[parentindex]) {

                swap( childindex,parentindex);
                childindex =parentindex;
                parentindex =( childindex-1 )/2;
            }
        }

        private void swap(int i, int j){

            int temp= heap[i];
            heap[i] =heap[j];
            heap[j]= temp;
        }

         private void sink( int i, int n ){



            int left= 2*i+1;
            int right= 2*i+2;
            int maxIndex= i;

            if( left<n && heap[maxIndex]< heap[left] ){
                maxIndex=left;
            }

            if(right<n &&  heap[maxIndex] <heap[right]){
                maxIndex = right;
            }
            
            if( maxIndex!=i ){

                swap( i,maxIndex );
                sink( maxIndex,n );
            }
        }

         private int extractMax(){

            int data = heap[0];
            swap(0,size-1);
            size--;
            sink(0,size);

            return data;
        }

        public boolean isEmpty(){
            return size == 0;
         }
    }
