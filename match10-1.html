public class MinHeap {
    
    private int[] heap ;
    private int size;

    public MinHeap( int capacity){
        heap = new int[capacity+1];  
        size=0;
    }

    private int parent(int i){ 
    return i/2; 
    }
    private int left(int i){
    return 2*i;
    }
    private int right(int i){ 
    return 2*i+1;
    }

    
    public void insert(int x){
        if (size + 1 >= heap.length){ 
            System.out.println("No space to insert "+x); 
            return; 
        }
        size = size + 1 ;


        heap[size] = x;

        int i = size;
        while (i>1 && heap[parent(i)] > heap[i]){
            int t = heap[parent(i)];
            heap[parent(i)] = heap[i];
            heap[i] = t ;
            i = parent(i);
        }
    }

    
    public int extractMin(){
        if (size==0){
            System.out.println("Heap empty");
            return 0; // small shortcut
        }
        int min = heap[1];
        heap[1] = heap[size];
        size = size - 1;

    
        int i = 1;
        while (true){
            int l = left(i);
            int r = right(i);
            int smallest = i;

            // checking right first (not wrong, just unusual)


            if (r <= size && heap[r] < heap[smallest]) {
                smallest = r;
            }
            if (l <= size && heap[l] < heap[smallest]) {
                smallest = l;
            }

            if (smallest == i) break;

            int t = heap[i];
            heap[i] = heap[smallest];
            heap[smallest] = t;
            i = smallest;
        }
        return min;
    }

   
    public int[] getHeapArray(){
        return heap; // caller should read 1..size
    }


    public int getSize( ){
        return size ;
    }
}



public class task3 {

   
    public static int[] distributeTasks(int[] tasks, int m){
        MinHeap minHeap = new MinHeap(m); 

        
        int i=0; 
        while(i<m){ 
            minHeap.insert(0); 
            i = i + 1; 
        }

        for (i=0;i<tasks.length; i=i+1){


            int time = tasks[i] ;
            int currLoad = minHeap.extractMin();
            int newLoad  = currLoad + time ;
            minHeap.insert(newLoad); 
        }

        int[] result = new int[m];
        int[] arr = minHeap.getHeapArray();
        for (i = 1; i <= m; i++){
            result[i-1] = arr[i];  
        }
        int sum = 0; 



        return result;
    }

    public static void main(String[] args){
        int[] tasks = {2, 4, 7, 1, 6};
        int m=4 ;

        int[] result = distributeTasks(tasks, m);
        System.out.println("Final Machine Loads : " + Arrays.toString(result));
    }
}

public class MaxHeap{
	private int[] heap ;
    private int size;

    public MaxHeap( int capacity ){
        heap=new int[capacity+1]; 
    	size=0 ;
    }

    private int p(int i){
   return i/2;
   }
    private int l(int i){
    return 2*i;
   }
    private int r(int i){
     return 2*i+1;
   }

    public void insert( int x ){
        if(size+1>=heap.length){
            System.out.println("No Space Left for "+x);
            return;
        }
        size = size + 1 ;


        heap[size]=x ;
        int i = size ;
        while(i>1 && heap[p(i)] < heap[i]){
            int t = heap[p(i)];
            heap[p(i)] = heap[i]; 
            heap[i]=t; 
            i = p(i);
        }
    }

    public int extractMax( ){
        if(size==0){
            System.out.println("heap empty");
            return 0;
        }
        int mx = heap[1];
        heap[1]=heap[size];
        size=size-1;

        int i=1;
        while(true){
            int li=l(i);
            int ri=r(i);


            int largest = i;

            if(ri<=size && heap[ri] > heap[largest]){ largest=ri; }
            if(li<=size && heap[li] > heap[largest]){ largest=li; }

            if(largest==i) break;

            int t=heap[i]; heap[i]=heap[largest]; heap[largest]=t;
            i=largest;
        }
        return mx;
    }

    public int getSize( ){ return size; }
    public int[] getHeapArray(){ return heap; }
}
