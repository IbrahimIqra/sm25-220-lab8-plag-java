        MinHeap h = new MinHeap(m);
        for (int i = 0; i < m; i++) {
            h.insert(0);
        }
        int index = 0;
        while (index < tasks.length) {
            int val = h.extractMin();

            index++;
        }

            System.out.println(h.extractMin());
        }
    }

     public int extractMin() {
        int data = arr[0];
        swap(0, size - 1);
        size--;
        sink(0, size);
        return data;
    }
     private void swap(int i, int j) {
        int temp = arr[i];
        arr[i] = arr[j];


        arr[j] = temp;
    }
     private void sink(int i, int n) {
        int left = 2 * i + 1;
        int right = 2 * i + 2;
        int minIndex = i;

        if (left < n && arr[minIndex] > arr[left]) {
            minIndex = left;
        }
        if (right < n && arr[minIndex] > arr[right]) {
            minIndex = right;
        }
        if (minIndex != i) {
            swap(i, minIndex);
            sink(minIndex, n);
        }
    }
