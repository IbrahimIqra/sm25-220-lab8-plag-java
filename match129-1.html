//Task 3
public class Task3 {



    public static class SmallHeap {
        private int[] data;
        private int count;
        
        public SmallHeap(int max) {
            data = new int[max];
            count = 0;
        }
        
        public void put(int x) {
            data[count] = x;
            goUp(count);
            count = count + 1;
        }
        
        private void goUp(int pos) {
            while (pos > 0) {
                int dad = (pos - 1) / 2;
                if (data[dad] <= data[pos]) {
                    break;
                }
                int temp = data[dad];
                data[dad] = data[pos];
                data[pos] = temp;
                pos = dad;
            }
        }
        
        public int takeSmall() {
            int small = data[0];
            data[0] = data[count - 1];
            count = count - 1;
            goDown(0);
            return small;
        }
        
        private void goDown(int pos) {
            while (2 * pos + 1 < count) {
                int left = 2 * pos + 1;


                int right = 2 * pos + 2;
                int small = left;
                
                if (right < count && data[right] < data[left]) {
                    small = right;
                }
                
                if (data[pos] <= data[small]) {
                    break;
                }
                
                int temp = data[pos];
                data[pos] = data[small];
                data[small] = temp;


                pos = small;
            }
        }
        
        public int[] getData() {
            int[] output = new int[count];
            for (int i = 0; i < count; i++) {
                output[i] = data[i];
            }
            return output;
        }
    }
    


    public static void main(String[] args) {
        int[] work = {2, 4, 7, 1, 6};
        int boxes = 4;
        


        SmallHeap heap = new SmallHeap(boxes);
        
        for (int i = 0; i < boxes; i++) {
            heap.put(0);
        }
        
        for (int i = 0; i < work.length; i++) {
            int load = heap.takeSmall();
            heap.put(load + work[i]);
        }
        
        int[] ans = heap.getData();
        
        System.out.print("[");
        for (int i = 0; i < ans.length; i++) {
            System.out.print(ans[i]);
            if (i < ans.length - 1) {
                System.out.print(", ");
            }
        }
        System.out.println("]");
    }
}

public class TaskFour {



    public static class BigHeap {
        private int[] data;
        private int count;
        
        public BigHeap(int max) {
            data = new int[max];
            count = 0;
        }
        
        public void put(int x) {
            data[count] = x;
            goUp(count);
            count = count + 1;
        }
        
        private void goUp(int pos) {
            while (pos > 0) {
                int dad = (pos - 1) / 2;
                if (data[dad] >= data[pos]) {
                    break;
                }
                int temp = data[dad];
                data[dad] = data[pos];
                data[pos] = temp;
                pos = dad;
            }
        }
        
        public int takeBig() {
            int big = data[0];
            data[0] = data[count - 1];
            count = count - 1;
            goDown(0);
            return big;
        }
        
        private void goDown(int pos) {
            while (2 * pos + 1 < count) {
                int left = 2 * pos + 1;
                int right = 2 * pos + 2;


                int big = left;
                
                if (right < count && data[right] > data[left]) {
                    big = right;
                }
                
                if (data[pos] >= data[big]) {
                    break;
                }
                
                int temp = data[pos];
                data[pos] = data[big];
                data[big] = temp;
                pos = big;
            }
        }
    }
    


    public static void main(String[] args) {
        int[] nums = {4, 10, 2, 8, 6, 7};
        int top = 3;
        


        BigHeap heap = new BigHeap(nums.length);
        
        for (int i = 0; i < nums.length; i++) {
            heap.put(nums[i]);
        }
        
        int[] ans = new int[top];
        for (int i = 0; i < top; i++) {
            ans[i] = heap.takeBig();
        }
        
        System.out.print("[");
        for (int i = 0; i < ans.length; i++) {
            System.out.print(ans[i]);
            if (i < ans.length - 1) {
                System.out.print(", ");
            }
        }
        System.out.println("]");
    }
}
