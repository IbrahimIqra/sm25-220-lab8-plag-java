public class MaxHeap {
    private int[] heap;
    private int size;
    private int capacity;

    public MaxHeap(int cap) {
         capacity = cap ;
        heap = new int[cap ];
          size = 0;
    }

    public boolean insert(int x) {
      if (size == capacity ) {
        return false;
      }
        heap[size] = x;
        swim(size);
        size++;
        return true;
    }



    private void swim(int x) {
        while ( x > 0) {
            int parent = ( x - 1) / 2;
            if (heap[ x] > heap[parent]) {
                swap( x, parent);
                x = parent;
            } else break;
        }
    }

    public int extractMax() {
      if (size == 0) {
        return -1;
      }
        int max = heap[0];
        heap[0] = heap[size - 1];
        size--;
        sink(0);
        return max;
    }

    private void sink(int  x) {
        while (true) {


            int left = 2 *  x + 1;
            int right = 2 *  x + 2;
            int largest =  x;

            if (left < size && heap[left] > heap[largest]) largest = left;


            if (right < size && heap[right] > heap[largest]) largest = right;

            if (largest !=  x) {
                swap( x, largest);
                x = largest;
            } else break;
        }
    }

    private void swap(int x, int y) {
        int temp = heap[x];
        heap[x] = heap[y];
        heap[y] = temp;
    }

    public int[] sort() {
        int[] sorted = new int[size];
        int original  = size;
        for (int i = 0; i < original ; i++) {
            sorted[i] = extractMax();
        }
        size = 0;
        return sorted;
    }

    public void printHeap() {
        for (int i = 0; i < size; i++) System.out.print(heap[i] + " ");
        System.out.println();
    }

    public int getSize() {
        return size;
    }
}

public class task3 {

    public static int[] distributeTasks(int[] tasks, int x) {
        MinHeap heap = new MinHeap(x);

       
        for (int i = 0; i < x; i++) {
            heap.insert(0);
        }

        
        for (int i = 0; i < tasks.length; i++) {
            int time = tasks[i];
            int min  = heap.extractMin();
            int updatedLoad = min  + time;
            heap.insert(updatedLoad);


            System.out.println(updatedLoad);  
        }

       
        int[] finalLoads = new int[x];
        for (int i = 0; i < x; i++) {
            finalLoads[i] = heap.extractMin();
        }

        return finalLoads;
    }



    public static void main(String[] args) {
        int[] tasks = {2, 4, 7, 1, 6};
        int x = 4;

        int[] loads = distributeTasks(tasks, x);

        System.out.println(Arrays.toString(loads));  
    }
