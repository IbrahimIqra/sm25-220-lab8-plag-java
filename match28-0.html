public class Task3{ 
  private static int min_find(int [] array){ 
    int min= 0; 
    for(int p=1; p<array.length; p++){ 
      if(array[p] < array[min]){ 
        min=p; 
      } 
    } 
    return min; 
  } 
    
  private static void min_swim(int[]x, int indx){ 
    int parent_node = indx-1/2; 
    while(indx > 0 && x[indx] < x[parent_node]){ 
      swap(x,indx,parent_node); 
      indx = parent_node; 
      parent_node = indx-1/2; 
    } 
  } 
   
  private static void swap(int[] x, int p, int q){ 
    int tempor = x[p]; 
    x[p] = x[q]; 
    x[q] = tempor; 
  } 
 
   
  public static int[] method_1(int[] tasks, int m){  
    int[] hp = new int[m]; 
    for(int q=0; q<m; q++){ 
      hp[q]=0; 
    } 
    
    for(int z=0; z<tasks.length; z++){ 
      int tempor = tasks[z]; 
      int min_indx = min_find(hp); 
      hp[min_indx] += tempor; 
      min_swim(hp, min_indx); 
    } 
    return hp; 
  } 
} 
 
 
 
public class Task3_Tester{

    int m = 4; 
    int[] ar = Task3.method1(tasks,m); 
    for(int i=0;i <ar.length;i++){ 
      System.out.print(ar[i]+" "); 
    } 
  } 
}

public class Task4{ 
  private static void swap(int[] x, int p, int q){ 
    int tempor = x[p]; 
    x[p] = x[q]; 
    x[q] = tempor; 
  } 
 
  private static void max_swim(int[]y, int indx){ 
    int parent_node = (indx-1)/2; 
    while(indx > 0 && y[indx] > y[parent_node]){ 
      swap(y, indx, parent_node); 
      indx = parent_node; 
      parent_node = (indx-1)/2; 
    } 
  } 
   
  private static int max_find(int[]z, int var){ 
    if(var==0){ 
      return -1; 
    } 
    int max = z[0]; 
    z[0] = z[var-1];  
    z[var-1] = 0; 
    max_heap(z, var-1, 0); 
    return max; 
  } 
    
  private static void max_heap(int[]ar, int num, int indx){ 
        int max = indx; 
        int left = 2*indx+1; 
        int right = 2*indx+2; 
        if(left<num && ar[left] > ar[max]){ 
            max=left; 
        } 


        if(right<num && ar[right] > ar[max]){ 
            max=right; 
        } 
        if(max != indx){ 
            swap(ar,indx,max); 
            max_heap(ar,num,max); 
        } 
    } 
 
  public static int[] find(int[]ar, int r){ 
    int a = ar.length; 
    int[] hp = new int[a]; 
    for(int i=0; i<a; i++){ 
      hp[i] = ar[i]; 
      max_swim(hp,i);  
    } 
    int[] result =new int[r]; 
      for(int i=0; i<r; i++){ 
        result[i]= max_find(hp, a-i); 
      } 
      return result; 
   } 
 }
