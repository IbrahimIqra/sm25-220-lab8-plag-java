// Task 3
public class Task3 {


    public void TaskDistributor(int[] tasks, int m){
        MinHeap min_heap = new MinHeap(m);

        for(int i = 0; i < m; i++){
            min_heap.insert(0);
        }

        for (int task : tasks) {
            int minLoad = min_heap.extractMin(); 
            min_heap.insert(minLoad + task);     
        }

        int[] res = new int[m];
    for (int i = 0; i < m; i++) {
        res[i] = min_heap.extractMin(); 
    }
        System.out.print("Final machine loads: ");
        for (int x : res) {
            System.out.print(x + " ");
        }
        System.out.println();
    }
}

        MaxHeap max_heap = new MaxHeap(nums.length);
        for(int num : nums){
            max_heap.insert(num);
        }

        int[] res = new int[k];

        for(int i = 0; i < k; i++){
            int temp = max_heap.extractMax();
            res[i] = temp;
        }
        for(int val : res){
            System.out.print(val + " ");
        }

    }
}

// MinHeap class
class MinHeap{
    private int[] heap;
    private int size;

    public MinHeap(int cap){
        this.heap = new int[cap];


        this.size = 0;
    }

    private int parent(int i){
        return (i-1)/2;
    }
    private int left(int i){
        return 2*i+1;
    }
    private int right(int i){
        return 2*i+2;
    }

    private void swap(int i, int j) {
        int temp = heap[i];
        heap[i] = heap[j];
        heap[j] = temp;
    }

    public void insert(int val) {
        if (size == heap.length) {
            System.out.println("Heap is full");
            return;
        }
        heap[size] = val;
        swim(size);   
        size++;
    }

    public void swim(int i) {


        while (i > 0 && heap[i] < heap[parent(i)]) {
            swap(i, parent(i));
            i = parent(i);
        }
    }

    public int extractMin() {
        if (size <= 0) {
            System.out.println("Heap is empty");
            return -1;
        }


        int min = heap[0];
        heap[0] = heap[size - 1];
        size--;
        sink(0);   
        return min;
    }

    private void sink(int i) {
        while (true) {
            int smallest = i;
            int l = left(i), r = right(i);



            if (l < size && heap[l] < heap[smallest]) {
                smallest = l;
            }
            if (r < size && heap[r] < heap[smallest]) {
                smallest = r;
            }
            if (smallest != i) {
                swap(i, smallest);
                i = smallest;
            } else{
                break;
            }
            
        }
    }

    public int[] sort() {
        
        int[] copy = new int[size];
        for (int i = 0; i < size; i++) {
            copy[i] = heap[i];
        }

        int originalSize = size;

        int[] sorted = new int[size];
        for (int i = 0; i < sorted.length; i++) {
            sorted[i] = extractMin();
        }

        heap = new int[copy.length];
        for (int i = 0; i < copy.length; i++) {
            heap[i] = copy[i];
        }
        size = originalSize;

        return sorted;
    }


}




// MaxHeap class
public class MaxHeap {
    private int[] heap;
    private int size;

    public MaxHeap(int cap){
        this.heap = new int[cap];


        this.size = 0;
    }

    private int parent(int i){
        return (i-1)/2;
    }
    private int left(int i){
        return 2*i+1;
    }
    private int right(int i){
        return 2*i+2;
    }

    private void swap(int i, int j) {
        int temp = heap[i];
        heap[i] = heap[j];
        heap[j] = temp;
    }

    public void insert(int val) {
        if (size == heap.length) {
            System.out.println("Heap is full");
            return;
        }
        heap[size] = val;
        swim(size);   
        size++;
    }

    private void swim(int i) {


        while (i > 0 && heap[i] > heap[parent(i)]) {
            swap(i, parent(i));
            i = parent(i);
        }
    }

    public int extractMax() {
        if (size <= 0) {
            System.out.println("Heap is empty!");
            return -1;
        }



        int max = heap[0];
        heap[0] = heap[size - 1];
        size--;
        sink(0);
        return max;
    }

    private void sink(int i) {
        while (true) {
            int largest = i;


            int l = left(i), r = right(i);

            if (l < size && heap[l] > heap[largest]){

            largest = l;
            }
            if (r < size && heap[r] > heap[largest]) {
                largest = r;
            }

            if (largest != i) {
                swap(i, largest);
                i = largest;
            } else {
                break;
            }
        }
    }

    public int[] sort() {
        int[] copy = new int[size];
        for (int i = 0; i < size; i++) copy[i] = heap[i];

        int originalSize = size;

        int[] sorted = new int[size];
        for (int i = 0; i < sorted.length; i++) {
            sorted[i] = extractMax();
        }


        heap = new int[copy.length];
        for (int i = 0; i < copy.length; i++) heap[i] = copy[i];
        size = originalSize;

        return sorted;
    }

}
