public class MinHeap {
    private int[] heap;
    private int size;
    
    public MinHeap(int cap) {
        heap = new int[cap];
        size = 0;
    }
    private void swap(int m, int n) {
        int store = heap[m];
        heap[m] = heap[n];
        heap[n] = store;
    }


    private void swim(int i) {
        while (i>0) {
            int parent = (i-1)/2;
            if (heap[parent]>heap[i]) {
                swap(parent, i);
                i = parent;
                continue;
            }
            else {
                break;
            }
        }
    }
    private void sink(int l) {
        while (l<size) {
            int left = 2*l+1;
            int right = 2*l+2;
            int smallest = l;


            if (left<size && heap[left]<heap[smallest]) {
                smallest = left;
            }
            if (right<size && heap[right]<heap[smallest]) {
                smallest = right;
            }
            if (smallest!=l) {
                swap(l, smallest);
                l = smallest;
            }
            else {
              break;
            }
        }
    }
    public void insert(int elem) {
        heap[size] = elem;
        size++;
        swim(size-1);
    }
    public int Min() {
        int min = heap[0];
        heap[0] = heap[size-1];
        size--;
        sink(0);
        return min;
    }
}

public class MaxHeap {
    private int[] heap;
    private int size;

    public MaxHeap(int cap) {
        heap = new int[cap];
        size = 0;
    }
    private void swap(int m, int n) {
        int store = heap[m];
        heap[m] = heap[n];
        heap[n] = store;
    }
    public void insert(int elem) {
        if (size < heap.length) {
            heap[size] = elem;
            size++;
            swim(size-1);
        }
        }
        private void swim(int i) {
            int parent = (i-1)/2;
            while (parent>=0 && heap[parent] < heap[i]) {
                swap(i, parent);
                i = parent;
                parent = (i-1)/2;
            }
        }
        private void sink(int i) {
            int parent = i;
            while (parent<size) {
                int left = 2*parent+1;
                int right = 2*parent+2;
                int largest = parent;


                if (left<size && heap[left]>heap[largest]) {
                    largest = left;
                }
                if (right<size && heap[right]>heap[largest]) {
                    largest = right;
                }
                if (largest!=parent) {
                    swap(parent, largest);
                    parent = largest;
                }
                else {
                    break;
                }
            }
        }
        public int Max() {
            int max = heap[0];
            heap[0] = heap[size - 1];
            size--;
            sink(0);
            return max;
        }
    }
