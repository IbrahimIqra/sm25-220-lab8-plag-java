public class Task3 {
    public static int[] result_tasks(int[] tasks, int m) {
        MinHeap machines = new MinHeap(m);
        int i;

        for (i = 0; i < m; i++) {
            machines.insert(0);
        }
        for (i = 0; i < tasks.length; i++) {
            int minLoad = machines.extractMin();
            int newLoad = minLoad + tasks[i];
            machines.insert(newLoad);
        }
        int[] result = new int[m];
        for (i = 0; i < m; i++) {
            result[i] = machines.extractMin();
        }
        return result;
    }

    public static void main(String[] args) {


        int[] tasks = {2, 4, 7, 1, 6};
        int m = 4;
        int[] result = result_tasks(tasks, m);

        for (int i = 0; i < result.length; i++) {
            System.out.print(result[i]);
            if (i < result.length - 1) {
                System.out.print(", ");
            }
        }
        System.out.println();
    }
}


class MinHeap {
    private int[] heap;
    private int size;
    private int capacity;

    public MinHeap(int capacity) {
        this.capacity = capacity;
        this.size = 0;
        this.heap = new int[capacity + 1]; 
    }

    public void insert(int value) {
        size++;


        heap[size] = value;
        swim(size);
    }

    private void swim(int k) {
        while (k > 1 && heap[k] < heap[k / 2]) {
            swap(k, k / 2);
            k = k / 2;
        }
    }

    public int extractMin() {
        int minVal = heap[1];
        heap[1] = heap[size];


        size--;
        sink(1);
        return minVal;
    }

    private void sink(int k) {
        while (2 * k <= size) {
            int j = 2 * k;
            if (j < size && heap[j] > heap[j + 1]) {
                j++;
            }
            if (heap[k] <= heap[j]) {
                break;
            }
            swap(k, j);
            k = j;
        }
    }

    private void swap(int i, int j) {
        int temp = heap[i];
        heap[i] = heap[j];
        heap[j] = temp;
    }
}



public class Task4 {
    public static int[] result_arr(int[] nums, int k) {
        if (k <= 0) {
            return new int[0];
        }
        if (k >= nums.length) {
            for (int i = 0; i < nums.length - 1; i++) {
                for (int j = i + 1; j < nums.length; j++) {
                    if (nums[i] < nums[j]) {
                        int temp = nums[i];
                        nums[i] = nums[j];
                        nums[j] = temp;
                    }
                }
            }
            return nums;
        }



        MaxHeap maxHeap = new MaxHeap(nums.length);
        for (int i = 0; i < nums.length; i++) {
            maxHeap.insert(nums[i]);
        }

        int[] topK = new int[k];
        for (int i = 0; i < k; i++) {
            topK[i] = maxHeap.extractMax();
        }
        return topK;
    }

            System.out.print(result[i]);

        }

    }
}
