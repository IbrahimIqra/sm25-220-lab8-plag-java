class MinHeap{
  private int[] heap;
  private int size;
  public MinHeap(int cap){
    heap = new int[cap];
    size = 0;
  }
  public void insert(int val){
    if (size >= heap.length){
      return;
    }
    heap[size] = val;
    swim(size);
    size++;
  }
  private void swim(int idx){
    while (idx > 0) {
      int parent = (idx - 1) / 2;
      if (heap[idx] >= heap[parent]){
        break;
      }
      int temp = heap[idx];
      heap[idx] = heap[parent];
      heap[parent] = temp;
      idx = parent;
    }
  }
  public int extractMin(){
    if (size == 0){
      return -1;
    }
    int min = heap[0];
    heap[0] = heap[size - 1];
    size--;
    if (size > 0){
      sink(0);
    }
    return min;
  }
  private void sink(int idx){
    while (true){
      int leftChild = 2 * idx + 1;


      int rightChild = 2 * idx + 2;
      int smallest = idx;
      
      if (leftChild < size && heap[leftChild] < heap[smallest]){
        smallest = leftChild;
      }
      if (rightChild < size && heap[rightChild] < heap[smallest]){
        smallest = rightChild;
      }
      
      if (smallest == idx){
        break;
      }
      int temp = heap[idx];
      heap[idx] = heap[smallest];
      heap[smallest] = temp;
      idx = smallest;
    }
  }
  public void sort(){
    int ogSize = size;
    for (int i = size - 1; i > 0; i--){
      int temp = heap[0];
      heap[0] = heap[i];
      heap[i] = temp;
      size--;
      sink(0);
    }
    size = ogSize;
  }
  public int[] getHeapArray(){
    int[] result = new int[size];
    for (int i = 0; i < size; i++){
      result[i] = heap[i];
    }
    return result;
  }
  public int getSize(){
    return size;
  }
}
class Task3 {
  public static int[] distributeTasks(int[] tasks, int m){
    MinHeap machineLoads = new MinHeap(m);
    for (int i = 0; i < m; i++) {
      machineLoads.insert(0);
    }
    for (int i = 0; i < tasks.length; i++) {
      int minLoad = machineLoads.extractMin();
      int newLoad = minLoad + tasks[i];
      machineLoads.insert(newLoad);
    }
    return machineLoads.getHeapArray();
  }
}
public class Task3_Driver {
  public static void main(String[] args) {
    int[] tasks = {2, 4, 7, 1, 6};
    int m = 4;
    
    int[] result = Task3.distributeTasks(tasks, m);
    for (int val : result) {
      System.out.print(val + " ");
    }
  }
}

class MaxHeap{
  private int[] heap;
  private int size;
  private int capacity;
  
  public MaxHeap(int capacity){
    this.capacity = capacity;
    this.heap = new int[capacity];


    this.size = 0;
  }
  
  private int parent(int i){ 
    return (i - 1) / 2;
  }
  private int left(int i){ 
    return 2 * i + 1; 
  }
  private int right(int i){
    return 2 * i + 2; 
  }
  
  public void insert(int val){
    if (size == capacity){
      System.out.println("Heap is full!");
      return;
    }
    heap[size] = val;
    size++;
    swim(size - 1);
  }
  
  private void swim(int i){


    while (i > 0 && heap[parent(i)] < heap[i]){
      swap(i, parent(i));
      i = parent(i);
    }
  }
  
  public int extractMax(){
    if (size <= 0) return -1;
    int max = heap[0];


    heap[0] = heap[size - 1];
    size--;
    sink(0);
    return max;
  }
  
  private void sink(int i){
    int largest = i;
    int l = left(i);
    int r = right(i);
    
    if (l < size && heap[l] > heap[largest]) largest = l;
    if (r < size && heap[r] > heap[largest]) largest = r;
    
    if (largest != i){
      swap(i, largest);
      sink(largest);
    }
  }
  
  private void swap(int i, int j){
    int temp = heap[i];
    heap[i] = heap[j];
    heap[j] = temp;
  }
}
class Task4{
  public static int[] topK(int[] nums, int k){
    MaxHeap heap = new MaxHeap(nums.length);
    for (int n : nums){
      heap.insert(n);
    }
    int[] result = new int[k];
    for (int i = 0; i < k; i++){
      result[i] = heap.extractMax();
    }
    
    return result;
  }
}
