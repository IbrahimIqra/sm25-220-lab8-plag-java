public class MinHeap{
  private int []heap;
  private int cap;
  private int index;
  public MinHeap(int x){
    cap = x;
    heap = new int[x];
    index =0;
  }
  private int parent (int i){
    return (i-1)/2;
  }
  private int leftchild(int i){
    return 2*i+1;
  }
  private int rightchild (int i){
    return 2*i+2;
  }


  public void swap (int a, int b){
    int temp = heap[a];
    heap[a]=heap[b];
    heap[b] = temp;
  }
  public void insert (int val){
    int i = index;
    heap[index++]=val;
    while(i!=0 && heap[parent(i)]>heap[i]){
      swap(i, parent (i));
      i=parent(i);
    }
  }
  public int getmin(){
    if(index<=0){


      return -1;
    }
      int root = heap[0];
      heap[0]= heap[index -1];
      index--;
      heapifyDown(0);
      return root;
    }
  public void heapifyDown(int i){
    int s = i;
    int left = leftchild(i);
    int right = rightchild(i);


    if(left<index&&heap[left]<heap[s]){
      s = left;
    }
    if(right<index&&heap[right]<heap[s]){
      s=right;
    }
    if (s!=i){
      swap(i,s);
      heapifyDown(s);
    }
  }
