public class Task3 {




    public static int[] processingTime(int[] tasks, int m){


        MinHeap time = new MinHeap(m);


        for (int i = 0 ; i < m ; i ++){


            time.insert(0);


        }


        for (int i = 0 ; i < tasks.length ; i++){


            int small = time.extractMin();
            int temp = tasks[i] + small;
            time.insert(temp);


        }


        int[] finalTime = new int [m];


        for (int i = 0 ; i < m ; i ++){


            finalTime [i] = time.extractMin();


        }


        return finalTime;


    }


    public static void main(String[] args) {




        int[] tasks = {2, 4, 7, 1, 6};
        int m = 4;


        int[] task4 = processingTime(tasks, m);


        System.out.print("[");


        for (int i = 0 ; i < task4.length ; i ++){


            System.out.print(task4[i]);


            if (i < task4.length -1 ){


                System.out.print(", ");


            }


        }




        System.out.println("]");
    }
}

public class Task4 {


    public static int[] topK(int[] nums, int k) {


        MaxHeap max = new MaxHeap(nums.length);


        for (int i = 0 ; i < nums.length ; i++){


            max.insert(nums[i]);


        }


        int[] largeVal = new int[k];


        for (int i = 0 ; i < k ; i++){


            largeVal[i] = max.extractMax();


        }


        return largeVal;


    }




    public static void main(String[] args) {




        int[] nums = {4, 10, 2, 8, 6, 7};
        int k = 3;


        int[] topK = topK(nums, k);




        System.out.print("[");


        for (int i = 0; i < topK.length; i++) {


            System.out.print(topK[i]);


            if (i < topK.length - 1) {


                System.out.print(", ");


            }


        }


        System.out.println("]");
    }


}
