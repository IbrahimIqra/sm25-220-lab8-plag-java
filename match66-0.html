public class Task3 {


   static class myMinHeap{
      private int arr[];
      private int s;
      public myMinHeap(int cap){
         arr=new int[cap];
         s=cap;
      }
      private void down(int i){


         int sm=i;
         int l =2*i+1;
         int r=2*i+2;


         if(l<s && arr[l] < arr[sm]){
            sm=l;
         }
         if(r<s && arr[r] < arr[sm]){
            sm=r;
         }
         if(sm!=i){
            int t =arr[i];
            arr[i] = arr[sm];
            arr[sm]=t;
            down(sm);
         }
      }
      private void up(int i){
         while(i>0 && arr[i] < arr[(i-1)/2]){
            int t=arr[i];
            arr[i]=arr[(i-1)/2];
            arr[(i-1)/2]=t;
            i = (i-1)/2;
         }
      }


      public int takeMin(){
         if(s==0){
            return -1;
         }
         int r =arr[0];
         arr[0] = arr[s-1];
         s=s-1;
         down(0);
         return r;
      }
      public void put(int v){
         if(s == arr.length){
            return;
         }
         arr[s]=v;
         int i=s;


         s=s+1;
         up(i);
      }
      public int[] getArr(){
         int[] r=new int[s];
         for(int i=0;i<s;i=i+1){
            r[i]=arr[i];
         }
         return r;


      }
   }


  


   public static int[] assign(int[] t, int m){
      myMinHeap h =new myMinHeap(m);
      for(int i=0;i<m;i=i+1){
         h.arr[i]=0;
      }
      for(int i=0;i<t.length; i=i+1){


         int mn =h.takeMin();
         mn =mn+t[i];


         h.put(mn);
      }




      return h.getArr();
   }
   //tester
   public static void main(String[] a){
      int[] tasks = {2,4,7,1,6};
      int mm =4;


      int[] r = assign(tasks, mm);
      for(int i=0; i<r.length-1;i=i+1){
         int min=i;

            if(r[j] < r[min]){
               min=j;
            }
         }

      }

      public int takeMax(){

   public static int[] topK(int[] nums, int k){
      myMaxHeap h = new myMaxHeap(nums.length);
