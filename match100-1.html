class Task3 {


   public static void main(String[] args) {


       int[] tasks = {2, 4, 7, 1, 6};
       int m = 4;
       int[] result = distask(tasks, m);
       for (int load : result) {


           System.out.print(load + " ");
       }
       // Expected output: 2 4 7 7 (order may vary due to heap structure)
   }


   public static int[] distask(int[] tasks, int m) {
       MinHeap heap = new MinHeap(m);
       for (int i = 0; i < m; i++) {
           heap.inst(0);
       }


       for (int task : tasks) {
           int minLoad = heap.genMin();
           int newLoad = minLoad + task;
           heap.inst(newLoad);
       }


       return heap.get();
   }


   public static class MinHeap {
       int[] heap;
       int size;
       int capacity;


       public MinHeap(int capacity) {
           this.capacity = capacity;
           this.size = 0;
           this.heap = new int[capacity + 1];
       }


       public void inst(int value) {
           if (size == capacity) {
               return;
           }
           size++;


           heap[size] = value;
           swim(size);
       }


       public void swim(int idx) {
           while (idx > 1 && heap[idx] < heap[idx / 2]) {
               swap(idx, idx / 2);
               idx = idx / 2;
           }
       }


       public int genMin() {
           if (size == 0) {
               return 0 ;
           }
           int min = heap[1];
           heap[1] = heap[size];


           size--;
           sink(1);
           return min;
       }


       public void sink(int idx) {
           while (2 * idx <= size) {
               int j = 2 * idx;
               if (j < size && heap[j] > heap[j + 1]) {
                   j++;
               }
               if (heap[idx] <= heap[j]) {
                   break;
               }
               swap(idx, j);
               idx = j;
           }
       }


       public void swap(int i, int j) {
           int temp = heap[i];
           heap[i] = heap[j];


           heap[j] = temp;
       }


       public int[] get() {
           int[] result = new int[size];
           for (int i = 0; i < size; i++) {
               result[i] = heap[i + 1];
           }
           return result;
       }
   }
}

class Task4 {


   public static void main(String[] args) {


       int[] nums = {4, 10, 2, 8, 6, 7};
       int k = 3;
       int[] result = large(nums, k);
       for (int num : result) {


           System.out.print(num + " ");
       }


   }


   public static int[] large(int[] nums, int k) {
       MaxHeap heap = new MaxHeap(nums.length);
       for (int num : nums) {
           heap.inst(num);
       }


       int[] result = new int[k];


       for (int i = 0; i < k; i++) {
           result[i] = heap.genMin();
       }
       return result;
   }


   public static class MaxHeap {
       int[] heap;
       int size;
       int cap;


       public MaxHeap(int cap) {
           this.cap = cap;
           this.size = 0;
           this.heap = new int[cap + 1];
       }


       public void inst(int value) {
           if (size == cap) {
               return ;
           }
           size++;


           heap[size] = value;
           swim(size);
       }


       public void swim(int idx) {
           while (idx > 1 && heap[idx] > heap[idx / 2]) {
               swap(idx, idx / 2);
               idx = idx / 2;
           }
       }


       public int genMin() {
           if (size == 0) {
               return 0 ;
           }
           int max = heap[1];
           heap[1] = heap[size];


           size--;
           sink(1);
           return max;
       }


       public void sink(int idx) {
           while (2 * idx <= size) {
               int j = 2 * idx;
               if (j < size && heap[j] < heap[j + 1]) {
                   j++;
               }
               if (heap[idx] >= heap[j]) {
                   break;
               }
               swap(idx, j);
               idx = j;
           }
       }


       public void swap(int i, int j) {
           int temp = heap[i];
           heap[i] = heap[j];
           heap[j] = temp;
       }
   }
