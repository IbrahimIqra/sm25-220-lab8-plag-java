//MinHeap Class
//The task 3's function is inside and at the end of the MinHeap class.

public class MinHeap{
    
    private int [] heap;
    private int size;

    public MinHeap(int capacity){
        heap = new int [capacity];
        size=0;
    }

    public void insert(int val){
        heap[size] = val;
        swim(size);
        size++;
    }

    private void swim(int index){
    while(index>0 && heap[(index-1)/2]>heap[index]){
        swap(index,(index-1)/2);


        index = (index-1)/2;
    }
    }

    private void swap(int child,int parent){
        int temp = heap[child];
        heap[child] = heap[parent];
        heap[parent] = temp;

    }

    public int extractMin(){
        if(size==0){
            System.err.println("Heap is Empty!");
        }
        int min = heap[0];


        swap(0,size-1);
        size--;
        sink(0);
        return min;
    }
    private void sink(int index){
        int small = index;
        int left_child_ind = (2*index)+1;
        int right_child_ind = (2*index)+2;
        if(left_child_ind<size && heap[small]>heap[left_child_ind]){
            small = left_child_ind;
        }
        if(right_child_ind<size && heap[small]>heap[right_child_ind]){
            small = right_child_ind;
        }
        if(small!=index){
            swap(index,small);
            sink(small);
        }

    }

    public void sort(){


        int [] sorted_arr = new int [size];
        int temp_size = size;
        for(int i=0; i<sorted_arr.length;i++){
            sorted_arr[i] = extractMin();
        }
        for(int i=0;i<sorted_arr.length;i++){
            heap[i] = sorted_arr[i];
        }


        size = temp_size;
    }
    public void printHeap(){
        for(int i=0;i<size;i++){
            System.out.print(heap[i]+" ");
        }
        System.out.println();
    }

    //Task 3

    public int[] loadManagement(int [] tasks, int m){
        MinHeap heap = new MinHeap(m);
        for(int i=0;i<m;i++){
            heap.insert(0);
        }
        for(int i=0;i<tasks.length;i++){
            int min = heap.extractMin();
            min = min+tasks[i];
            heap.insert(min);
        }
        heap.sort();
        return heap.heap;
    }


}

public class MaxHeap {
    
    private int [] heap;
    private int size;

    public MaxHeap(int capacity){
        heap = new int [capacity];
        size=0;
    }

    public void insert(int val){
        heap[size] = val;
        swim(size);
        size++;
    }

    private void swim(int index){
    while(index>0 && heap[(index-1)/2]<heap[index]){
        swap(index,(index-1)/2);
        index = (index-1)/2;
    }
    }

    private void swap(int child,int parent){
        int temp = heap[child];
        heap[child] = heap[parent];
        heap[parent] = temp;

    }

    public int extractMax(){
        if(size==0){
            System.err.println("Heap is Empty!");
        }
        int max = heap[0];
        swap(0,size-1);
        size--;
        sink(0);
        return max;
    }
    private void sink(int index){
        int large = index;


        int left_child_ind = (2*index)+1;
        int right_child_ind = (2*index)+2;
        if(left_child_ind<size && heap[large]<heap[left_child_ind]){
            large = left_child_ind;
        }
        if(right_child_ind<size && heap[large]<heap[right_child_ind]){
            large = right_child_ind;
        }
        if(large!=index){
            swap(index,large);
            sink(large);
        }

    }

    public void sort(){


        int [] sorted_arr = new int [size];
        int temp_size = size;
        for(int i=0; i<sorted_arr.length;i++){
            sorted_arr[i] = extractMax();
        }
        for(int i=0;i<sorted_arr.length;i++){
            heap[i] = sorted_arr[i];
        }


        size = temp_size;
    }

    public void printHeap(){
        for(int i=0;i<size;i++){
            System.out.print(heap[i]+" ");
        }
        System.out.println();
    }

    //Task 04

    public int [] max_elems(int [] arr, int k){
        MaxHeap heap = new MaxHeap(arr.length);
        int [] new_arr = new int[k];
        for(int i=0; i<arr.length;i++){
            heap.insert(arr[i]);
        }
        heap.sort();
        for(int i=k-1;i>=0;i--){
            new_arr[i] = heap.heap[i];
        }
        return new_arr;

    }

}
