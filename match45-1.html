public class TaskDistribute{


   public static void insert (int[]heap, int value, int size){


       heap[size] = value;


       while(size>0){
           int now = (size-1)/2;
           if(heap[now]>heap[size]){
               int temp = heap[now];
               heap[now] = heap[size];
               heap[size] = temp;
               size = now;
           }
           else{
               break;
           }
       }
   }


   public static int MinFinder(int[]heap, int size){


       int min = heap[0];
       heap[0] = heap[size-1];
       int val = 0;
       while(true){
           int left = 2*val+1;
           int right = 2*val+2;
           int small = val;


           if(left<size && heap[left]<heap[small]){
               small = left;
           }
           if(right<size && heap[right]<heap[small]){
               small = right;
           }
           if(small!=val){
               int temp = heap[val];
               heap[val] = heap[small];
               heap[small] = temp;
               val = small;
           }
           else{
               break;
           }
       }
       return min;
   }


   public static int[] TaskDistribution(int[]tasks, int m){
       int[] heap = new int[tasks.length+m];
       int hsize = 0;
       for(int i=0;i<m;i++){
           insert(heap,0, hsize);
           hsize++;
       }


       for(int i=0;i<tasks.length;i++){
           int minVal = MinFinder(heap, hsize);
           hsize--;
           int newVal = minVal+tasks[i];
           insert(heap, newVal, hsize);
           hsize++;
       }
       int[] output = new int[m];
       for(int i=0;i<m;i++){
           output[i] = MinFinder(heap, hsize);
           hsize--;
       }


       return output;
   }


   public static void main(String[]args){
       int[]tasks = {2, 4, 7, 1, 6};
       int m = 4;
       int[] result = TaskDistribution(tasks, m);
       for(int i=0;i<result.length;i++){
           System.out.print(result[i]+" ");
       }
   }
}

public class TopK {


   public static void heapify(int[]arr, int n, int i){
       int left = 2*i+1;
       int right = 2*i+2;
       int largest = i;


       if(left<n && arr[left]>arr[largest]){
           largest = left;
       }
       if(right<n && arr[right]>arr[largest]){
           largest = right;
       }
       if(largest!=i){
           int temp = arr[i];
           arr[i] = arr[largest];
           arr[largest] = temp;
           heapify(arr, n, largest);
       }
   }


   public static void maxHeap(int[]arr, int n){
       for(int i=(n/2)-1;i>=0;i--){


           heapify(arr, n, i);
       }
   }


   public static int MaxFinder(int[]arr, int n){
       if(n<=0){
           return -1;
       }
       int m = arr[0];
       arr[0] = arr[n-1];
       heapify(arr, n-1, 0);
       return m;
   }


   public static int[] topKfinder(int[] arr, int k){
       int n = arr.length;
       if(k>n){
           k=n;
       }
       maxHeap(arr, n);
       int[] output = new int[k];
       int hsize = n;


       for(int i=0;i<k;i++){
           output[i] = MaxFinder(arr, hsize);
           hsize--;
       }
       return output;
   }


   public static void main(String[]args){
       int[] given = {4, 10, 2, 8, 6, 7};
       int k = 3;
       int[] topK = topKfinder(given, k);


       for(int i=0;i<k;i++){
           System.out.print(topK[i]+" ");
       }
   }
}
