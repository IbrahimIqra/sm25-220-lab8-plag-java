public class Task3 {


    public static int[] distributeTasks(int[] tasks, int m) {
        MinHeap heap = new MinHeap(m);

        for (int i = 0; i < m; i++) {
            heap.insert(0);
        }

        for (int i = 0; i < tasks.length; i++) {


            heap.insertRebalance(tasks[i]);
        }

        return heap.getArray();
    }

    public static void main(String[] args) {
        int[] tasks = {2, 4, 7, 1, 6};
        int m = 4;

        int[] result = distributeTasks(tasks, m);

        System.out.print("[");
        for (int i = 0; i < result.length; i++) {


            System.out.print(result[i]);
            if (i < result.length - 1) System.out.print(", ");
        }
        System.out.println("]");
    }
}

class MinHeap {
    private int[] heap;
    private int size;
    private int capacity;

    public MinHeap(int capacity) {
        this.capacity = capacity;
        this.size = 0;
        this.heap = new int[capacity];
    }

    public void insert(int value) {
        if (size == capacity) {
            System.out.println("Heap is full !!!");
            return;
        }
        heap[size] = value;
        size++;
        swim(size - 1);
    }

    public void swim(int index) {
        while (index > 0) {
            int parent = (index - 1) / 2;
            if (heap[index] < heap[parent]) {
                swap(index, parent);
                index = parent;
            } else {
                break;
            }
        }
    }



    public int extractMin() {
        if (size == 0) return -1; 
        int root = heap[0];
        heap[0] = heap[size - 1];
        size--;
        sink(0);
        return root;
    }

    public void sink(int index) {
        while (true) {
            int left = 2 * index + 1;
            int right = 2 * index + 2;
            int smallest = index;



            if (left < size && heap[left] < heap[smallest]) {
                smallest = left;
            }
            if (right < size && heap[right] < heap[smallest]) {
                smallest = right;
            }
            if (smallest != index) {
                swap(index, smallest);
            } else {
                break;
            }
        }
    }

    public void insertRebalance(int value) {
        int minVal = extractMin();


        int newVal = minVal + value;
        insert(newVal);
    }

    public int[] getArray() {
        int[] result = new int[size];
        for(int i = 0; i < size; i++) {
            result[i] = heap[i];
        }
        return result;
    }

    public void swap(int i, int j) {
        int temp = heap[i];
        heap[i] = heap[j];
        heap[j] = temp;
    }
}

class MaxHeap {
    private int[] heap;
    private int size;
    private int capacity;

    public MaxHeap(int capacity) {
        this.capacity = capacity;
        this.size = 0;
        this.heap = new int[capacity];
    }

    public void insert(int value) {
        if (size == capacity) {
            System.out.println("Heap is full !!!");
            return;
        }
        heap[size] = value;
        size++;
        swim(size - 1);
    }

    public void swim(int index) {
        while (index > 0) {
            int parent = (index - 1) / 2;
            if (heap[index] > heap[parent]) {
                swap(index, parent);
                index = parent;
            } else {
                break;
            }
        }
    }



    public int extractMax() {
        if (size == 0) 
            return -1;
        int root = heap[0];
        heap[0] = heap[size - 1];
        size--;
        sink(0);
        return root;
    }

    public void sink(int index) {
        while (true) {
            int left = 2 * index + 1;
            int right = 2 * index + 2;
            int largest = index;



            if (left < size && heap[left] > heap[largest]) {
                largest = left;
            }
            if (right < size && heap[right] > heap[largest]) {
                largest = right;
            }
            if (largest != index) {
                swap(index, largest);
                index = largest;
            } else {
                break;
            }
        }
    }

    public void swap(int i, int j) {
        int temp = heap[i];
        heap[i] = heap[j];
        heap[j] = temp;
    }
}

public class Task4 {


    public static int[] topK(int[] nums, int k) {
        MaxHeap heap = new MaxHeap(nums.length);

        for (int i = 0; i < nums.length; i++) {


            heap.insert(nums[i]);
        }

        int[] result = new int[k];
        for (int i = 0; i < k; i++) {


            result[i] = heap.extractMax();
        }

        return result;
    }

    public static void main(String[] args) {
        int[] nums = {4, 10, 2, 8, 6, 7};
        int k = 3;

        int[] result = topK(nums, k);

        System.out.print("[");
        for (int i = 0; i < result.length; i++) {


            System.out.print(result[i]);
            if (i < result.length - 1) System.out.print(", ");
        }
        System.out.println("]");
    }
