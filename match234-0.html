import java.util.*;

public class Task3_MinHeapSimple {



    static class MinHeap {
        private int[] a;
        private int size;

        public MinHeap(int capacity) {
            a = new int[capacity];
            size = 0;
        }

        public void insert(int val) {
            if (size == a.length) {
                return;
            }
            a[size] = val;
            int i = size;
            while (i > 0) {
                int p = (i - 1) / 2;
                if (a[i] >= a[p]) {
                    break;
                }
                swap(i, p);
                i = p;
            }
            size++;
        }

        public int extractMin() {
            if (size == 0) {
                return 0;
            }
            int min = a[0];
            a[0] = a[size - 1];
            size--;
            sink(0);
            return min;
        }

        public void sink(int i) {
            while (true) {
                int l = 2 * i + 1;
                if (l >= size) {
                    break;
                }


                int r = l + 1;
                int s = l;
                if (r < size && a[r] < a[l]) {
                    s = r;
                }
                if (a[i] <= a[s]) {
                    break;
                }
                swap(i, s);
                i = s;
            }
        }

        public void sort() {
            int n = size;
            int[] out = new int[n];
            for (int i = 0; i < n; i++) {
                out[i] = extractMin();
            }
            System.arraycopy(out, 0, a, 0, n);
            size = n;
        }

        public void reinsert(int val) {
            insert(val);
        }

        public int[] getHeap() {
            return Arrays.copyOf(a, size);
        }

        private void swap(int i, int j) {
            int t = a[i];
            a[i] = a[j];
            a[j] = t;
        }
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int n = sc.nextInt();
        int[] tasks = new int[n];
        for (int i = 0; i < n; i++) {
            tasks[i] = sc.nextInt();
        }
        int m = sc.nextInt();

        MinHeap heap = new MinHeap(m);

        for (int i = 0; i < m; i++) {
            heap.insert(0);
        }

        for (int t : tasks) {
            int load = heap.extractMin();
            heap.reinsert(load + t);
        }

        System.out.println(Arrays.toString(heap.getHeap()));
    }
}

class MaxHeap {
    private int[] heap;
    private int size;
    private int capacity;

    public MaxHeap(int capacity) {
        this.capacity = capacity;
        this.size = 0;


        this.heap = new int[capacity];
    }

    private int parent(int i) { return (i - 1) / 2; }
    private int left(int i) { return 2 * i + 1; }
    private int right(int i) { return 2 * i + 2; }

    public void insert(int val) {
        if (size == capacity) return;
        heap[size] = val;
        int i = size;
        size++;


        while (i > 0 && heap[parent(i)] < heap[i]) {
            swap(i, parent(i));
            i = parent(i);
        }
    }

    public int extractMax() {
        if (size <= 0) return -1;
        if (size == 1) {
            size--;
            return heap[0];
        }
        int root = heap[0];


        heap[0] = heap[size - 1];
        size--;
        sink(0);
        return root;
    }

    private void sink(int i) {
        int largest = i;
        int l = left(i);
        int r = right(i);
        if (l < size && heap[l] > heap[largest]) largest = l;
        if (r < size && heap[r] > heap[largest]) largest = r;
        if (largest != i) {
            swap(i, largest);
            sink(largest);
        }
    }

    private void swap(int i, int j) {
        int temp = heap[i];
        heap[i] = heap[j];
        heap[j] = temp;
    }
}

public class Task4 {
    public static int[] topK(int[] nums, int k) {
        MaxHeap maxHeap = new MaxHeap(nums.length);
        for (int i = 0; i < nums.length; i++) {
            maxHeap.insert(nums[i]);
        }
        int[] result = new int[k];
        for (int i = 0; i < k; i++) {
            result[i] = maxHeap.extractMax();
        }
        return result;
    }



    public static void main(String[] args) {
        int[] nums = {4, 10, 2, 8, 6, 7};
        int k = 3;
        int[] result = topK(nums, k);
        System.out.print("[");
        for (int i = 0; i < result.length; i++) {
            System.out.print(result[i]);
            if (i < result.length - 1) System.out.print(", ");
        }
        System.out.println("]");
    }
