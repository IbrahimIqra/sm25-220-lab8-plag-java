class MinHeap {
   private int[] heap;
   private int size;
   private int capacity;


   MinHeap(int cap) {
       this.capacity = cap;
       heap = new int[cap];
       size = 0;
   }


   public void insert(int value) {
       if (this.size >= this.capacity) {
           System.out.println("Heap is full");
           return;
       }
       this.heap[this.size] = value;
       swim(this.size);
       size++;
   }
   private void swim(int indx) {
       while (indx > 0) {
           int parentIndx = (indx - 1) / 2;


           if (this.heap[indx] < this.heap[parentIndx]) {
               int temp = this.heap[indx];
               this.heap[indx] = this.heap[parentIndx];
               this.heap[parentIndx] = temp;
               indx = parentIndx;
           } else {
               break;
           }
       }
   }
   public int extractMin() {


       if (this.size == 0) {
           System.out.println("Heap is empty");
           return -1;
       }
       int temp = this.heap[0];
       this.heap[0] = this.heap[this.size - 1];
       this.size--;
       sink(0);
       return temp;
   }
   private void sink(int indx) {
       while (true) {


           int left = 2 * indx + 1;
           int right = 2 * indx + 2;
           int smallest = indx;


           if (left < this.size && this.heap[left] < this.heap[smallest]) {
               smallest = left;
           }


           if (right < this.size && this.heap[right] < this.heap[smallest]) {
               smallest = right;
           }
           if (smallest != indx) {
               int temp = this.heap[indx];
               this.heap[indx] = this.heap[smallest];
               this.heap[smallest] = temp;
               indx = smallest;
           } else {
               break;
           }
       }
   }
   public int[] customizeArr() {
       int[] arr = new int[size];
       for (int i = 0; i < size; i++) {
           arr[i] = heap[i];
       }
       return arr;
   }


   public static int[] assignTasks(int[] tasks, int m) {
       MinHeap minHeap = new MinHeap(m);
       for (int i = 0; i < m; i++) {
           minHeap.insert(0);
       }
       for (int i = 0; i < tasks.length; i++) {
           int load = minHeap.extractMin();


           load += tasks[i];
           minHeap.insert(load);
       }
       int[] result = new int[m];
       for (int i = 0; i < m; i++) {
           result[i] = minHeap.extractMin();
       }
       return result;
   }
}

class MaxHeap{
   private int [] heap;
   private int size;
   private int capacity;


   MaxHeap(int cap){
       this.capacity = cap;
       this.heap = new int[cap];
       this.size = 0;
   }




   public void insert(int value) {
       if (this.size >= this.capacity) {
           System.out.println("Heap is full");
           return;
       }
       this.heap[this.size] = value;
       swim(this.size);
       size++;
   }
   public void swim(int indx) {
       int parentIndx = indx / 2;


       if (this.heap[indx] > this.heap[parentIndx]) {
           int temp = this.heap[indx];
           this.heap[indx] = this.heap[parentIndx];
           this.heap[parentIndx] = temp;
       }
       indx = parentIndx;
       parentIndx = indx / 2;
   }
   public int extractMax() {


       if (this.size == 0) {
           System.out.println("Heap is empty");
           return -1;
       }
       int temp = this.heap[0];
       this.heap[0] = this.heap[this.size - 1];
       this.heap[this.size - 1] = 0;
       this.size--;
       sink(0);
       return temp;
   }
   public void sink(int indx) {


       int left = 2 * indx;
       int right = 2 * indx + 1;
       if (left >= this.size) {
           return;
       }
       int largest = left;
       if (right < this.size && this.heap[left] < this.heap[right]) {
           largest = right;
       }
       if (this.heap[indx] < this.heap[largest]) {


           int temp = this.heap[indx];
           this.heap[indx] = this.heap[largest];
           this.heap[largest] = temp;
           sink(largest);
       }
   }
}
