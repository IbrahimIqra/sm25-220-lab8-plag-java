class Minheap{
    private int[] arr;
    private int size;
    public Minheap(int c){
        arr = new int[c];


        size = 0;
    }
    private void swap(int i, int j){
        int t = arr[i];
        arr[i]=arr[j];
        arr[j]= t;
    }


    public void insert(int d){
        arr[size]=d;
        int ci = size;
        int pi = (ci-1)/2;
        swim(ci,pi);
        size++;
    }
    private void swim(int ci, int pi){
        while(ci>0 && arr[ci]<arr[pi]){
           swap(ci,pi);
           ci = pi;
           pi = (ci-1)/2;


        }
    }
    public int extractMin(){
        int d = arr[0];
        swap(0,size-1);
        size--;
        sink(0,size);
        return d;
    }
    private void sink(int i,int n){
        int l = 2*i+1;
        int r = 2*i+2;
        int mini = i;
        if(l<n && arr[mini]>arr[l]){
            mini = l;
        }if(r<n && arr[mini]>arr[r]){
            mini = r;
        }if(mini!=i){
            swap(i, mini);
            sink(mini, n);
        }
    }
    public void heapsort(){
        for(int i=size/2-1;i>=0;i--){
            sink(i, size);
        }for(int i=size - 1; i>=0;i--){
            swap(0, i);
            sink(0,i);
        }
    }
    public int peek(){
        return arr[0];
    }
    public boolean isEmpty(){
        return size==0;
    }
}

public class task3 {


    public static void main(String[] args){
        int[] tasks = {2, 4, 7, 1, 6};
        int m = 4;
        Task03(tasks,m);

    }


    public static void Task03(int[] t, int m){
        Minheap mh = new Minheap(m);
        for(int i=0; i<m; i++){
            mh.insert(0);
        }
        int index = 0;
        while(!mh.isEmpty()){
            int val = mh.extractMin();
            mh.insert(val+t[index]);
            index++;
            if(index==t.length){
                break;
            }
        }
        while(!mh.isEmpty()){
            System.out.println(mh.extractMin());
        }
    }
}
