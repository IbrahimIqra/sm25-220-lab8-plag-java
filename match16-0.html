class MinHeap {
private int[] array;
private int heapSize;
public MinHeap(int maxSize) {
array = new int[maxSize + 1];
heapSize = 0;
}
public void insert(int element) {
array[++heapSize] = element;
bubbleUp(heapSize);
}
private void bubbleUp(int pos) {
while (pos > 1 && array[pos / 2] > array[pos]) {
exchange(pos / 2, pos);
pos = pos / 2;
}
}
public int extractMin() {
int minimum = array[1];
exchange(1, heapSize--);
trickleDown(1);
return minimum;
}
private void trickleDown(int pos) {
while (2 * pos <= heapSize) {
int child = 2 * pos;
if (child < heapSize && array[child] > array[child + 1]) child++;
if (array[pos] <= array[child]) break;
exchange(pos, child);
pos = child;
}
}
private void exchange(int a, int b) {
int tmp = array[a];
array[a] = array[b];
array[b] = tmp;
}
public void sort() {
int tempSize = heapSize;
for (int idx = heapSize; idx > 1; idx--) {
exchange(1, idx);
heapSize--;
trickleDown(1);
}
heapSize = tempSize;
}
public int[] getArray() {
int[] resultArray = new int[heapSize];
for (int idx = 1; idx <= heapSize; idx++) {
resultArray[idx - 1] = array[idx];
}
return resultArray;
}
}

public int[] assignTasks(int[] tasks, int m) {
MinHeap minHeap = new MinHeap(m);
for (int count = 0; count < m; count++) {
minHeap.insert(0);
}
for (int processingTime : tasks) {
int minLoad = minHeap.extractMin();
minHeap.insert(minLoad + processingTime);
}
return minHeap.getArray();
}

public int[] topKLargest(int[] nums, int k) {
MaxHeap maxHeap = new MaxHeap(nums.length);
for (int number : nums) {
maxHeap.insert(number);
}
int[] topElements = new int[k];
for (int count = 0; count < k; count++) {
topElements[count] = maxHeap.extractMax();
}
return topElements;
}
