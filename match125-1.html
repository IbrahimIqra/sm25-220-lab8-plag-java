public class Maxheap{
  private int[] heap;
  private int size;
  
  
  public Maxheap(int cap){
    heap= new int[cap];
    size=0;
    
  }
  public void insert(int val){


    if(size==heap.length){
      System.out.println("Heap reached full capacity");
      
    }
    heap[size]=val;
    swim(size);
    size++;
    
  }
  
  private void swim(int n){
    while(n>0){
      int parent=(n-1)/2;
      if(heap[parent]<heap[n]){
        swap(parent, n);
        n=parent;
      }else{


        break;
      }
    }
  }
  
  public int maxExtract(){
    if(size==0){
      System.out.println("Empty Heap, add more elements");
    }
    int max=heap[0];
    heap[0]= heap[size-1];
    size--;
    sink(0);
    return max;
    
  }
  
  public int[] toArrays(){
    int[] arr1= new int[size];
    for(int n=0; n<size; n++){
      arr1[n]=heap[n];
    }
      return arr1;
    
  }
  private void sink(int i){
    while(2*i+1<size){
      int j= 2*1+1;
      if(j+1<size && heap[j+1]>heap[i]){
        j++;
      }
      if(heap[i]>= heap[j]){
        break;
      }
      swap(i,j);
      i=j;
      
    }
  }
  
  private void swap(int i, int j){
    int temp= heap[i];
    heap[i]=heap[j];
    heap[j]=temp;
  }
  
  private void sortArray(){
    int[] temp= new int[size];
    for(int i=0; i<size; i++){
      temp[i]= heap[i];
    }
    
    int len= temp.length;
    for(int k=(len/2)-1; k>=0; k--){
      sink(temp, k, len);
    }
    for(int f=len-1; f>0; f--){
      int swap= temp[0];
      temp[0]=temp[f];
      temp[f]=swap;
      sink(temp, 0, f);
    }
    System.out.print("Sorted: ");
    for(int val: temp){
    System.out.print(val+ " ");
    }
    System.out.println();
  }
    
  private void sink(int[] a, int o, int len){
    while(2*o+1<len){
    int n= 2*o+1;
    if(n+1<len && a[n+1]> a[n]){
    n++;
    }
    if(a[o]>= a[n]){
    break;
    }
    int temp= a[o];
    a[o]=a[n];
    a[n]=temp;
    o=n;
    }
  }
  
}
