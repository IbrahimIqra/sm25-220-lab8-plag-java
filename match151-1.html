// max_heap:




public class MaxHeap {
    private Integer[] max_heap_arr;
    private int mh_size;
    private int mh_capacity;

    public MaxHeap(int cap) {
        this.max_heap_arr = new Integer[cap];
        this.mh_size = 0;
        this.mh_capacity = cap;
    }

    public void mh_insert(int elem) {
        if (mh_size >= mh_capacity) {
            System.out.println("Heap full!");
            return;
        }

        max_heap_arr[mh_size] = elem;
        mh_swim(mh_size);
        mh_size++;
    }

    private void mh_swim(int index) {
        while (index > 0) {
            int parent = (index - 1) / 2;
            if (max_heap_arr[index] <= max_heap_arr[parent]) break;

            max_swap(index, parent);
            index = parent;
        }
    }

    public int remove_max() {
        if (mh_size <= 0) {
            System.out.println("Heap empty!");
            return -1;
        }

        int max = max_heap_arr[0];
        max_heap_arr[0] = max_heap_arr[mh_size - 1];
        max_heap_arr[mh_size - 1] = null;
        mh_size--;

        mh_heapify(0);
        return max;
    }

    private void mh_heapify(int index) {
        int largest = index;
        int left = (index * 2) + 1;
        int right = (index * 2) + 2;

        if (left < mh_size && max_heap_arr[left] > max_heap_arr[largest]) {
            largest = left;
        }

        if (right < mh_size && max_heap_arr[right] > max_heap_arr[largest]) {
            largest = right;
        }

        if (largest != index) {
            max_swap(index, largest);
            mh_heapify(largest);
        }
    }

    private void max_swap(int i, int j) {
        int temp = max_heap_arr[i];
        max_heap_arr[i] = max_heap_arr[j];
        max_heap_arr[j] = temp;
    }

    public int[] mh_sort() {
        int[] sorted = new int[mh_size];
        int originalSize = mh_size;

        for (int i = 0; i < originalSize; i++) {
            sorted[i] = remove_max();
        }
        
        return sorted;
    }

    public void mh_print() {
        for (int i = 0; i < mh_size; i++) {
            System.out.print(max_heap_arr[i] + " ");
        }
        System.out.println();
    }
}




// min_heap:



public class MinHeap {
    private Integer[] mih_heap_arr;
    private int mih_size;
    private int mih_capacity;

    public MinHeap(int cap) {


        this.mih_heap_arr = new Integer[cap];
        this.mih_size = 0;
        this.mih_capacity = cap;
    }

    public void mih_insert(int elem) {
        if (this.mih_size >= this.mih_capacity) {
            System.out.println("Heap full!");
            return;
        }
        this.mih_heap_arr[this.mih_size] = elem;
        mih_swim(this.mih_size);
        this.mih_size++;
    }



    private void mih_swim(int index) {
        int pae_pos = (index - 1) / 2;

        while (index > 0 && this.mih_heap_arr[index] < this.mih_heap_arr[pae_pos]) {
            mih_swap(index, pae_pos);


            index = pae_pos;
            pae_pos = (index - 1) / 2;
        }
    }

    public int remove_min() {
        if (this.mih_size <= 0) {
            System.out.println("Heap empty!");
            return -1;
        }

        int min = this.mih_heap_arr[0];
        this.mih_heap_arr[0] = this.mih_heap_arr[this.mih_size - 1];
        this.mih_heap_arr[this.mih_size - 1] = null;
        this.mih_size--;

        mih_heapify(0);
        return min;
    }

    private void mih_heapify(int index) {
        int smallest = index;
        int left = (index * 2) + 1;


        int right = (index * 2) + 2;

        if (left < this.mih_size && this.mih_heap_arr[left] < this.mih_heap_arr[smallest]) {
            smallest = left;
        }
        if (right < this.mih_size && this.mih_heap_arr[right] < this.mih_heap_arr[smallest]) {
            smallest = right;
        }

        if (smallest != index) {
            mih_swap(index, smallest);
            mih_heapify(smallest);
        }
    }

    public void mih_swap(int i, int j) {


        int temp = this.mih_heap_arr[i];
        this.mih_heap_arr[i] = this.mih_heap_arr[j];
        this.mih_heap_arr[j] = temp;
    }

    public int[] mih_sort() {
        int[] sorted = new int[this.mih_size];
        int sizeCopy = this.mih_size;

        for (int i = 0; i < sizeCopy; i++) {
            sorted[i] = remove_min();
        }
        return sorted;
    }

    public void mih_print() {
        for (int i = 0; i < this.mih_size; i++) {
            System.out.print(this.mih_heap_arr[i] + " ");
        }


        System.out.println();
    }
}

    public static int[] task3(int[] tasks, int m) {

        MinHeap heap = new MinHeap(m);

        for (int i = 0; i < m; i++) {
            heap.mih_insert(0);
        }

        for (int task : tasks) {
            int minLoad = heap.remove_min();
            heap.mih_insert(minLoad + task);
        }

        int[] result = new int[m];

        for (int i = 0; i < m; i++) {
            result[i] = heap.remove_min();
        }

        return result;
    }

        }

        
        int[] result = new int[k];
        for (int i = 0; i < k; i++) {
            result[i] = heap.remove_max();
        }
        return result;
    }
