public class minHeap{
    private int[] heapArr ;
    private int lastIdx = 0 ;
    private int capacity ;
    public minHeap(int capacity){
        this.capacity = capacity ;
        this.heapArr = new int[capacity] ;
        lastIdx = 0 ;
    }
    public void insert(int elem){
        if(lastIdx==capacity){
            return ;
        }
        heapArr[lastIdx] = elem ;
        swim(lastIdx,(lastIdx-1)/2) ;
        lastIdx++ ;
    }
    private void swim(int cIdx,int pIdx){
        if(cIdx==0){
            return ;
        }
        if(heapArr[cIdx]<heapArr[pIdx]){
            int temp = heapArr[pIdx] ;
            heapArr[pIdx] = heapArr[cIdx] ;
            heapArr[cIdx] = temp ;
            swim(pIdx,(pIdx-1)/2) ;
        }
    }


    public int extractMin(){
        if(lastIdx==0){
            return -1 ;
        }
        int min = heapArr[0] ;
        heapArr[0] = heapArr[lastIdx-1] ;
        lastIdx-- ;
        sink(0,lastIdx) ;
        return min ;
    }
    private void sink(int i , int n){


        int left = 2*i+1 ;
        int right = 2*i+2 ;
        int minIdx = i ;
        if(left<n && heapArr[minIdx]>heapArr[left]){
            minIdx = left ;
        }
        if(right<n && heapArr[minIdx]>heapArr[right]){
            minIdx = right ;
        }
        if(minIdx!=i){
            int temp = heapArr[i] ;
            heapArr[i] = heapArr[minIdx] ;
            heapArr[minIdx] = temp ;
            sink(minIdx,n) ;
        }
    }


    public static int[] distTask(int tasks[] , int m){
        minHeap minH = new minHeap(m) ;
        for(int i=0 ; i<m ; i++){
            minH.insert(0) ;
        }
        for(int i=0 ; i<tasks.length ; i++){
            int elem = minH.extractMin() ;
            minH.insert(elem+tasks[i]) ;
        }
        int result[] = new int[m] ;
        for(int i=0 ; i<m ; i++){
            result[i] = minH.extractMin() ;
        }
        return result ;
    }
}

public class maxHeap{
    private int[] heapArr ;
    private int lastIdx = 0 ;
    private int capacity ;
    public maxHeap(int capacity){
        this.capacity = capacity ;
        this.heapArr = new int[capacity] ;
        lastIdx = 0 ;
    }
    public void insert(int elem){
        if(lastIdx==capacity){
            return ;
        }
        heapArr[lastIdx] = elem ;
        swim(lastIdx,(lastIdx-1)/2) ;
        lastIdx++ ;
    }
    private void swim(int cIdx,int pIdx){
        if(cIdx==0){
            return ;
        }
        if(heapArr[cIdx]>heapArr[pIdx]){
            int temp = heapArr[pIdx] ;
            heapArr[pIdx] = heapArr[cIdx] ;
            heapArr[cIdx] = temp ;
            swim(pIdx,(pIdx-1)/2) ;
        }
    }


    public int extractMax(){
        if(lastIdx==0){
            return -1 ;
        }
        int max = heapArr[0] ;
        heapArr[0] = heapArr[lastIdx-1] ;
        lastIdx-- ;
        sink(0,lastIdx) ;
        return max ;
    }
    private void sink(int i , int n){


        int left = 2*i+1 ;
        int right = 2*i+2 ;
        int maxIdx = i ;
        if(left<n && heapArr[maxIdx]<heapArr[left]){
            maxIdx = left ;
        }
        if(right<n && heapArr[maxIdx]<heapArr[right]){
            maxIdx = right ;
        }
        if(maxIdx!=i){
            int temp = heapArr[i] ;
            heapArr[i] = heapArr[maxIdx] ;
            heapArr[maxIdx] = temp ;
            sink(maxIdx,n) ;
        }
    }


    public static int[] largest(int nums[] , int k){
        maxHeap maxH = new maxHeap(nums.length) ;
        for(int i=0 ; i<nums.length ; i++){
            maxH.insert(nums[i]) ;
        }
        int result[] = new int[k] ;
        for(int i=0 ; i<k ; i++){
            result[i] = maxH.extractMax() ;
        }
        return result ;
    }
}
