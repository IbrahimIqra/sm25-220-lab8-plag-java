public class Lab8Heap {
    //------ TASK 3-------------


    static void hMin(int[] hp,int n,int i){
        int smaller = i;
        int left = 2*i+1;
        int right = 2*i+2;
        if(left<n && hp[left]<hp[smaller]) smaller=left;
        if(right<n && hp[right]<hp[smaller]) smaller=right;
        if(smaller!=i){
            
            int temporary = hp[i];
            hp[i]=hp[smaller];
            hp[smaller]=temporary;
            hMin(hp,n,smaller);
        }
    }

    static int[] task(int[] task,int m){
        int[] hp = new int[m]; 
        for(int i=m/2-1;i>=0;i--) {
            hMin(hp,m,i);
        }
        for(int t : task){
            hp[0] = hp[0]+t;      
            hMin(hp,m,0);
        }
        return hp;
    }

  //-------- TASK 4------------------------------
    static void hMax(int[] hp,int n,int i){


        int large = i;
        int left = 2*i+1;
        int right = 2*i+2;
        if(left<n && hp[left]>hp[large]) large=left;
        if(right<n && hp[right]>hp[large]) large=right;
        if(large!=i){
            int temporary = hp[i];
            hp[i]=hp[large];
            hp[large]=temporary;
            hMax(hp,n,large);
        }
    }



    static void buildHp(int[] hp){
        int n = hp.length;
        for(int i= (n/2-1);i>=0;i--) hMax(hp,n,i);
    }

    static int[] largeTopk(int[] numbers,int k){
        int n = numbers.length;


        int [] hp = new int[n];
        for(int i=0;i<n;i++){ 
            hp[i]=numbers[i];
        }
        buildHp(hp);
        int hpSize = n;
        int[] res = new int[k];
        for(int i=0;i<k;i++){
            res[i]=hp[0];
            hp[0]=hp[hpSize-1];
            hpSize--;
            hMax(hp, hpSize,0);
        }


        return res;
    }

    public static void main(String[] args){
        int[] tasks = {2,4,7,1, 6};
        int m = 4;
        int[] machineLoad = task(tasks,m);
        
    
    System.out.print("Task 3-Machine Loads: [");
        for(int i=0;i<machineLoad.length;i++){
            System.out.print(machineLoad[i]);
            if(i!=machineLoad.length-1) {
                System.out.print(", ");
            }
        }
        System.out.println("]");
        
        
        
        
        
        



        int [] nums = {4,10,2, 8,6,7};
        int k=3;
        int[] largeTopk = largeTopk(nums,k);
        
        
        System.out.print("Task 4-Top " + k + " Largest:[");
        for(int i=0;i<largeTopk.length;i++){
            System.out.print(largeTopk[i]);
            if(i!= largeTopk.length-1){
                System.out.print(", ");
            } 
        }
        System.out.println("]");
    }
}
