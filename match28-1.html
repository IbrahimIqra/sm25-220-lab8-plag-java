public class Task3{
   private static int findmini(int[]t){
        int minimum=0;
      for(int i=1;i<t.length;i++){
            if(t[i]<t[minimum]){
                minimum=i;
            }
        }
        return minimum;
    }
   private static void mini_swim(int[]p,int indx){
        int prnt=indx-1/2;
        while(0<indx && p[indx]<p[prnt]){
          swap(p,indx,prnt);
          indx=prnt;
          prnt=indx-1/2;
        }
    }
   private static void swap(int[]x,int y,int z){
        int temporary=x[y];
        x[y]=x[z];
        x[z]=temporary;
    }

  
  public static int[] method1(int[]tasks,int s){ 
  int[]v= new int[s];
   for(int i=0;i<s;i++){
     v[i]=0;
        }
   for(int i=0;i<tasks.length;i++){
            int u=tasks[i];
            int mini_indx=findmini(v);
            v[mini_indx]+=u;
            mini_swim(v,mini_indx);
        }
           return v;
    }
  }

public class Task4{
  private static void swap(int[]x,int y,int z){


        int temporary=x[y];
        x[y]=x[z];
        x[z]=temporary;
    }

  private static void maxi_swim(int[]s,int indx){
        int prnt=(indx-1)/2;
        while(0<indx && s[indx]>s[prnt]){
          swap(s,indx,prnt);
          indx=prnt;
          prnt=(indx-1)/2;
        }
    }
  private static int find_maxi(int[]u,int v){
    if(v==0){
      return -1;
        }
    int maxi=u[0];
    u[0]=u[v-1]; 
    u[v-1]=0;
    max_he(u,v-1,0);
    return maxi;
    }
 private static void max_he(int[]p,int n,int indx){
        int maxi=indx;
        int left=2*indx+1;
        int right=2*indx+2;
        if(n>left && p[left]>p[maxi]){
            maxi=left;
        }


        if(n>right && p[right]>p[maxi]){
            maxi=right;
        }
        if(maxi!=indx){
            swap(p,indx,maxi);
            max_he(p,n,maxi);
        }
    }

    public static int[] findi(int[]d,int k){
        int o=d.length;
        int[]p=new int[o];
        for(int i=0;i<o;i++){
            p[i]=d[i];
            maxi_swim(p,i); 
        }
        int[]r=new int[k];
        for (int i=0;i<k;i++){
            r[i]=find_maxi(p,o-i);
        }
            return r;
    }
 }
